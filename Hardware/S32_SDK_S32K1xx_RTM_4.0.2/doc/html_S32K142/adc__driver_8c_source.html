86 /*FUNCTION**********************************************************************
87  *
88  * Function Name : ADC_DRV_ConfigConverter
89  * Description : This function configures the ADC converter with the options
90  * provided in the configuration structure.
91  *
92  * Implements : ADC_DRV_ConfigConverter_Activity
93  *END**************************************************************************/
94 void ADC_DRV_ConfigConverter(const uint32_t instance,
95  const adc_converter_config_t * const config)
96 {
97  DEV_ASSERT(instance < ADC_INSTANCE_COUNT);
98  DEV_ASSERT(config != NULL);
99  /* Some alternative clocks can be unavailable depending on the device */
100  DEV_ASSERT(config->inputClock <= NUMBER_OF_ALT_CLOCKS);
101 
102  ADC_Type * const base = s_adcBase[instance];
103  clock_names_t adc_clocks[ADC_INSTANCE_COUNT] = ADC_CLOCKS;
104  uint32_t adc_freq = 0u;
105  status_t clk_status = CLOCK_SYS_GetFreq(adc_clocks[instance], &adc_freq);
106  DEV_ASSERT(clk_status == STATUS_SUCCESS);
107  (void) clk_status;
108 
109  adc_freq = adc_freq / (uint32_t)(1UL << ((uint32_t)(config->clockDivide)));
110  DEV_ASSERT((adc_freq >= ADC_CLOCK_FREQ_MIN_RUNTIME) && (adc_freq <= ADC_CLOCK_FREQ_MAX_RUNTIME));
111 
112  ADC_SetClockDivide(base, config->clockDivide);
113  ADC_SetSampleTime(base, config->sampleTime);
114  ADC_SetResolution(base, config->resolution);
115  ADC_SetInputClock(base, config->inputClock);
116  ADC_SetTriggerMode(base, config->trigger);
117  ADC_SetPretriggerSelect(instance, config->pretriggerSel);
118  ADC_SetTriggerSelect(instance, config->triggerSel);
119  ADC_SetDMAEnableFlag(base, config->dmaEnable);
120  ADC_SetVoltageReference(base, config->voltageRef);
121  ADC_SetContinuousConvFlag(base, config->continuousConvEnable);
122 
123  /* Supply monitoring is only available for ADC 0. */
124  DEV_ASSERT((config->supplyMonitoringEnable == false) || (instance == 0u));
125  if(instance == 0u)
126  {
127  SIM_Type * const simBase = SIM;
128  ADC_SetSupplyMonitoringEnableFlag(simBase, config->supplyMonitoringEnable);
129  }
130 }

170 /*FUNCTION**********************************************************************
171  *
172  * Function Name : ADC_DRV_Reset
173  * Description : This function writes all the internal ADC registers with
174  * their Reference Manual reset values.
175  *
176  * Implements : ADC_DRV_Reset_Activity
177  *END**************************************************************************/
178 void ADC_DRV_Reset(const uint32_t instance)
179 {
180  DEV_ASSERT(instance < ADC_INSTANCE_COUNT);
181 
182  ADC_Type * const baseAddr = s_adcBase[instance];
183  uint8_t idx = 0U;
184 
185  for(idx = 0U; idx < ADC_SC1_COUNT; idx++)
186  {
187  baseAddr->SC1[idx] = ADC_SC1_ADCH(ADC_INPUTCHAN_DISABLED) | ADC_SC1_AIEN(0x00U);
188  }
189 
190  baseAddr->CFG1 = ADC_CFG1_ADICLK(ADC_CLK_ALT_1) | ADC_CFG1_MODE(ADC_RESOLUTION_8BIT) | ADC_CFG1_ADIV(ADC_CLK_DIVIDE_1);
191  baseAddr->CFG2 = ADC_CFG2_SMPLTS(ADC_DEFAULT_SAMPLE_TIME);
192 
193  for(idx = 0U; idx < ADC_CV_COUNT; idx++)
194  {
195  baseAddr->CV[idx] = ADC_CV_CV(0U);
196  }
197 
198  baseAddr->SC2 = ADC_SC2_REFSEL(ADC_VOLTAGEREF_VREF) | ADC_SC2_DMAEN(0x00U) | ADC_SC2_ACREN(0x00U) | ADC_SC2_ACFGT(0x00U) | ADC_SC2_ACFE(0x00U) |
199  ADC_SC2_ADTRG(0x00U);
200  baseAddr->SC3 = ADC_SC3_AVGS(ADC_AVERAGE_4) | ADC_SC3_AVGE(0x00U) | ADC_SC3_ADCO(0x00U) | ADC_SC3_CAL(0x00U);
201  baseAddr->USR_OFS = ADC_USR_OFS_USR_OFS(0U);
202  baseAddr->UG = ADC_UG_UG(ADC_DEFAULT_USER_GAIN);
203 
204 #if FEATURE_ADC_HAS_EXTRA_NUM_REGS
205  for(idx = 0U; idx < ADC_aSC1_COUNT; idx++)
206  {
207  baseAddr->aSC1[idx] = ADC_aSC1_ADCH(ADC_INPUTCHAN_DISABLED) | ADC_aSC1_AIEN(0x00U);
208  }
209 #endif /* FEATURE_ADC_HAS_EXTRA_NUM_REGS */
210 
211  ADC_SetPretriggerSelect(instance, ADC_PRETRIGGER_SEL_PDB);
212  ADC_SetTriggerSelect(instance, ADC_TRIGGER_SEL_PDB);
213  ADC_DRV_SetSwPretrigger(instance, ADC_SW_PRETRIGGER_DISABLED);
214 
215  /* Reset ADC Supply Monitoring - available only for ADC 0 */
216  if(instance == 0u)
217  {
218  SIM_Type * const simBase = SIM;
219  ADC_SetSupplyMonitoringEnableFlag(simBase, false);
220 
221  simBase->CHIPCTL &= ~SIM_CHIPCTL_ADC_SUPPLY_MASK;
222  }
223 }

419 /*FUNCTION**********************************************************************
420  *
421  * Function Name : ADC_DRV_SetSwPretrigger
422  * Description : This function sets the software pretrigger - affects only first 4 control channels.
423  *
424  * Implements : ADC_DRV_SetSwPretrigger_Activity
425  *END**************************************************************************/
426 void ADC_DRV_SetSwPretrigger(const uint32_t instance,
427  const adc_sw_pretrigger_t swPretrigger)
428 {
429  DEV_ASSERT(instance < ADC_INSTANCE_COUNT);
430  SIM_Type * const simBase = SIM;
431  uint32_t intermValue = 0U;
432  uint32_t mask[ADC_INSTANCE_COUNT] = {0U};
433 #if (ADC_INSTANCE_COUNT == 1U)
434  mask[0] = SIM_ADCOPT_ADC0SWPRETRG_MASK;
435 #elif (ADC_INSTANCE_COUNT == 2U)
436  mask[0] = SIM_ADCOPT_ADC0SWPRETRG_MASK;
437  mask[1] = SIM_ADCOPT_ADC1SWPRETRG_MASK;
438 #else
439 #error "Maximum supported value for ADC_INSTANCE_COUNT is 2."
440 #endif
441  /* If SW Pretrigger Select is not enabled, the SW pretriggers will be ignored by ADC. */
442  DEV_ASSERT((ADC_GetPretriggerSelect(instance) == ADC_PRETRIGGER_SEL_SW) || \
443  (swPretrigger == ADC_SW_PRETRIGGER_DISABLED));
444 
445  intermValue = simBase->ADCOPT & (~ mask[instance]);
446  switch(instance)
447  {
448  case 0:
449  intermValue |= SIM_ADCOPT_ADC0SWPRETRG(swPretrigger);
450  break;
451  case 1:
452  intermValue |= SIM_ADCOPT_ADC1SWPRETRG(swPretrigger);
453  break;
454  default:
455  DEV_ASSERT(false);
456  break;
457  }
458 
459  simBase->ADCOPT = intermValue;
460 }

543 /*FUNCTION**********************************************************************
544  *
545  * Function Name : ADC_DRV_AutoCalibration
546  * Description : This functions executes an Auto-Calibration sequence. It
547  * is recommended to run this sequence before using the ADC converter.
548  * this function will check and reset clock divide based the adc frequency.
549  * an error will be displayed if adc_freq too big
550  * this function will set satisfy clock divide,start calibration.
551  * final this function: restore adc clock divide,hardware average and trigger settings.
552  *
553  * Implements : ADC_DRV_AutoCalibration_Activity
554  *END**************************************************************************/
555 void ADC_DRV_AutoCalibration(const uint32_t instance)
556 {
557  DEV_ASSERT(instance < ADC_INSTANCE_COUNT);
558 
559  ADC_Type * const base = s_adcBase[instance];
560  /* set hardware average to maximum and set software trigger*/
561  bool hwavgen = ADC_GetHwAverageEnableFlag(base);
562  adc_average_t hwavg = ADC_GetHwAverageMode(base);
563  adc_trigger_t trig = ADC_GetTriggerMode(base);
564  uint8_t sampletime = ADC_GetSampleTime(base);
565  ADC_SetHwAverageMode(base, ADC_AVERAGE_32);
566  ADC_SetHwAverageEnableFlag(base, true);
567  ADC_SetTriggerMode(base, ADC_TRIGGER_SOFTWARE);
568  /* Set the sample time to the reset value because it affects the
569  calibration duration but not the results
570  */
571  ADC_SetSampleTime(base, ADC_RESET_SAMPLE_TIME_VALUE);
572 
573  base->CLPS = 0x00u;
574  base->CLP3 = 0x00u;
575  base->CLP2 = 0x00u;
576  base->CLP1 = 0x00u;
577  base->CLP0 = 0x00u;
578  base->CLPX = 0x00u;
579  base->CLP9 = 0x00u;
580 
581  /*Set clock divider */
582  clock_names_t adc_clocks[ADC_INSTANCE_COUNT] = ADC_CLOCKS;
583  uint32_t adc_freq = 0u;
584  adc_clk_divide_t adc_clk_divide_res = ADC_GetClockDivide(base);
585  adc_clk_divide_t adc_clk_divide = ADC_CLK_DIVIDE_1;
586  status_t clk_status = CLOCK_SYS_GetFreq(adc_clocks[instance], &adc_freq);
587  DEV_ASSERT(clk_status == STATUS_SUCCESS);
588  (void) clk_status;
589  DEV_ASSERT(adc_freq >= ADC_CLOCK_FREQ_MIN_RUNTIME);
590  if ((adc_freq / (uint32_t)(1UL << ((uint32_t)(adc_clk_divide_res)))) <= (ADC_CLOCK_FREQ_MAX_RUNTIME / 2U))
591  {
592  /* no action if adc_freq is satisfy */
593  }
594  else
595  {
596  if ((adc_freq / 2U) <= (ADC_CLOCK_FREQ_MAX_RUNTIME / 2U))
597  {
598  adc_clk_divide = ADC_CLK_DIVIDE_2;
599  }
600  else if ((adc_freq / 4U) <= (ADC_CLOCK_FREQ_MAX_RUNTIME / 2U))
601  {
602  adc_clk_divide = ADC_CLK_DIVIDE_4;
603  }
604  else if ((adc_freq / 8U) <= (ADC_CLOCK_FREQ_MAX_RUNTIME / 2U))
605  {
606  adc_clk_divide = ADC_CLK_DIVIDE_8;
607  }
608  else
