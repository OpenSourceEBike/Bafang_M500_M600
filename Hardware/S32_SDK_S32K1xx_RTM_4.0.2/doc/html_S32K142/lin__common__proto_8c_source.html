67 /*FUNCTION**********************************************************************
68  *
69  * Function Name : lin_pid_resp_callback_handler
70  * Description : Callback handler for LIN low level events.
71  *
72  * Implements : lin_pid_resp_callback_handler_Activity
73  *END**************************************************************************/
74 void lin_pid_resp_callback_handler(l_ifc_handle iii,
75  const lin_lld_event_id_t event_id,
76  l_u8 id)
77 {
78  switch (event_id)
79  {
80  case LIN_LLD_PID_OK:
81  lin_process_id(iii, id);
82  break;
83  case LIN_LLD_TX_COMPLETED:
84  lin_update_tx(iii, id);
85  break;
86  case LIN_LLD_RX_COMPLETED:
87  lin_update_rx(iii, id);
88  break;
89  case LIN_LLD_BUS_ACTIVITY_TIMEOUT:
90  lin_bus_activity_timeout(iii);
91  break;
92  case LIN_LLD_PID_ERR:
93  case LIN_LLD_FRAME_ERR:
94  case LIN_LLD_CHECKSUM_ERR:
95  case LIN_LLD_READBACK_ERR:
96  case LIN_LLD_NODATA_TIMEOUT:
97  lin_handle_error(iii, event_id, id);
98  break;
99  default:
100  /* do nothing */
101  break;
102  }
103 }

105 /*FUNCTION**********************************************************************
106  *
107  * Function Name : lin_process_id
108  * Description : Handle the received ID and decide the next action of Stack.
109  *
110  * Implements : lin_process_id_Activity
111  *END**************************************************************************/
112 static void lin_process_id(l_ifc_handle iii,
113  l_u8 id)
114 {
115  l_u8 action = 0U;
116  const lin_frame_t * lin_frame_ptr;
117  l_u8 frame_index;
118  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
119  lin_protocol_state_t * prot_state_ptr = &g_lin_protocol_state_array[iii];
120 
121  frame_index = lin_get_frame_index(iii, id);
122  lin_frame_ptr = &(prot_user_config_ptr->frame_tbl_ptr[frame_index]);
123  if (0xFFU != frame_index)
124  {
125  action = 1U;
126  prot_state_ptr->last_pid = lin_process_parity(id,MAKE_PARITY);
127  /* PID belongs to this node, then check type of frame */
128  switch (lin_frame_ptr->frm_type)
129  {
130  /* Unconditional frame */
131  case LIN_FRM_UNCD:
132  if (LIN_RES_PUB == lin_frame_ptr->frm_response)
133  {
134  lin_make_uncd_frame(iii, id);
135  /* Set response */
136  action = 2U;
137  }
138  break;
139  #if ((1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_PROTOCOL_20) || (1U == SUPPORT_PROTOCOL_13))
140  /* Event trigger frame */
141  case LIN_FRM_EVNT:
142  #if (1U == SUPPORT_SLAVE_MODE)
143  if ((bool)LIN_SLAVE == prot_user_config_ptr->function)
144  {
145  action = lin_process_evnt_frame(iii,id);
146  }
147  #endif /* (1U == SUPPORT_SLAVE_MODE) */
148  break;
149  #endif /* ((1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_PROTOCOL_20) || (1U == SUPPORT_PROTOCOL_13)) */
150 
151  /* Diagnostic frame */
152  case LIN_FRM_DIAG:
153  if(LIN_PROTOCOL_13 != prot_user_config_ptr->protocol_version)
154  {
155  action = lin_process_diag_frame(iii,id);
156  }
157  else
158  {
159  action = 0U;
160  }
161  break;
162  default:
163  /* do nothing */
164  break;
165  }
166  }
167 
168  switch (action)
169  {
170  case 1U:
171  /* Receive response */
172  (void)lin_lld_rx_response(iii, lin_frame_ptr->frm_len);
173  break;
174  case 2U:
175  /* Set response */
176  (void)lin_lld_set_response(iii, lin_frame_ptr->frm_len);
177  break;
178  default:
179  /* ignore response */
180  (void)lin_lld_ignore_response(iii);
181  break;
182  }
183 }

185 /*FUNCTION**********************************************************************
186  *
187  * Function Name : lin_update_rx
188  * Description : This function update signal, status, and flag after received response.
189  *
190  * Implements : lin_update_rx_Activity
191  *END**************************************************************************/
192 static void lin_update_rx(l_ifc_handle iii,
193  l_u8 id)
194 {
195  l_u8 frame_index, flag_size, i, associate_id;
196  l_u16 flag_offset;
197  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
198  lin_protocol_state_t * prot_state_ptr = &g_lin_protocol_state_array[iii];
199 #if (1U == SUPPORT_MASTER_MODE)
200  lin_master_data_t * master_data_ptr;
201 #endif /* (1U == SUPPORT_MASTER_MODE) */
202 
203  /* Set successful transfer */
204  prot_state_ptr->successful_transfer = 1U;
205  /* if two or more frames are processed since the previous call to l_ifc_read_status, set overrun flag to 1 */
206  if (prot_state_ptr->num_of_processed_frame < 0xFFU)
207  {
208  prot_state_ptr->num_of_processed_frame++;
209  }
210 
211  if ((prot_state_ptr->num_of_processed_frame > 1U) && (prot_state_ptr->overrun_flg == 0U))
212  {
213  prot_state_ptr->overrun_flg = 1U;
214  }
215 
216  frame_index = lin_get_frame_index(iii, id);
217  if (frame_index < LIN_NUM_OF_FRMS)
218  {
219  /* This PID doesn't belong to this node */
220  /* PID belongs to this node, then check type of frame */
221  switch (prot_user_config_ptr->frame_tbl_ptr[frame_index].frm_type)
222  {
223  /* Unconditional frame */
224  case LIN_FRM_UNCD:
225  lin_update_uncd_frame(iii, id);
226  break;
227  /* Event trigger frame */
228  case LIN_FRM_EVNT:
229  lin_update_rx_evnt_frame(iii, id);
230  /* Recalculate frame_index by updating associate PID */
231  associate_id = lin_process_parity(prot_state_ptr->response_buffer_ptr[0], CHECK_PARITY);
232  frame_index = lin_get_frame_index(iii, associate_id);
233  break;
234  /* Diagnostic frame */
235  case LIN_FRM_DIAG:
236  /* If goto sleep signal */
237  if (0U == prot_state_ptr->response_buffer_ptr[0])
238  {
239  /* In a slave node, set go to sleep flag in state structure */
240  prot_state_ptr->go_to_sleep_flg = true;
241  (void)lin_lld_set_low_power_mode(iii);
242  }
243  else
244  {
245  /* Call transport layer callback to process TL_RX_COMPLETED event*/
246  (void)LIN_TL_CALLBACK_HANDLER(iii, TL_RX_COMPLETED, id);
247  }
248 
249  break;
250  default:
251  /* do nothing */
252  break;
253  }
254 
255  if (frame_index < LIN_NUM_OF_FRMS)
256  {
257  /* Update rx frame flag */
258  g_lin_frame_flag_handle_tbl[frame_index] = true;
259 
260  if ((bool)LIN_SLAVE == prot_user_config_ptr->function)
261  {
262  /* Update rx signal flags */
263  flag_offset = prot_user_config_ptr->frame_tbl_ptr[frame_index].flag_offset;
264  flag_size = prot_user_config_ptr->frame_tbl_ptr[frame_index].flag_size;
265  for (i = 0U; i < flag_size; i++)
266  {
267  g_lin_flag_handle_tbl[flag_offset] = 0xFFU;
268  flag_offset++;
269  }
270  }
271  #if (1U == SUPPORT_MASTER_MODE)
272  else /* If master node */
273  {
274  master_data_ptr = &g_lin_master_data_array[g_lin_protocol_user_cfg_array[iii].master_ifc_handle];
275  master_data_ptr->flag_offset = prot_user_config_ptr->frame_tbl_ptr[frame_index].flag_offset;
276  master_data_ptr->flag_size = prot_user_config_ptr->frame_tbl_ptr[frame_index].flag_size;
277  }
278  #endif /* (1U == SUPPORT_MASTER_MODE) */
279  }
280 
281  #if (1U == SUPPORT_SLAVE_MODE)
282  /* update word status */
283  if ((bool)LIN_SLAVE == prot_user_config_ptr->function)
284  {
285  if ((prot_user_config_ptr->protocol_version == LIN_PROTOCOL_21) || (prot_user_config_ptr->protocol_version == LIN_PROTOCOL_20))
286  {
287  lin_update_word_status_lin21(iii, LIN_LLD_RX_COMPLETED);
288  }
289  else
290  {
291  if (prot_user_config_ptr->protocol_version == LIN_PROTOCOL_J2602)
292  {
293  lin_update_word_status_j2602(iii);
294  }
295  }
296  }
297  #endif /* (1U == SUPPORT_SLAVE_MODE) */
298  }
299 }

301 /*FUNCTION**********************************************************************
302  *
303  * Function Name : lin_update_tx
304  * Description : This function update status and flags after transmit response
305  *
306  * Implements : lin_update_tx_Activity
307  *END**************************************************************************/
308 static void lin_update_tx(l_ifc_handle iii,
309  l_u8 id)
310 {
311  l_u8 frame_index;
312  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
313  lin_protocol_state_t * prot_state_ptr = &g_lin_protocol_state_array[iii];
314 #if (1U == SUPPORT_MASTER_MODE)
315  const lin_master_data_t * master_data_ptr;
316 #endif /* (1U == SUPPORT_MASTER_MODE) */
317 #if (1U == SUPPORT_SLAVE_MODE)
318  const lin_node_attribute_t * node_attr_ptr;
319  l_u8 i;
320 #if (1U == SUPPORT_PROTOCOL_J2602)
321  l_u8* bit_offset_temp_ptr;
322  l_u16* byte_offset_temp_ptr;
323 #endif /* ((1U == SUPPORT_PROTOCOL_J2602) */
324 #endif /* (1U == SUPPORT_SLAVE_MODE) */
325 
326  /* Set successful transfer */
327  prot_state_ptr->successful_transfer = 1U;
328  /* if two or more frames are processed since the previous call to l_ifc_read_status, set overrun flag to 1 */
329  if (prot_state_ptr->num_of_processed_frame < 0xFFU)
330  {
331  prot_state_ptr->num_of_processed_frame++;
332  }
333 
334  if ((prot_state_ptr->num_of_processed_frame > 1U) && (prot_state_ptr->overrun_flg == 0U))
335  {
336  prot_state_ptr->overrun_flg = 1U;
337  }
338 
339 #if (1U == SUPPORT_MASTER_MODE)
340  /* If the Master node sent go to sleep command successfully, then set go to sleep flag*/
341  if (((bool)LIN_MASTER == prot_user_config_ptr->function) && (0x3CU == id))
342  {
343  master_data_ptr = &g_lin_master_data_array[g_lin_protocol_user_cfg_array[iii].master_ifc_handle];
344  /* At this point, the schedule table is LIN_SCH_TBL_NULL */
345  if (prot_user_config_ptr->schedule_tbl[master_data_ptr->active_schedule_id].sch_tbl_type == LIN_SCH_TBL_NULL)
346  {
347  /* Set go to sleep flag in state structure */
348  prot_state_ptr->go_to_sleep_flg = true;
349  (void)lin_lld_set_low_power_mode(iii);
350  }
351  }
352 #endif /* (1U == SUPPORT_MASTER_MODE) */
353  /* Find frame index by pid */
354  frame_index = lin_get_frame_index(iii, id);
355  if ((((bool)LIN_MASTER == prot_user_config_ptr->function) && (0x3CU == id) && (DIAG_INTERLEAVE_MODE == prot_state_ptr->diagnostic_mode)) ||
356  (((bool)LIN_SLAVE == prot_user_config_ptr->function) && (0x3DU == id)))
357  {
358  /* Call transport layer callback to process TL_TX_COMPLETED event*/
359  (void)LIN_TL_CALLBACK_HANDLER(iii, TL_TX_COMPLETED, id);
360  }
361  if (frame_index < LIN_NUM_OF_FRMS)
362  {
363  /* Update signal flag */
364  lin_update_tx_flags(iii, frame_index);
365  /* Update tx frame flag */
366  g_lin_frame_flag_handle_tbl[frame_index] = true;
367  }
368 
369 #if (1U == SUPPORT_SLAVE_MODE)
370  /* Update word status */
371  if ((bool)LIN_SLAVE == prot_user_config_ptr->function)
372  {
373  node_attr_ptr = &g_lin_node_attribute_array[prot_user_config_ptr->slave_ifc_handle];
374  #if ((1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_PROTOCOL_20))
375  if ((prot_user_config_ptr->protocol_version == LIN_PROTOCOL_21) || (prot_user_config_ptr->protocol_version == LIN_PROTOCOL_20))
376  {
377  for (i = 0U; i < node_attr_ptr->num_frame_have_esignal; i++)
378  {
379  /* Check if frame contain error signal */
380  if (id == node_attr_ptr->resp_err_frm_id_ptr[i])
381  {
382  /* Clear error signal in frame data buffer */
383  prot_state_ptr->error_in_response = 0U;
384  lin_update_err_signal(iii, frame_index);
385  break;
386  }
387  }
388  lin_update_word_status_lin21(iii, LIN_LLD_TX_COMPLETED);
389  }
390  #endif /* ((1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_PROTOCOL_20)) */
391  #if (1U == SUPPORT_PROTOCOL_J2602)
392  if (prot_user_config_ptr->protocol_version == LIN_PROTOCOL_J2602)
393  {
394  /* This status byte which contains error signal shall be transmitted as the first byte of every slave transmission */
395  if (0x3DU != id)
396  {
397  /* Clear error signal in frame data buffer */
398  prot_state_ptr->error_in_response = 0U;
399  for (i = 0; i < node_attr_ptr->num_frame_have_esignal; i++)
400  {
401  byte_offset_temp_ptr = node_attr_ptr->response_error_byte_offset_ptr + i;
402  bit_offset_temp_ptr = node_attr_ptr->response_error_bit_offset_ptr + i;
403  /* Clear error signal in frame data buffer */
404  g_lin_frame_data_buffer[*byte_offset_temp_ptr] = (l_u8)(g_lin_frame_data_buffer[*byte_offset_temp_ptr] & (~(0x07U << (*bit_offset_temp_ptr))));
405  }
406  }
407  lin_update_word_status_j2602(iii);
408  }
409  #endif /* (1U == SUPPORT_PROTOCOL_J2602) */
410  }
411 #endif /* (1U == SUPPORT_SLAVE_MODE) */
412 }

414 /*FUNCTION**********************************************************************
415  *
416  * Function Name : lin_handle_error
417  * Description : This function handle error raised by low-level
418  *
419  * Implements : lin_handle_error_Activity
420  *END**************************************************************************/
421 static void lin_handle_error(l_ifc_handle iii,
422  lin_lld_event_id_t event_id,
423  l_u8 id)
424 {
425 #if ((1U == SUPPORT_PROTOCOL_20) || (1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_SLAVE_MODE))
426  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
427 #endif /* ((1U == SUPPORT_PROTOCOL_20) || (1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_SLAVE_MODE)) */
428  lin_protocol_state_t * prot_state_ptr = &g_lin_protocol_state_array[iii];
429 
430 #if ((1U == SUPPORT_MASTER_MODE) && ((1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_PROTOCOL_20)))
431  lin_master_data_t * master_data_ptr;
432 #endif /* ((1U == SUPPORT_MASTER_MODE) && ((1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_PROTOCOL_20))) */
433 
434 #if ((1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_PROTOCOL_20))
435  l_u8 frame_index;
436  frame_index = lin_get_frame_index(iii, id);
437 #endif /* ((1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_PROTOCOL_20)) */
438 
439  /* if two or more frames are processed since the previous call to l_ifc_read_status, set overrun flag to 1 */
440  if (prot_state_ptr->num_of_processed_frame < 0xFFU)
441  {
442  prot_state_ptr->num_of_processed_frame++;
443  }
444 
445  if ((prot_state_ptr->num_of_processed_frame > 1U) && (prot_state_ptr->overrun_flg == 0U))
446  {
447  prot_state_ptr->overrun_flg = 1U;
448  }
449 
450  switch (event_id)
451  {
452  /* PID error */
453  case LIN_LLD_PID_ERR:
454  /* do nothing here */
455  break;
456  /* Frame error */
457  case LIN_LLD_FRAME_ERR:
458  case LIN_LLD_CHECKSUM_ERR:
459  case LIN_LLD_READBACK_ERR:
460  case LIN_LLD_NODATA_TIMEOUT:
461  #if ((1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_PROTOCOL_20))
462  /* Check if frame type is event trigger */
463  /* Find the corresponding frame */
464 
465  if (LIN_FRM_EVNT == prot_user_config_ptr->frame_tbl_ptr[frame_index].frm_type)
466  {
467  #if (1U == SUPPORT_MASTER_MODE)
468  if (prot_user_config_ptr->function == (bool)LIN_MASTER)
469  {
470  master_data_ptr = &g_lin_master_data_array[g_lin_protocol_user_cfg_array[iii].master_ifc_handle];
471  master_data_ptr->event_trigger_collision_flg = true;
472  }
473  #endif /* (1U == SUPPORT_MASTER_MODE) */
474  }
475  else
476  #endif /* ((1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_PROTOCOL_20)) */
477  {
478  /* Set response error */
479  prot_state_ptr->error_in_response = 1U;
480  /* TL support */
481  if ((0x3CU == id) || (0x3DU == id))
482  {
483  /* Call transport layer callback to handle TL_ERROR */
484  (void)LIN_TL_CALLBACK_HANDLER(iii, TL_ERROR, id);
485  }
486  }
487  break;
488  default:
489  /* do nothing */
490  break;
491  }
492 
493 #if (1U == SUPPORT_SLAVE_MODE)
494  /* Update word status */
495  if ((bool)LIN_SLAVE == prot_user_config_ptr->function)
496  {
497  #if ((1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_PROTOCOL_20))
498  if ((prot_user_config_ptr->protocol_version == LIN_PROTOCOL_21) || (prot_user_config_ptr->protocol_version == LIN_PROTOCOL_20))
499  {
500  if (prot_state_ptr->error_in_response == 1U)
501  {
502  lin_update_err_signal(iii, frame_index);
503  }
504  lin_update_word_status_lin21(iii, event_id);
505  }
506  #endif /* ((1U == SUPPORT_PROTOCOL_21) || (1U == SUPPORT_PROTOCOL_20)) */
507  #if (1U == SUPPORT_PROTOCOL_J2602)
508  if (prot_user_config_ptr->protocol_version == LIN_PROTOCOL_J2602)
509  {
510  lin_update_status_byte(iii, event_id);
511  lin_update_word_status_j2602(iii);
512  }
513  #endif /* (1U == SUPPORT_PROTOCOL_J2602) */
514  }
515 #endif /* (1U == SUPPORT_SLAVE_MODE) */
516 }

555 #if (1U == SUPPORT_MASTER_MODE)
556 /*FUNCTION**********************************************************************

564 void lin_switch_sch_table(l_ifc_handle iii)
