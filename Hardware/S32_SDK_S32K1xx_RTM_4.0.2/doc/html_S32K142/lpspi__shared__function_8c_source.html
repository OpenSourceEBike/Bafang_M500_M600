123 void LPSPI_DRV_FillupTxBuffer(uint32_t instance)
124 {
125  /* Instantiate local variable of type dspi_master_state_t and point to global state. */
126  lpspi_state_t * lpspiState = g_lpspiStatePtr[instance];
127  LPSPI_Type *base = g_lpspiBase[instance];
128  uint32_t wordToSend = 0;
129  uint16_t numOfBytes;
130  uint8_t availableSpace = (uint8_t)(lpspiState->fifoSize - (uint8_t)LPSPI_ReadTxCount(base));
131 
132  /* Fill the TX buffer. */
133  while(availableSpace != 0U)
134  {
135  if (lpspiState->isPcsContinuous == true)
136  {
137  if(lpspiState->txCount == 1U)
138  {
139  /* Disable continuous PCS */
140  LPSPI_ClearContCBit(base);
141  lpspiState->txCount = 0U;
142  break;
143  }
144  }
145  /* Get the number of bytes which can be written in a single 32 bits word. */
146  if ((lpspiState->bytesPerFrame - lpspiState->txFrameCnt) <= (uint16_t)4)
147  {
148  numOfBytes = (uint16_t)(lpspiState->bytesPerFrame - lpspiState->txFrameCnt);
149  }
150  else
151  {
152  numOfBytes = 4U;
153  }
154  wordToSend = 0;
155 
156  if (lpspiState->txBuff != NULL)
157  {
158  switch(lpspiState->bytesPerFrame)
159  {
160  case 1:
161  wordToSend = *((const uint8_t *)(lpspiState->txBuff));
162  lpspiState->txBuff += sizeof(uint8_t);
163  break;
164  case 2:
165  wordToSend = *((const uint16_t *)(lpspiState->txBuff));
166  lpspiState->txBuff += sizeof(uint16_t);
167  break;
168  default:
169  wordToSend = *((const uint32_t *)(lpspiState->txBuff));
170  lpspiState->txBuff += sizeof(uint32_t);
171  break;
172  }
173  lpspiState->txFrameCnt = (uint16_t)((lpspiState->txFrameCnt + numOfBytes) % lpspiState->bytesPerFrame);
174  }
175  LPSPI_WriteData(base, wordToSend);
176  /* Update internal variable used in transmission. */
177  lpspiState->txCount = (uint16_t)(lpspiState->txCount - numOfBytes);
178  /* Verify if all bytes were send. */
179  if (lpspiState->txCount == 0U)
180  {
181  break;
182  }
183  availableSpace = (uint8_t)(availableSpace - 1U);
184  }
185 }
192 void LPSPI_DRV_ReadRXBuffer(uint32_t instance)
193 {
194  lpspi_state_t * lpspiState = g_lpspiStatePtr[instance];
195  const LPSPI_Type *base = g_lpspiBase[instance];
196  uint32_t receivedWord;
197  uint16_t numOfBytes;
198  uint16_t j;
199  uint8_t filledSpace = (uint8_t)LPSPI_ReadRxCount(base);
200  while (filledSpace!= 0U)
201  {
202  receivedWord = LPSPI_ReadData(base);
203  /* Get the number of bytes which can be read from this 32 bites */
204  if ((lpspiState->bytesPerFrame - lpspiState->rxFrameCnt) <= (uint16_t)4)
205  {
206  numOfBytes = (uint16_t)(lpspiState->bytesPerFrame - lpspiState->rxFrameCnt);
207  }
208  else
209  {
210  numOfBytes = 4U;
211  }
212  /* Generate the word which will be write in buffer. */
213  for (j = 0; j < numOfBytes; j++)
214  {
215  *(lpspiState->rxBuff) = (uint8_t)(receivedWord >> (j * 8U));
216  lpspiState->rxBuff++;
217  }
218  lpspiState->rxFrameCnt = (uint16_t)((lpspiState->rxFrameCnt + numOfBytes) % lpspiState->bytesPerFrame);
219 
220  /* Update internal variable used in transmission. */
221  lpspiState->rxCount = (uint16_t)(lpspiState->rxCount - numOfBytes);
222  /* Verify if all bytes were sent. */
223  if (lpspiState->rxCount == 0U)
224  {
225  break;
226  }
227  filledSpace = (uint8_t)(filledSpace - 1U);
228  }
229 }
