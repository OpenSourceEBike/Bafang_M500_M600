50 /*FUNCTION**********************************************************************
51  *
52  * Function Name : EIM_DRV_Init
53  * Description : Initializes EIM module.
54  * This function configures for EIM channels.
55  * The EIM channel configuration structure array and number of configured channels
56  * shall be passed as arguments and enables module.
57  * This function should be called before calling any other EIM driver function
58  * and make sure that EIM channel has disabled before calling this function.
59  *
60  * Implements : EIM_DRV_Init_Activity
61  *END**************************************************************************/
62 void EIM_DRV_Init(uint32_t instance,
63  uint8_t channelCnt,
64  const eim_user_channel_config_t *channelConfigArr)
65 {
66  DEV_ASSERT(instance < EIM_INSTANCE_COUNT);
67  DEV_ASSERT(channelCnt > 0U);
68  DEV_ASSERT(channelCnt <= EIM_EICHDn_COUNT);
69  DEV_ASSERT(channelConfigArr != NULL);
70  EIM_Type * base = s_eimBase[instance];
71  uint8_t index;
72 
73  for (index = 0U; index < channelCnt; index++)
74  {
75  /* Configures for EIM module */
76  EIM_DRV_ConfigChannel(instance, &channelConfigArr[index]);
77  }
78 
79  /* Enables EIM module */
80  EIM_Enable(base);
81 }

83 /*FUNCTION**********************************************************************
84  *
85  * Function Name : EIM_DRV_Deinit
86  * Description : De-initializes EIM module.
87  * This function sets all registers to reset value and disables EIM channel.
88  * In order to use the EIM channel again, EIM_DRV_Init must be called.
89  *
90  * Implements : EIM_DRV_Deinit_Activity
91  *END**************************************************************************/
92 void EIM_DRV_Deinit(uint32_t instance)
93 {
94  DEV_ASSERT(instance < EIM_INSTANCE_COUNT);
95  EIM_Type * base = s_eimBase[instance];
96 
97  /* Disables EIM module */
98  EIM_Disable(base);
99 
100  /* Resets EIM descriptors and disables EIM channels */
101  EIM_InitChannel(base);
102 }

104 /*FUNCTION**********************************************************************
105  *
106  * Function Name : EIM_DRV_ConfigChannel
107  * Description : This function configures check bit mask, data mask and
108  * operation status(enable/disable) for EIM channel.
109  * The EIM channel configuration structure shall be
110  * passed as arguments.
111  * Make sure that EIM module has disabled before calling this function.
112  *
113  * Implements : EIM_DRV_ConfigChannel_Activity
114  *END**************************************************************************/
115 void EIM_DRV_ConfigChannel(uint32_t instance,
116  const eim_user_channel_config_t *userChannelConfig)
117 {
118  DEV_ASSERT(instance < EIM_INSTANCE_COUNT);
119  DEV_ASSERT(userChannelConfig != NULL);
120  DEV_ASSERT(userChannelConfig->channel < EIM_EICHDn_COUNT);
121  EIM_Type * base = s_eimBase[instance];
122 
123  /* Disables EIM channel to configure EIM channel */
124  EIM_EnableChannelCmd(base, userChannelConfig->channel, false);
125  /* Configures check-bit mask for EIM channel */
126  EIM_SetCheckBitMask(base, userChannelConfig->channel, userChannelConfig->checkBitMask);
127  /* Configures data mask for EIM channel */
128  EIM_SetDataMask(base, userChannelConfig->channel, userChannelConfig->dataMask);
129  /* Configures dataMask1 for EIM channel if CPU S32Rx7x defined */
130 #if defined(FEATURE_EIM_DATAMASK1)
131  EIM_SetDataMask1(base, userChannelConfig->channel, userChannelConfig->dataMask1);
132 #endif
133  /* Enables or disables EIM channel operation corresponding to channel configuration */
134  EIM_EnableChannelCmd(base, userChannelConfig->channel, userChannelConfig->enable);
135 }

148 {
149  DEV_ASSERT(instance < EIM_INSTANCE_COUNT);
150  DEV_ASSERT(channel < EIM_EICHDn_COUNT);
151  DEV_ASSERT(channelConfig != NULL);
152  const EIM_Type * base = s_eimBase[instance];
153 
154  /* Gets the channel number */
155  channelConfig->channel = channel;
156  /* Gets check-bit mask of EIM channel */
157  channelConfig->checkBitMask = EIM_GetCheckBitMask(base, channel);
158  /* Gets data mask of EIM channel */
159  channelConfig->dataMask = EIM_GetDataMask(base, channel);
160  /* Gets dataMask1 of EIM channel if CPU of S32Rx7x defined */
161 #if defined(FEATURE_EIM_DATAMASK1)
162  channelConfig->dataMask1 = EIM_GetDataMask1(base, channel);
163 #endif
164  /* Gets operation status */
165  channelConfig->enable = EIM_IsChannelEnabled(base, channel);
166 }
