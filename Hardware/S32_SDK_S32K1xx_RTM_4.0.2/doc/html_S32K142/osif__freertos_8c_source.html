87  *
88  * Function Name : osif_IsIsrContext
89  * Description : This function returns true if the current execution context
90  * is from an ISR and false if execution is in normal thread mode.
91  *
92  *END**************************************************************************/
93 #if FEATURE_OSIF_FREERTOS_ISR_CONTEXT_METHOD == 1
94 /* Cortex M device - read ICSR[IPSR] value */
95 static inline bool osif_IsIsrContext(void)
96 {
97  bool is_isr = false;
98  uint32_t ipsr_code = (uint32_t)( (S32_SCB->ICSR & S32_SCB_ICSR_VECTACTIVE_MASK) >> S32_SCB_ICSR_VECTACTIVE_SHIFT );
99  if (ipsr_code != 0u)
100  {
101  is_isr = true;
102  }
103 
104  return is_isr;
105 }
106 #elif FEATURE_OSIF_FREERTOS_ISR_CONTEXT_METHOD == 2
107 /* PowerPC device, for FreeRTOS 9.0.0 read the SPRG0 reg that denotes the interrupt nesting level */
108 #define SPRG0_ADDR (272)
109 static inline bool osif_IsIsrContext(void)
110 {
111  uint32_t interrupt_level = MFSPR(SPRG0_ADDR);
112  bool is_isr = (bool)(interrupt_level > 0u);
113  return is_isr;
114 }
115 #else
116  #error "No method to check ISR Context"
117 #endif /* FEATURE_OSIF_FREERTOS_ISR_CONTEXT_METHOD */

158 /*FUNCTION**********************************************************************
159  *
160  * Function Name : OSIF_MutexLock
161  * Description : This function obtains the mutex lock or returns error if timeout.
162  *
163  * Implements : OSIF_MutexLock_freertos_Activity
164  *END**************************************************************************/
165 status_t OSIF_MutexLock(const mutex_t * const pMutex,
166  const uint32_t timeout)
167 {
168  /* The (pMutex == NULL) case is a valid option, signaling that the mutex does
169  * not need to be locked - do not use DEV_ASSERT in this case */
170 
171  uint32_t timeoutTicks;
172  status_t osif_ret_code = STATUS_SUCCESS;
173  TaskHandle_t mutex_holder_handle;
174  TaskHandle_t current_task_handle;
175  BaseType_t operation_status = pdFAIL;
176 
177  if (pMutex != NULL)
178  {
179  SemaphoreHandle_t mutex_handle = SEM_HANDLE(*pMutex);
180  /* Two dependencies for FreeRTOS config file */
181  /* INCLUDE_xQueueGetMutexHolder */
182  /* INCLUDE_xTaskGetCurrentTaskHandle */
183  mutex_holder_handle = xSemaphoreGetMutexHolder(mutex_handle);
184  current_task_handle = xTaskGetCurrentTaskHandle();
185 
186  /* If pMutex has been locked by current task, return error. */
187  if (mutex_holder_handle == current_task_handle)
188  {
189  osif_ret_code = STATUS_ERROR;
190  }
191  else
192  {
193  /* Convert timeout from millisecond to tick. */
194  if (timeout == OSIF_WAIT_FOREVER)
195  {
196  timeoutTicks = portMAX_DELAY;
197  }
198  else
199  {
200  timeoutTicks = MSEC_TO_TICK(timeout);
201  }
202 
203  /* Try to take the semaphore */
204  operation_status = xSemaphoreTake(mutex_handle, timeoutTicks);
205 
206  osif_ret_code = (operation_status == pdPASS) ? STATUS_SUCCESS : STATUS_TIMEOUT;
207  }
208  }
209 
210  return osif_ret_code;
211 }

347 /*FUNCTION**********************************************************************
348  *
349  * Function Name : OSIF_SemaPost
350  * Description : This function performs the 'post' (increment) operation on a semaphore.
351  *
352  * Implements : OSIF_SemaPost_freertos_Activity
353  *END**************************************************************************/
354 status_t OSIF_SemaPost(semaphore_t * const pSem)
355 {
356  DEV_ASSERT(pSem);
357 
358  BaseType_t operation_status = pdFAIL;
359  status_t osif_ret_code;
360 
361  SemaphoreHandle_t sem_handle = SEM_HANDLE(*pSem);
362  /* Check if the post operation is executed from ISR context */
363  bool is_isr = osif_IsIsrContext();
364  if (is_isr)
365  {
366  /* Execution from exception handler (ISR) */
367  BaseType_t taskWoken = pdFALSE;
368  operation_status = xSemaphoreGiveFromISR(sem_handle, &taskWoken);
369 
370  if (operation_status == pdPASS)
371  {
372  /* Perform a context switch if necessary */
373  portYIELD_FROM_ISR(taskWoken);
374  }
375  }
376  else
377  {
378  /* Execution from task */
379  operation_status = xSemaphoreGive(sem_handle);
380  }
381 
382  /* pdFAIL in case that the semaphore is full */
383  osif_ret_code = (operation_status == pdPASS) ? STATUS_SUCCESS : STATUS_ERROR;
384 
385  return osif_ret_code;
386 }
