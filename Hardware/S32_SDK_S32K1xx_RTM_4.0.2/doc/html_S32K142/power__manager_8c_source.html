55 /*FUNCTION**********************************************************************
56  *
57  * Function Name : POWER_SYS_Init
58  * Description : Initializes the Power manager for operation.
59  * This function initializes the Power manager and its run-time state structure.
60  * Reference to an array of Power mode configuration structures has to be passed
61  * as parameter along with parameter specifying its size. At least one power mode
62  * configuration is required. Optionally, reference to array of predefined
63  * call-backs can be passed with its size parameter.
64  * For details about call-backs refer to the power_manager_callback_user_config_t.
65  * As Power manager stores only references to array of these structures they have
66  * to exist while Power manager is used.
67  *
68  * Implements POWER_SYS_Init_Activity
69  *END**************************************************************************/
70 status_t POWER_SYS_Init(power_manager_user_config_t * (*powerConfigsPtr)[],
71  uint8_t configsNumber,
72  power_manager_callback_user_config_t * (*callbacksPtr)[],
73  uint8_t callbacksNumber)
74 {
75  DEV_ASSERT(powerConfigsPtr != NULL); /* Reference to the power configurations is valid. */
76  DEV_ASSERT(configsNumber != 0U); /* Power configuration index is valid. */
77  DEV_ASSERT(gPowerManagerState.configs == NULL); /* Driver is not initialized, reference to configuration is not valid. */
78  DEV_ASSERT(gPowerManagerState.configsNumber == 0U); /* Driver is not initialized, number of configurations is zero. */
79 
80  /* Store references to user-defined power mode configurations */
81  gPowerManagerState.configs = (power_manager_user_config_t * (*)[])powerConfigsPtr;
82  gPowerManagerState.configsNumber = configsNumber;
83  gPowerManagerState.currentConfig = 0U;
84 
85  /* Store references to user-defined callback configurations and increment call-back handle counter */
86  if (callbacksPtr != NULL)
87  {
88  gPowerManagerState.staticCallbacks = (power_manager_callback_user_config_t * (*)[])callbacksPtr;
89  gPowerManagerState.staticCallbacksNumber = callbacksNumber;
90  /* Default value of handle of last call-back that returned error */
91  gPowerManagerState.errorCallbackIndex = callbacksNumber;
92  }
93  else
94  {
95  gPowerManagerState.staticCallbacks = NULL;
96  gPowerManagerState.staticCallbacksNumber = 0U;
97  gPowerManagerState.errorCallbackIndex = 0U;
98  }
99 
100  return POWER_SYS_DoInit();
101 }

216 /*FUNCTION**********************************************************************************************
217 * Function Name : POWER_SYS_CallbacksManagement
218 * Description : Internal function used by POWER_SYS_SetMode function for callback management
219 * notifyStruct callback notification structure
220 * currentStaticCallback index to array of statically registered call-backs
221 * policy transaction policy
222 *
223 *END***************************************************************************************************/
224 static status_t POWER_SYS_CallbacksManagement(power_manager_notify_struct_t * notifyStruct,
225  uint8_t * currentStaticCallback,
226  power_manager_policy_t policy)
227 {
228  uint8_t callbacksNumber; /* The total number of callbacks */
229  status_t returnCode = STATUS_SUCCESS; /* Function return */
230  status_t errorCode = STATUS_SUCCESS; /* Error code to be returned(error case) */
231  status_t callbackStatusCode; /* Status code returned by callback function */
232  power_manager_callback_type_t callbackTypeFilter; /* Callback types to be excluded */
233 
234  switch (notifyStruct->notifyType)
235  {
236  /* notify before */
237  case POWER_MANAGER_NOTIFY_BEFORE:
238  callbacksNumber = gPowerManagerState.staticCallbacksNumber;
239  callbackTypeFilter = POWER_MANAGER_CALLBACK_AFTER;
240  errorCode = STATUS_MCU_NOTIFY_BEFORE_ERROR;
241  break;
242  /* notify after */
243  case POWER_MANAGER_NOTIFY_AFTER:
244  callbacksNumber = gPowerManagerState.staticCallbacksNumber;
245  callbackTypeFilter = POWER_MANAGER_CALLBACK_BEFORE;
246  errorCode = STATUS_MCU_NOTIFY_AFTER_ERROR;
247  break;
248  /* notify recover */
249  case POWER_MANAGER_NOTIFY_RECOVER:
250  callbacksNumber = gPowerManagerState.staticCallbacksNumber;
251  callbackTypeFilter = POWER_MANAGER_CALLBACK_AFTER;
252  errorCode = STATUS_MCU_NOTIFY_BEFORE_ERROR;
253  break;
254  default:
255  /* invalid value for notifyType */
256  DEV_ASSERT(false);
257  callbacksNumber = 0U;
258  callbackTypeFilter = POWER_MANAGER_CALLBACK_BEFORE;
259  break;
260  }
261 
262  /* From all statically registered call-backs... */
263  for ((*currentStaticCallback) = 0U; (*currentStaticCallback) < callbacksNumber; (*currentStaticCallback)++)
264  {
265  /* Pointer to callback configuration */
266  const power_manager_callback_user_config_t * const callbackConfig = ((*gPowerManagerState.staticCallbacks)[*currentStaticCallback]);
267 
268  /* Check pointer to static callback configuration */
269  if ((callbackConfig != NULL) && (callbackTypeFilter != callbackConfig->callbackType))
270  {
271  /* In case that call-back returned error code mark it, store the call-back handle and eventually cancel the mode switch */
272 
273  callbackStatusCode = callbackConfig->callbackFunction(notifyStruct, callbackConfig->callbackData);
274  if (STATUS_SUCCESS != callbackStatusCode)
275  {
276  returnCode = errorCode;
277  gPowerManagerState.errorCallbackIndex = *currentStaticCallback;
278  /* If not forcing power mode switch, call all already notified call-backs to revert their state as the mode change is canceled */
279  if (policy != POWER_MANAGER_POLICY_FORCIBLE)
280  {
281  break;
282  }
283  }
284  }
285  }
286 
287  return returnCode;
288 }

290 /*FUNCTION**********************************************************************
291  *
292  * Function Name : POWER_SYS_SetMode
293  * Description : Configures the power mode.
294  *
295  * This function switches to one of the defined power modes. Requested mode number is passed
296  * as an input parameter. This function notifies all registered callback functions before
297  * the mode change (using POWER_MANAGER_CALLBACK_BEFORE set as callback type parameter),
298  * sets specific power options defined in the power mode configuration and enters the specified
299  * mode. In case of success switch, this function also invokes all registered callbacks after
300  * the mode change (using POWER_MANAGER_CALLBACK_AFTER).
301  * The actual mode switch is performed by POWER_SYS_DoSetMode in the specific implementation.
302  * Callbacks are invoked in the following order: All registered callbacks are notified
303  * ordered by index in the callbacks array (see callbacksPtr parameter of POWER_SYS_Init()).
304  * The same order is used for before and after switch notifications.
305  * The notifications before the power mode switch can be used to obtain confirmation about
306  * the change from registered callbacks. If any registered callback denies the power
307  * mode change, further execution of this function depends on mode change policy: the mode
308  * change is either forced(POWER_MANAGER_POLICY_FORCIBLE) or exited(POWER_MANAGER_POLICY_AGREEMENT).
309  * When mode change is forced, the result of the before switch notifications are ignored. If
310  * agreement is required, if any callback returns an error code then further notifications
311  * before switch notifications are cancelled and all already notified callbacks are re-invoked
312  * with POWER_MANAGER_CALLBACK_AFTER set as callback type parameter. The index of the callback
313  * which returned error code during pre-switch notifications is stored(any error codes during
314  * callbacks re-invocation are ignored) and POWER_SYS_GetErrorCallback() can be used to get it.
315  * Regardless of the policies, if any callback returned an error code, an error code denoting in which phase
316  * the error occurred is returned when POWER_SYS_SetMode() exits.
317  * It is possible to enter any mode supported by the processor. Refer to the chip reference manual
318  * for list of available power modes. If it is necessary to switch into intermediate power mode prior to
319  * entering requested mode, then the intermediate mode is entered without invoking the callback mechanism.
320  *
321  * Implements POWER_SYS_SetMode_Activity
322  *END**************************************************************************/
323 status_t POWER_SYS_SetMode(uint8_t powerModeIndex,
324  power_manager_policy_t policy)
325 {
326  power_manager_user_config_t * configPtr; /* Local pointer to the requested user-defined power mode configuration */
327  status_t returnCode; /* Function return */
328  status_t errorCode;
329  bool successfulSwitch; /* Power mode switch is successful or not */
330  uint8_t currentStaticCallback = 0U; /* Index to array of statically registered call-backs */
331  power_manager_notify_struct_t notifyStruct; /* Callback notification structure */
332 
333  /* Driver is already initialized. */
334  DEV_ASSERT(gPowerManagerState.configs != NULL);
335  DEV_ASSERT(gPowerManagerState.configsNumber != 0U);
336 
337  /* Power mode index is valid. */
338  DEV_ASSERT(powerModeIndex < gPowerManagerState.configsNumber);
339 
340  /* Initialization of local pointer to the requested user-defined power mode configuration */
341  configPtr = (*gPowerManagerState.configs)[powerModeIndex];
342 
343  /* Reference to the requested user-defined power mode configuration is valid. */
344  DEV_ASSERT(configPtr != NULL);
345 
346  /* Default value of handle of last call-back that returned error */
347  gPowerManagerState.errorCallbackIndex = gPowerManagerState.staticCallbacksNumber;
348 
349  /* Set the transaction policy in the notification structure */
350  notifyStruct.policy = policy;
351 
352  /* Set the target power mode configuration in the notification structure */
353  notifyStruct.targetPowerConfigIndex = powerModeIndex;
354  notifyStruct.targetPowerConfigPtr = configPtr;
355 
356  /* Notify those which asked to be called before the power mode change */
357  notifyStruct.notifyType = POWER_MANAGER_NOTIFY_BEFORE;
358  returnCode = POWER_SYS_CallbacksManagement(&notifyStruct, &currentStaticCallback, policy);
359 
360  /* Power mode switch */
361  /* In case that any call-back returned error code and policy doesn't force the mode switch go to after switch call-backs */
362  if ((policy == POWER_MANAGER_POLICY_FORCIBLE) || (returnCode == STATUS_SUCCESS))
363  {
364  returnCode = POWER_SYS_DoSetMode(configPtr);
365  successfulSwitch = (STATUS_SUCCESS == returnCode);
366  }
367  else
368  {
369  /* Unsuccessful switch */
370  successfulSwitch = false;
371  }
372 
373  if (successfulSwitch)
374  {
375  /* End of successful switch */
376 
377  /* Update current configuration index */
378  gPowerManagerState.currentConfig = powerModeIndex;
379 
380  /* Notify those which asked to be called after the power mode change */
381  notifyStruct.notifyType = POWER_MANAGER_NOTIFY_AFTER;
382  returnCode = POWER_SYS_CallbacksManagement(&notifyStruct, &currentStaticCallback, POWER_MANAGER_POLICY_FORCIBLE);
383  }
384  else
385  {
386  /* End of unsuccessful switch */
387 
388  /* Notify those which have been called before the power mode change */
389  notifyStruct.notifyType = POWER_MANAGER_NOTIFY_RECOVER;
390  errorCode = POWER_SYS_CallbacksManagement(&notifyStruct, &currentStaticCallback, POWER_MANAGER_POLICY_FORCIBLE);
391  (void)(errorCode);
392  }
393 
394  return returnCode;
395 }
