125 status_t LPSPI_DRV_SlaveInit(uint32_t instance,
126  lpspi_state_t * lpspiState,
127  const lpspi_slave_config_t * slaveConfig)
128 {
129  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
130  DEV_ASSERT(lpspiState != NULL);
131  DEV_ASSERT(slaveConfig != NULL);
132  LPSPI_Type * base = g_lpspiBase[instance];
133  status_t errorCode = STATUS_SUCCESS;
134 
135  lpspiState->lsb = slaveConfig->lsbFirst;
136  lpspiState->bitsPerFrame = slaveConfig->bitcount;
137  lpspiState->transferType = slaveConfig->transferType;
138  lpspiState->isBlocking = false;
139  /* Store DMA channels numbers used for DMA transfer */
140  lpspiState->rxDMAChannel = slaveConfig->rxDMAChannel;
141  lpspiState->txDMAChannel = slaveConfig->txDMAChannel;
142  /* Store callback */
143  lpspiState->callback = slaveConfig->callback;
144  lpspiState->callbackParam = slaveConfig->callbackParam;
145  /* Calculate the bytes/frame for lpspiState->bytesPerFrame. */
146  lpspiState->bytesPerFrame = (uint16_t)((lpspiState->bitsPerFrame + 7U) / 8U);
147  /* Due to DMA limitations frames of 3 bytes/frame will be internally handled as 4 bytes/frame. */
148  if (lpspiState->bytesPerFrame == 3U)
149  {
150  lpspiState->bytesPerFrame = 4U;
151  }
152  /* Due to some limitations all frames bigger than 4 bytes/frame must be composed only from 4 bytes chunks. */
153  if (lpspiState->bytesPerFrame > 4U)
154  {
155  lpspiState->bytesPerFrame = (((lpspiState->bytesPerFrame - 1U) / 4U) + 1U) * 4U;
156  }
157  lpspiState->isTransferInProgress = false;
158  /* Initialize the semaphore */
159  errorCode = OSIF_SemaCreate(&(lpspiState->lpspiSemaphore), 0);
160  DEV_ASSERT(errorCode == STATUS_SUCCESS);
161  g_lpspiStatePtr[instance] = lpspiState;
162 
163  /* Configure registers */
164  LPSPI_Init(base);
165 
166  /* Configure lpspi to slave mode */
167  (void)LPSPI_SetMasterSlaveMode(base, LPSPI_SLAVE);
168  /* Set Pin settings */
169  (void)LPSPI_SetPinConfigMode(base, LPSPI_SDI_IN_SDO_OUT, LPSPI_DATA_OUT_RETAINED, true);
170  /* Calculate the FIFO size for the LPSPI */
171  LPSPI_GetFifoSizes(base, &(lpspiState->fifoSize));
172 
173  /* Set polarity */
174  (void)LPSPI_SetPcsPolarityMode(base, slaveConfig->whichPcs, slaveConfig->pcsPolarity);
175 
176  /* Write the TCR for this transfer */
177  lpspi_tx_cmd_config_t txCmdCfg = {
178  .frameSize = lpspiState->bitsPerFrame,
179  .width = LPSPI_SINGLE_BIT_XFER,
180  .txMask = false,
181  .rxMask = false,
182  .byteSwap = false,
183  .lsbFirst = slaveConfig->lsbFirst,
184  .clkPhase = slaveConfig->clkPhase,
185  .clkPolarity = slaveConfig->clkPolarity,
186  .whichPcs = slaveConfig->whichPcs
187  };
188 
189  /* Write to the TX CMD register */
190  LPSPI_SetTxCommandReg(base, &txCmdCfg);
191  LPSPI_Enable(base);
192  /* Enable the interrupt source */
193  INT_SYS_EnableIRQ(g_lpspiIrqId[instance]);
194 
195  return errorCode;
196 }

230 status_t LPSPI_DRV_SlaveTransferBlocking(uint32_t instance,
231  const uint8_t *sendBuffer,
232  uint8_t *receiveBuffer,
233  uint16_t transferByteCount,
234  uint32_t timeout)
235 {
236  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
237  DEV_ASSERT(g_lpspiStatePtr[instance] != NULL);
238  lpspi_state_t * state = (lpspi_state_t *)g_lpspiStatePtr[instance];
239  const LPSPI_Type *base = g_lpspiBase[instance];
240  status_t error;
241  status_t osifError;
242 
243  /* Check if another transfer is in progress */
244  if (LPSPI_GetStatusFlag(base, LPSPI_MODULE_BUSY))
245  {
246  return STATUS_BUSY;
247  }
248 
249  /* Dummy wait to ensure the semaphore is 0, no need to check result */
250  (void)OSIF_SemaWait(&(state->lpspiSemaphore), 0);
251  state->isBlocking = true;
252 
253  error = LPSPI_DRV_SlaveTransfer(instance, sendBuffer, receiveBuffer, transferByteCount);
254  if(error != STATUS_SUCCESS)
255  {
256  state->isBlocking = false;
257  LPSPI_DRV_DisableTEIEInterrupts(instance);
258  return error;
259  }
260  /* As this is a synchronous transfer, wait until the transfer is complete.*/
261  osifError = OSIF_SemaWait(&(state->lpspiSemaphore), timeout);
262 
263  if (osifError == STATUS_TIMEOUT)
264  {
265  /* Set isBlocking variable to false to avoid dummy semaphore post. */
266  state->isBlocking = false;
267  /* Complete transfer. */
268  (void)LPSPI_DRV_SlaveAbortTransfer(instance);
269  return(STATUS_TIMEOUT);
270  }
271 
272  LPSPI_DRV_DisableTEIEInterrupts(instance);
273 
274  return STATUS_SUCCESS;
275 }

280 status_t LPSPI_DRV_SlaveTransfer(uint32_t instance,
281  const uint8_t *sendBuffer,
282  uint8_t *receiveBuffer,
283  uint16_t transferByteCount)
284 {
285  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
286  DEV_ASSERT(g_lpspiStatePtr[instance] != NULL);
287  DEV_ASSERT(!((sendBuffer == NULL) && (receiveBuffer == NULL)));
288  LPSPI_Type * base = g_lpspiBase[instance];
289  lpspi_state_t * state = (lpspi_state_t *)g_lpspiStatePtr[instance];
290  state->dummy = 0xFFU;
291  edma_transfer_size_t dmaTransferSize = EDMA_TRANSFER_SIZE_1B;
292  const uint8_t * buffer;
293 
294  /* The number of transferred bytes should be divisible by frame size */
295  if ((uint16_t)(transferByteCount % state->bytesPerFrame) != (uint16_t)0)
296  {
297  return STATUS_ERROR;
298  }
299  /* Check if LPSPI module isn't busy */
300  if (state->isTransferInProgress == true)
301  {
302  return STATUS_BUSY;
303  }
304  /* Initialize the status of the current transfer */
305  state->status = LPSPI_TRANSFER_OK;
306  /* Clean RX and TX buffers */
307  LPSPI_SetFlushFifoCmd(base, true, true);
308  /* The second flush command is used to avoid the case when one word is still in shifter. */
309  LPSPI_SetFlushFifoCmd(base, true, true);
310  /* Clear all interrupts sources */
311  (void)LPSPI_ClearStatusFlag(base, LPSPI_ALL_STATUS);
312  /* Enable fault interrupts sources */
313  LPSPI_SetIntMode(base,LPSPI_TRANSMIT_ERROR , true);
314  LPSPI_SetIntMode(base,LPSPI_RECEIVE_ERROR , true);
315  /* Fill out the other members of the run-time state structure. */
316  state->txBuff = (const uint8_t *)sendBuffer;
317  state->rxBuff = (uint8_t *)receiveBuffer;
318  if (state->transferType == LPSPI_USING_INTERRUPTS)
319  {
320 
321  if(state->txBuff != NULL)
322  {
323  state->txCount = transferByteCount;
324  LPSPI_ClearTxmaskBit(base);
325  }
326  else
327  {
328  state->txCount = 0;
329  LPSPI_SetTxmskBit(base);
330  }
331 
332  if(state->rxBuff != NULL)
333  {
334  state->rxCount = transferByteCount;
335  LPSPI_ClearRxmaskBit(base);
336  }
337  else
338  {
339  state->rxCount = 0;
340  LPSPI_SetRxmskBit(base);
341  }
342 
343  state->txFrameCnt = 0;
344  state->rxFrameCnt = 0;
345  state->isPcsContinuous = false;
346  /* Configure watermarks */
347  LPSPI_SetRxWatermarks(base, 0U);
348  LPSPI_SetTxWatermarks(base, 2U);
349 
350  state->isTransferInProgress = true;
351  /* Enable interrupts for RX and TX only if it's necessary */
352  if(state->txBuff != NULL)
353  {
354  LPSPI_SetIntMode(base,LPSPI_TX_DATA_FLAG , true);
355  }
356  if(state->rxBuff != NULL)
357  {
358  LPSPI_SetIntMode(base, LPSPI_RX_DATA_FLAG, true);
359  }
360 
361  }
362  else
363  {
364  /* Configure watermarks */
365  LPSPI_SetRxWatermarks(base, 0U);
366  LPSPI_SetTxWatermarks(base, 3U);
367  /* When LPSPI use DMA frames with 3 bytes size are not accepted. */
368  switch(state->bytesPerFrame)
369  {
370  case 1: dmaTransferSize = EDMA_TRANSFER_SIZE_1B; break;
371  case 2: dmaTransferSize = EDMA_TRANSFER_SIZE_2B; break;
372  case 4: dmaTransferSize = EDMA_TRANSFER_SIZE_4B; break;
373  default: dmaTransferSize = EDMA_TRANSFER_SIZE_4B; break;
374  }
375 
376  if(receiveBuffer != NULL)
377  {
378  state->rxCount = transferByteCount;
379  buffer = receiveBuffer;
380  }
381  else
382  {
383  state->rxCount = 0U;
384  /* if there is no data to receive, use dummy data as destination for DMA transfer */
385  buffer = (uint8_t *)(&(state->dummy));
386  }
387  (void)EDMA_DRV_ConfigMultiBlockTransfer(state->rxDMAChannel,
388  EDMA_TRANSFER_PERIPH2MEM,
389  (uint32_t)(&(base->RDR)),
390  (uint32_t)buffer,
391  dmaTransferSize,
392  (uint32_t)1U<<(uint8_t)(dmaTransferSize),
393  (uint32_t)transferByteCount/(uint32_t)((uint32_t)1U <<(uint8_t)(dmaTransferSize)),
394  true);
395  if(receiveBuffer == NULL)
396  {
397  /* if there is no data to receive, don't increment destination offset */
398  EDMA_DRV_SetDestOffset(state->rxDMAChannel, 0);
399  }
400 
401  if(sendBuffer != NULL)
402  {
403  state->txCount = transferByteCount;
404  buffer = sendBuffer;
405  }
406  else
407  {
408  state->txCount = 0U;
409  /* if there is no data to send, use dummy data as source for DMA transfer */
410  buffer = (uint8_t *)(&(state->dummy));
411  }
412  (void)EDMA_DRV_ConfigMultiBlockTransfer(state->txDMAChannel, EDMA_TRANSFER_MEM2PERIPH,
413  (uint32_t)buffer, (uint32_t)(&(base->TDR)), dmaTransferSize, (uint32_t)1U<<(uint8_t)(dmaTransferSize),
414  (uint32_t)transferByteCount/(uint32_t)((uint32_t)1U <<(uint8_t)(dmaTransferSize)), true);
415  if (sendBuffer == NULL)
416  {
417  /* if there is no data to transmit, don't increment source offset */
418  EDMA_DRV_SetSrcOffset(state->txDMAChannel, 0);
419  }
420 
421  (void)EDMA_DRV_InstallCallback(state->rxDMAChannel, (LPSPI_DRV_SlaveCompleteDMATransfer),(void*)(instance));
422 
423  state->isTransferInProgress = true;
424 
425  /* Start RX channel */
426  (void)EDMA_DRV_StartChannel(state->rxDMAChannel);
427  /* Start TX channel */
428  (void)EDMA_DRV_StartChannel(state->txDMAChannel);
429  /* Enable LPSPI DMA requests */
430  LPSPI_SetRxDmaCmd(base, true);
431  LPSPI_SetTxDmaCmd(base, true);
432 
433  }
434  return STATUS_SUCCESS;
435 }

437 void LPSPI_DRV_SlaveIRQHandler(uint32_t instance)
438 {
439  LPSPI_Type * base = g_lpspiBase[instance];
440  lpspi_state_t * lpspiState = (lpspi_state_t *)g_lpspiStatePtr[instance];
441 
442  /* If an error is detected the transfer will be aborted */
443  if ((bool)LPSPI_GetStatusFlag(base, LPSPI_TRANSMIT_ERROR) && (lpspiState->txBuff != NULL))
444  {
445  (void)LPSPI_DRV_SlaveAbortTransfer(instance);
446  lpspiState->status = LPSPI_TRANSMIT_FAIL;
447  return;
448  }
449  if (LPSPI_GetStatusFlag(base, LPSPI_RECEIVE_ERROR) && (lpspiState->rxBuff != NULL))
450  {
451  (void)LPSPI_DRV_SlaveAbortTransfer(instance);
452  lpspiState->status = LPSPI_RECEIVE_FAIL;
453  return;
454  }
455 
456  /* Receive data */
457  if (LPSPI_GetStatusFlag(base,LPSPI_RX_DATA_FLAG))
458  {
459  if ((lpspiState->rxCount != (uint8_t)0))
460  {
461  LPSPI_DRV_ReadRXBuffer(instance);
462  }
463  }
464  /* Transmit data */
465  if (LPSPI_GetStatusFlag(base,LPSPI_TX_DATA_FLAG))
466  {
467  if ((lpspiState->txCount != (uint8_t)0))
468  {
469  LPSPI_DRV_FillupTxBuffer(instance);
470  }
471  }
472  /* If all bytes are sent disable interrupt TDF */
473  if (lpspiState->txCount == (uint8_t)0)
474  {
475  LPSPI_SetIntMode(base, LPSPI_TX_DATA_FLAG, false);
476  }
477  /* If all bytes are received disable interrupt RDF */
478  if (lpspiState->rxCount == (uint8_t)0)
479  {
480  LPSPI_SetIntMode(base, LPSPI_RX_DATA_FLAG, false);
481  }
482  if (lpspiState->rxCount == (uint8_t)0)
483  {
484  if (lpspiState->txCount == (uint8_t)0)
485  {
486  /* Disable fault interrupts sources */
487  LPSPI_SetIntMode(base,LPSPI_TRANSMIT_ERROR , false);
488  LPSPI_SetIntMode(base,LPSPI_RECEIVE_ERROR , false);
489 
490  /* Call the callback if it is defined */
491  if (lpspiState->callback != NULL)
492  {
493  lpspiState->callback(lpspiState, SPI_EVENT_END_TRANSFER, lpspiState->callbackParam);
494  }
495 
496  /* If the transfer is blocking post the semaphore */
497  if(lpspiState->isBlocking == true)
498  {
499  (void)OSIF_SemaPost(&(lpspiState->lpspiSemaphore));
500  lpspiState->isBlocking = false;
501  }
502 
503  /* Update internal state of the transfer */
504  lpspiState->isTransferInProgress = false;
505  }
506  }
507 }

512 status_t LPSPI_DRV_SlaveAbortTransfer(uint32_t instance)
513 {
514  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
515  DEV_ASSERT(g_lpspiStatePtr[instance] != NULL);
516  LPSPI_Type * base = g_lpspiBase[instance];
517  lpspi_state_t * state = (lpspi_state_t *)g_lpspiStatePtr[instance];
518

520  {

524  }
525  else
526  {

528  LPSPI_SetRxDmaCmd(base, false);
529  LPSPI_SetTxDmaCmd(base, false);
530  }
531 
532  LPSPI_DRV_DisableTEIEInterrupts(instance);
533 
534  state->isTransferInProgress = false;
535  /* Clean RX and TX buffers */
536  LPSPI_SetFlushFifoCmd(base, true, true);
537  /* The second flush command is used to avoid the case when one word is still in shifter. */
538  LPSPI_SetFlushFifoCmd(base, true, true);
539  if(state->isBlocking == true)
540  {
541  (void)OSIF_SemaPost(&(state->lpspiSemaphore));
542  state->isBlocking = false;
543  }
544  return STATUS_SUCCESS;
