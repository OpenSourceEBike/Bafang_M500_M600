226  *

229  *
230  * Implements : lin_condittional_change_nad_Activity

232 static void lin_condittional_change_nad(l_ifc_handle iii)
233 {
234  l_u8 id;
235  l_u8 byte;
236  l_u8 mask;
237  l_u8 invert;
238  l_bool give_positive_flg = (bool)0U;
239  const lin_node_attribute_t * node_attr_ptr = &g_lin_node_attribute_array[g_lin_protocol_user_cfg_array[iii].slave_ifc_handle];
240  const lin_transport_layer_queue_t * rx_queue;
241  lin_product_id_t product_id = node_attr_ptr->product_id;
242  lin_serial_number_t serial_number = node_attr_ptr->serial_number;
243 
244  /* Get receive queue */
245  rx_queue = &(g_lin_tl_descriptor_array[iii].tl_rx_queue);
246  id = rx_queue->tl_pdu_ptr[rx_queue->queue_header][3];
247  byte = rx_queue->tl_pdu_ptr[rx_queue->queue_header][4];
248  mask = rx_queue->tl_pdu_ptr[rx_queue->queue_header][5];
249  invert = rx_queue->tl_pdu_ptr[rx_queue->queue_header][6];
250 
251  switch (id)
252  {
253  case LIN_PRODUCT_ID:
254  if ((byte > 0U) && (byte < 6U))
255  {
256  /* Byte 1: Supplier ID LSB; Byte 2: Supplier ID MSB */
257  if ((byte > 0U) && (byte < 3U))
258  {
259  byte = (l_u8)(product_id.supplier_id >> ((byte - 1U) * 8U));
260  }
261  /* Byte 3: Function ID LSB; Byte 4: Function ID MSB */
262  else if ((byte > 2U) && (byte < 5U))
263  {
264  byte = (l_u8)(product_id.function_id >> ((byte - 3U) * 8U));
265  }
266  /* Byte 5: Variant */
267  else
268  {
269  byte = product_id.variant;
270  }
271 
272  /* Do a bitwise XOR with Invert and Do a bitwise AND with Mask */
273  byte = (l_u8)((byte ^ invert) & mask);
274  if (byte == 0U)
275  {
276  give_positive_flg = (l_bool)1U;
277  }
278  }
279 
280  break;
281  case LIN_SERIAL_NUMBER:
282  if ((byte > 0U) && (byte < 5U))
283  {
284  switch (byte)
285  {
286  case 1U:
287  byte = serial_number.serial_0;
288  break;
289  case 2U:
290  byte = serial_number.serial_1;
291  break;
292  case 3U:
293  byte = serial_number.serial_2;
294  break;
295  case 4U:
296  byte = serial_number.serial_3;
297  break;
298  default:
299  /* Do nothing */
300  break;
301  }
302 
303  /* Do a bitwise XOR with Invert and Do a bitwise AND with Mask */
304  byte = (l_u8)((byte ^ invert) & mask);
305  if (byte == 0U)
306  {
307  give_positive_flg = (l_bool)1U;
308  }
309  }
310 
311  break;
312  default:
313  /* Do nothing */
314  break;
315  }
316 
317  if (give_positive_flg == (bool)1U)
318  {
319  /* Make response PDU before change configuration NAD */
320  ld_make_slave_response_pdu(iii, SERVICE_CONDITIONAL_CHANGE_NAD, POSITIVE, 0U);
321  /* If the final result is zero then change the NAD to New NAD */
322  *node_attr_ptr->configured_NAD_ptr = rx_queue->tl_pdu_ptr[rx_queue->queue_header][7U];
323  }
324 }

333 static void lin_assign_nad(l_ifc_handle iii)
334 {
335  lin_tl_pdu_data_t lin_tl_pdu;
336  l_u16 supid;
337  l_u16 fid;
338  lin_product_id_t product_id = g_lin_node_attribute_array[g_lin_protocol_user_cfg_array[iii].slave_ifc_handle].product_id;
339  const lin_transport_layer_queue_t * rx_queue = &(g_lin_tl_descriptor_array[iii].tl_rx_queue);
340  l_u8 i;
341  for (i = 0U; i < 8U; i++)
342  {
343  lin_tl_pdu[i] = rx_queue->tl_pdu_ptr[rx_queue->queue_header][i];
344  }
345 
346  /* Get supplier and function identification in request */
347  supid = (l_u16)(((l_u16)lin_tl_pdu[4U]) << 8U);
348  supid = (l_u16)(supid | ((l_u16)lin_tl_pdu[3]));
349 
350  fid = (l_u16)(((l_u16)lin_tl_pdu[6U]) << 8U);
351  fid = (l_u16)(fid | ((l_u16)lin_tl_pdu[5]));
352 
353  /* Check Supplier ID and Function ID */
354  if (((supid == product_id.supplier_id) || (supid == LD_ANY_SUPPLIER)) &&
355  ((fid == product_id.function_id) || (fid == LD_ANY_FUNCTION)))
356  {
357  ld_make_slave_response_pdu(iii, SERVICE_ASSIGN_NAD, POSITIVE, 0U);
358  }

370 static void lin_diagservice_assign_frame_id_range(l_ifc_handle iii)
371 {
372  DEV_ASSERT((l_u8)iii < LIN_NUM_OF_IFCS);
373 
374  l_u8 j;
375  l_u8 i;
376  l_u8 start_index;
377  l_u8 cfg_frame_num = 0U;
378  lin_tl_pdu_data_t lin_tl_pdu;
379  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
380  const lin_transport_layer_queue_t * rx_queue = &(g_lin_tl_descriptor_array[iii].tl_rx_queue);
381  l_u8 storePID = 1U;
382  for (i = 0U; i < 8U; i++)
383  {
384  lin_tl_pdu[i] = rx_queue->tl_pdu_ptr[rx_queue->queue_header][i];
385  }
386 
387  /* Get start index in request */
388  start_index = lin_tl_pdu[3U];
389 
390  /* Find the number of configurable frames plus 2, including 0x3C and 0x3D */
391  i = 1U;
392  while (0xFFFFU != (prot_user_config_ptr->list_identifiers_ROM_ptr)[i++])
393  {
394  cfg_frame_num++;
395  }
396 
397  /* Calculate number of configurable frames */
398  cfg_frame_num = (l_u8)(cfg_frame_num - 3U);
399 
400  i = 4U;
401  /* Check request validity */
402  for (j = start_index; j < (start_index + 4U); j++)
403  {
404  if ((0xFFU != lin_tl_pdu[i++]) && (j > cfg_frame_num))
405  {
406  ld_make_slave_response_pdu(iii, SERVICE_ASSIGN_FRAME_ID_RANGE, NEGATIVE, GENERAL_REJECT);
407  storePID = 0U;
408  break;
409  }
410  }
411 
412  if (storePID == 1U)
413  {
414  /* Store PIDs */
415  for (i = 4U; i < 8U; i++)
416  {
417  switch (lin_tl_pdu[i])
418  {
419  case 0x00U:
420  /* Unassign frame */
421  start_index++;
422  (prot_user_config_ptr->list_identifiers_RAM_ptr)[start_index] = 0xFFU;
423  break;
424  case 0xFFU:
425  /* keep the previous assigned value of this frame */
426  break;
427  default:
428  /* Calculate frame ID and Assign ID to frame */
429  start_index++;
430  (prot_user_config_ptr->list_identifiers_RAM_ptr)[start_index] = lin_process_parity(lin_tl_pdu[i], CHECK_PARITY);
431  break;
432  }
433  } /* End of for statement */
434 
435  ld_make_slave_response_pdu(iii, SERVICE_ASSIGN_FRAME_ID_RANGE, POSITIVE, 0U);
436  }
437 }
