120  * Function Name : UART_AllocateState
121  * Description : Allocates one of the available state structures.
122  *
123  *END**************************************************************************/
124 static uint8_t UART_AllocateState(bool* isAllocated,
125  uint8_t * instanceMapping,
126  const uart_instance_t* const instance,
127  uint8_t numberOfinstances)
128 {
129  DEV_ASSERT((isAllocated != NULL) &&
130  (instanceMapping != NULL) &&
131  (instance != NULL));
132 
133  uint8_t i;
134  uint32_t uartIdx = instance->instIdx;
135 
136  /* Allocate one of the UART state structures for this instance */
137  for (i = 0; i < numberOfinstances; i++)
138  {
139  if (isAllocated[i] == false)
140  {
141  instanceMapping[i] = (uint8_t)uartIdx;
142  isAllocated[i] = true;
143  break;
144  }
145  }
146 
147  return i;
148 }

207 /*FUNCTION**********************************************************************
208  *
209  * Function Name : UART_GetConfigurationUsingFlexio
210  * Description : Assign all user configuration to flexio_uart configuration
211  *
212  *END**************************************************************************/
213 static void UART_GetConfigurationUsingFlexio(const uart_user_config_t * config,
214  flexio_uart_user_config_t * flexioUartTxConfig,
215  flexio_uart_user_config_t * flexioUartRxConfig)
216 {
217  /* FlexIO driver only can be used with parity disabled and one stop bit */
218  DEV_ASSERT((config->parityMode == UART_PARITY_DISABLED) &&
219  (config->stopBitCount == UART_ONE_STOP_BIT));
220 
221  uint32_t baudRate = config->baudRate;
222 
223  /* Set baud rate for Tx and Rx */
224  flexioUartTxConfig->baudRate = baudRate;
225  flexioUartRxConfig->baudRate = baudRate;
226 
227  /* Set transfer type for Tx and Rx */
228  switch (config->transferType)
229  {
230  case UART_USING_DMA:
231  flexioUartTxConfig->driverType = FLEXIO_DRIVER_TYPE_DMA;
232  flexioUartRxConfig->driverType = FLEXIO_DRIVER_TYPE_DMA;
233  break;
234  case UART_USING_INTERRUPTS:
235  flexioUartTxConfig->driverType = FLEXIO_DRIVER_TYPE_INTERRUPTS;
236  flexioUartRxConfig->driverType = FLEXIO_DRIVER_TYPE_INTERRUPTS;
237  break;
238  default:
239  /* Impossible type - do nothing */
240  break;
241  }
242 
243  /* Set bit count per char for Tx and Rx */
244  switch (config->bitCount)
245  {
246  case UART_7_BITS_PER_CHAR:
247  flexioUartTxConfig->bitCount = 7U;
248  flexioUartRxConfig->bitCount = 7U;
249  break;
250  case UART_8_BITS_PER_CHAR:
251  flexioUartTxConfig->bitCount = 8U;
252  flexioUartRxConfig->bitCount = 8U;
253  break;
254  case UART_9_BITS_PER_CHAR:
255  flexioUartTxConfig->bitCount = 9U;
256  flexioUartRxConfig->bitCount = 9U;
257  break;
258  case UART_10_BITS_PER_CHAR:
259  flexioUartTxConfig->bitCount = 10U;
260  flexioUartRxConfig->bitCount = 10U;
261  break;
262  case UART_15_BITS_PER_CHAR:
263  flexioUartTxConfig->bitCount = 15U;
264  flexioUartRxConfig->bitCount = 15U;
265  break;
266  case UART_16_BITS_PER_CHAR:
267  flexioUartTxConfig->bitCount = 16U;
268  flexioUartRxConfig->bitCount = 16U;
269  break;
270  default:
271  /* Impossible type - do nothing */
272  break;
273  }
274 
275  /* Configure Tx */
276  flexioUartTxConfig->direction = FLEXIO_UART_DIRECTION_TX;
277  flexioUartTxConfig->dmaChannel = config->txDMAChannel;
278  flexioUartTxConfig->dataPin = ((extension_flexio_for_uart_t *)(config->extension))->dataPinTx;
279 
280  /* Configure Rx */
281  flexioUartRxConfig->direction = FLEXIO_UART_DIRECTION_RX;
282  flexioUartRxConfig->dmaChannel = config->rxDMAChannel;
283  flexioUartRxConfig->dataPin = ((extension_flexio_for_uart_t *)(config->extension))->dataPinRx;
284 
285  /* Link Flexio Callbacks to the callbacks defined in PAL */
286  flexioUartRxConfig->callback = config->rxCallback;
287  flexioUartRxConfig->callbackParam = config->rxCallbackParam;
288  flexioUartTxConfig->callback = config->txCallback;
289  flexioUartTxConfig->callbackParam = config->txCallbackParam;
290 }

412 /*FUNCTION**********************************************************************
413  *
414  * Function Name : UART_Init
415  * Description : Configures the UART module
416  *
417  * Implements : UART_Init_Activity
418  *END**************************************************************************/
419 status_t UART_Init(const uart_instance_t * const instance, const uart_user_config_t *config)
420 {
421  DEV_ASSERT((instance != NULL) &&
422  (config != NULL));
423 
424  status_t status = STATUS_ERROR;
425  uint8_t index = 0;
426 
427  switch (instance->instType)
428  {
429  /* Define UART PAL over LPUART */
430  #if (defined(UART_OVER_LPUART))
431  case UART_INST_TYPE_LPUART:
432  /*instance index is available */
433  DEV_ASSERT(instance->instIdx < LPUART_INSTANCE_COUNT);
434 
435  /* Assign all user configuration to lpuart configuration */
436  lpuart_user_config_t lpuartConfig;
437  UART_GetConfigurationUsingLpuart(config, &lpuartConfig);
438 
439  /* Allocate one of the LPUART state structure for this instance */
440  index = UART_AllocateState(s_lpuartStateIsAllocated,
441  s_lpuartStateInstanceMapping,
442  instance,
443  NO_OF_LPUART_INSTS_FOR_UART);
444  /* No permit if all elements in LPUART available resources table are chosen */
445  DEV_ASSERT(index < NO_OF_LPUART_INSTS_FOR_UART);
446  /* Initialize LPUART instance */
447  status = LPUART_DRV_Init(instance->instIdx, (lpuart_state_t*)(&s_lpuartState[index]), &lpuartConfig);
448 
449  /* Install Rx callback */
450  if (config->rxCallback != NULL)
451  {
452  (void)LPUART_DRV_InstallRxCallback(instance->instIdx,
453  config->rxCallback,
454  config->rxCallbackParam);
455  }
456 
457  /* Install Tx callback */
458  if (config->txCallback != NULL)
459  {
460  (void)LPUART_DRV_InstallTxCallback(instance->instIdx,
461  config->txCallback,
462  config->txCallbackParam);
463  }
464  break;
465  #endif
466  /* Define UART PAL over FLEXIO */
467  #if (defined(UART_OVER_FLEXIO))
468  case UART_INST_TYPE_FLEXIO_UART:
469  /*instance index is available */
470  DEV_ASSERT(instance->instIdx < FLEXIO_UART_INSTANCE_COUNT);
471 
472  /* Assign all user configuration to flexio_uart configuration */
473  flexio_uart_user_config_t flexioUartTxConfig;
474  flexio_uart_user_config_t flexioUartRxConfig;
475  UART_GetConfigurationUsingFlexio(config, &flexioUartTxConfig, &flexioUartRxConfig);
476 
477  status = FLEXIO_DRV_InitDevice(0U, &flexioDeviceState);
478  if (status == STATUS_SUCCESS)
479  {
480  /* Allocate one of the Flexio state structure for this instance */
481  index = UART_AllocateState(s_flexioUartStateIsAllocated,
482  s_flexioUartStateInstanceMapping,
483  instance,
484  NO_OF_FLEXIO_INSTS_FOR_UART);
485 
486  /* No permit if all elements in FLEXIO available resources table are chosen */
487  DEV_ASSERT(index < NO_OF_FLEXIO_INSTS_FOR_UART);
488 
489  /* Init FlexIO UART driver for Tx */
490  status = FLEXIO_UART_DRV_Init(0U,
491  &flexioUartTxConfig,
492  (flexio_uart_state_t*)&s_flexioUartTxState[index]);
493  if (status == STATUS_SUCCESS)
494  {
495  /* Init FlexIO UART driver for Rx */
496  status = FLEXIO_UART_DRV_Init(0U,
497  &flexioUartRxConfig,
498  (flexio_uart_state_t*)&s_flexioUartRxState[index]);
499  };
500  }
501  break;
502  #endif
503  /* Define UART PAL over LinFlexD */
504  #if (defined(UART_OVER_LINFLEXD))
505  case UART_INST_TYPE_LINFLEXD:
506  /*instance index is available */
507  DEV_ASSERT(instance->instIdx < LINFlexD_INSTANCE_COUNT);
508 
509  /* Assign all user configuration to linflexd_uart configuration */
510  linflexd_uart_user_config_t linFlexDConfig;
511  UART_GetConfigurationUsingLinFlexD(config, &linFlexDConfig);
512 
513  /* Allocate one of the LinFlexD state structures for this instance */
514  index = UART_AllocateState(s_linFlexDStateIsAllocated,
515  s_linFlexDStateInstanceMapping,
516  instance,
517  NO_OF_LINFLEXD_INSTS_FOR_UART);
518 
519  /* No permit if all elements in LINFLEXD available resources table are chosen */
520  DEV_ASSERT(index < NO_OF_LINFLEXD_INSTS_FOR_UART);
521 
522  /* Initialize LinFlexD instance */
523  status = LINFLEXD_UART_DRV_Init(instance->instIdx, &s_linFlexDState[index], &linFlexDConfig);
524 
525  /* Install Rx callback */
526  if (config->rxCallback != NULL)
527  {
528  (void)LINFLEXD_UART_DRV_InstallRxCallback(instance->instIdx,
529  config->rxCallback,
530  config->rxCallbackParam);
531  }
532 
533  /* Install Tx callback */
534  if (config->txCallback != NULL)
535  {
536  (void)LINFLEXD_UART_DRV_InstallTxCallback(instance->instIdx,
537  config->txCallback,
538  config->txCallbackParam);
539  }
540  break;
541  #endif
542  default:
543  /* Impossible type - do nothing */
544  break;
545  }
546  return status;
547 }

549 /*FUNCTION**********************************************************************
550  *
551  * Function Name : UART_Deinit
552  * Description : De-initializes the UART module
553  *
554  * Implements : UART_Deinit_Activity
555  *END**************************************************************************/
556 status_t UART_Deinit(const uart_instance_t * const instance)
557 {
558  DEV_ASSERT(instance != NULL);
559  status_t status = STATUS_ERROR;
560 #if (defined(UART_OVER_FLEXIO))
561  uint8_t index = 0;
562 #endif
563 
564  switch (instance->instType)
565  {
566  /* Define UART PAL over LPUART */
567  #if (defined(UART_OVER_LPUART))
568  case UART_INST_TYPE_LPUART:
569  DEV_ASSERT(instance->instIdx < LPUART_INSTANCE_COUNT);
570  status = LPUART_DRV_Deinit(instance->instIdx);
571  if (status == STATUS_SUCCESS)
572  {
573  UART_FreeState(s_lpuartStateIsAllocated,
574  s_lpuartStateInstanceMapping,
575  instance,
576  NO_OF_LPUART_INSTS_FOR_UART);
577  }
578  break;
579  #endif
580  /* Define UART PAL over FLEXIO */
581  #if (defined(UART_OVER_FLEXIO))
582  /* Define UART PAL over FLEXIO */
583  case UART_INST_TYPE_FLEXIO_UART:
584  DEV_ASSERT(instance->instIdx < FLEXIO_UART_INSTANCE_COUNT);
585  index = UART_FindFlexioState(instance);
586  /* Return STATUS_BUSY when Tx/Rx line is busy */
587  if ((!s_flexioUartTxState[index].driverIdle) || (!s_flexioUartRxState[index].driverIdle))
588  {
589  status = STATUS_BUSY;
590  }
591  else
592  {
593  (void)FLEXIO_UART_DRV_Deinit(&(s_flexioUartTxState[index]));
594  (void)FLEXIO_UART_DRV_Deinit(&(s_flexioUartRxState[index]));
595  /* Free flexio state array when whole lines are idle */
596  UART_FreeState(s_flexioUartStateIsAllocated,
597  s_flexioUartStateInstanceMapping,
598  instance,
599  NO_OF_FLEXIO_INSTS_FOR_UART);
600  status = STATUS_SUCCESS;
601  }
602  break;
603  #endif
604  /* Define UART PAL over LinFlexD */
605  #if (defined(UART_OVER_LINFLEXD))
606  /* Define UART PAL over LinFlexD */
607  case UART_INST_TYPE_LINFLEXD:
608  DEV_ASSERT(instance->instIdx < LINFlexD_INSTANCE_COUNT);
609  status = LINFLEXD_UART_DRV_Deinit(instance->instIdx);
610  if (status == STATUS_SUCCESS)
611  {
612  UART_FreeState(s_linFlexDStateIsAllocated,
613  s_linFlexDStateInstanceMapping,
614  instance,
615  NO_OF_LINFLEXD_INSTS_FOR_UART);
616  }
617  break;
618  #endif
619  default:
620  /* Impossible type - do nothing */
621  break;
622  }
623  return status;
624 }
