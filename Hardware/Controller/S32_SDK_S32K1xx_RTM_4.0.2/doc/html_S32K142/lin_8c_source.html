77 /*FUNCTION**********************************************************************
78  *
79  * Function Name : lin_lld_init
80  * Description : This function initializes a LIN hardware instance for operation.
81  * This function will initialize the run-time state structure to keep track of
82  * the on-going transfers, initialize the module to user defined settings and
83  * default settings, configure the IRQ state structure and enable the module-level
84  * interrupt to the core, and enable the LIN hardware module transmitter and receiver.
85  *
86  * Implements : lin_lld_init_Activity
87  *END**************************************************************************/
88 l_bool lin_lld_init(l_ifc_handle iii)
89 {
90  DEV_ASSERT((l_u8)iii < LIN_NUM_OF_IFCS);
91 
92  l_u32 lin_hardware_instance;
93  l_u8 index;
94  status_t retVal = STATUS_ERROR;
95  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
96  lin_protocol_state_t * prot_state_ptr = &g_lin_protocol_state_array[iii];
97  static l_u8 lin_lld_response_buffer[LIN_NUM_OF_IFCS][10];
98 
99  prot_state_ptr->response_buffer_ptr = lin_lld_response_buffer[iii];
100  /* Get the LIN hardware instance number */
101  lin_hardware_instance = g_lin_virtual_ifc[iii];
102 
103  /* Initialize time out values */
104  /* If autobaud is enabled, set frame response timeout array and header timeout */
105  /* to maximum values in case the Master baudrate is 2400 */
106  if (prot_user_config_ptr->lin_user_config_ptr->autobaudEnable)
107  {
108  s_lin_max_header_timeout[iii] = lin_calc_max_header_timeout_cnt(2400);
109 
110  for (index = 1U; index < 9U; index++)
111  {
112  s_lin_max_frame_res_timeout_val[iii][index - 1U] = lin_calc_max_res_timeout_cnt(2400, index);
113  }
114  }
115  /* If autobaud is disabled, set frame response timeout array and header timeout */
116  /* to values corresponding to node's configured baudrate */
117  else
118  {
119  s_lin_max_header_timeout[iii] = lin_calc_max_header_timeout_cnt(prot_user_config_ptr->lin_user_config_ptr->baudRate);
120  for (index = 1U; index < 9U; index++)
121  {
122  s_lin_max_frame_res_timeout_val[iii][index - 1U] = lin_calc_max_res_timeout_cnt(prot_user_config_ptr->lin_user_config_ptr->baudRate, index);
123  }
124  }
125 
126  /* Set frame timeout count */
127  prot_state_ptr->frame_timeout_cnt = s_lin_max_frame_res_timeout_val[iii][7U];
128  /* Set idle timeout count */
129  prot_state_ptr->idle_timeout_cnt = prot_user_config_ptr->max_idle_timeout_cnt;
130 
131  /* Initialize the LIN hardware interface */
132  retVal = LIN_DRV_Init(lin_hardware_instance, prot_user_config_ptr->lin_user_config_ptr, &s_lin_state_array[iii]);
133 
134  if (retVal == STATUS_SUCCESS)
135  {
136  /* Install callback function */
137  (void)LIN_DRV_InstallCallback(lin_hardware_instance, CallbackHandler);
138  /* Clear s_baudrate_adjusted_flg[iii] */
139  s_baudrate_adjusted_flg[iii] = false;
140  }
141 
142  return (retVal != STATUS_SUCCESS) ? true : false;
143 }

400 /*FUNCTION**********************************************************************
401  *
402  * Function Name : lin_lld_timeout_service
403  * Description : This is a part of Timer Interrupt Handler.
404  *
405  * Implements : lin_lld_timeout_service_Activity
406  *END**************************************************************************/
407 void lin_lld_timeout_service(l_ifc_handle iii)
408 {
409  DEV_ASSERT((l_u8)iii < LIN_NUM_OF_IFCS);
410  l_u32 lin_hardware_instance = g_lin_virtual_ifc[iii];
411  lin_state_t linState = s_lin_state_array[iii];
412  lin_protocol_state_t * prot_state_ptr = &g_lin_protocol_state_array[iii];
413  l_u8 index;
414 
415  LIN_DRV_TimeoutService(lin_hardware_instance);
416  /* Callback to transport layer to handle N_CR, N_AS or diagnostic interleave timeout */
417  (void)LIN_TL_CALLBACK_HANDLER(iii, TL_TIMEOUT_SERVICE, 0);
418 
419  switch (linState.currentNodeState)
420  {
421  case LIN_NODE_STATE_IDLE:
422  if (!(prot_state_ptr->idle_timeout_cnt-- > 0U))
423  {
424  /* Callback to handle timeout */
425  CALLBACK_HANDLER(iii, LIN_LLD_BUS_ACTIVITY_TIMEOUT, 0xFFU);
426 
427  /* Put current node to Idle state, reset idle timeout count */
428  prot_state_ptr->idle_timeout_cnt = g_lin_protocol_user_cfg_array[iii].max_idle_timeout_cnt;
429 
430  /* Put current node to sleep mode */
431  (void)LIN_DRV_GoToSleepMode(lin_hardware_instance);
432  }
433  /* Re-calculate s_lin_max_header_timeout and s_lin_max_frame_res_timeout_val */
434  if (s_baudrate_adjusted_flg[iii])
435  {
436  /* Clear baudrate adjusted flag */
437  s_baudrate_adjusted_flg[iii] = false;
438  s_lin_max_header_timeout[iii] = lin_calc_max_header_timeout_cnt(g_lin_protocol_user_cfg_array->lin_user_config_ptr->baudRate);
439  for (index = 1U; index < 9U; index++)
440  {
441  s_lin_max_frame_res_timeout_val[iii][index - 1U] = lin_calc_max_res_timeout_cnt(g_lin_protocol_user_cfg_array->lin_user_config_ptr->baudRate, index);
442  }
443  }
444  break;
445  case LIN_NODE_STATE_SEND_BREAK_FIELD:
446  case LIN_NODE_STATE_RECV_SYNC:
447  case LIN_NODE_STATE_RECV_PID:
448  case LIN_NODE_STATE_SEND_PID:
449  /* timeout send has occurred - change state of the node and inform core */
450  if (!(prot_state_ptr->frame_timeout_cnt-- > 0U))
451  {
452  /* Go to idle state */
453  (void)LIN_DRV_GotoIdleState(lin_hardware_instance);
454 
455  /* Reset frame count timeout */
456  prot_state_ptr->frame_timeout_cnt = s_lin_max_frame_res_timeout_val[iii][7U];
457  }
458  break;
459  case LIN_NODE_STATE_SEND_DATA:
460  /* timeout send has occurred - change state of the node and inform core */
461  if (!(prot_state_ptr->frame_timeout_cnt-- > 0U))
462  {
463  /* Abort frame data transferring */
464  (void)LIN_DRV_AbortTransferData(lin_hardware_instance);
465 
466  /* Reset frame count timeout */
467  prot_state_ptr->frame_timeout_cnt = s_lin_max_frame_res_timeout_val[iii][7U];
468  }
469  break;
470  case LIN_NODE_STATE_UNINIT:
471  case LIN_NODE_STATE_SLEEP_MODE:
472  case LIN_NODE_STATE_RECV_DATA:
473  case LIN_NODE_STATE_RECV_DATA_COMPLETED:
474  case LIN_NODE_STATE_SEND_DATA_COMPLETED:
475  default:
476  /* do nothing */
477  break;
478  }
479 }

481 /*FUNCTION**********************************************************************
482  *
483  * Function Name : CallbackHandler
484  * Description : This is the callback function for LIN Driver.
485  *
486  * Implements : CallbackHandler_Activity
487  *END**************************************************************************/
488 static void CallbackHandler(uint32_t instance,
489  void * linState)
490 {
491  const lin_state_t * linCurrentState = linState;
492  lin_lld_event_id_t event_id;
493  l_u8 id = linCurrentState->currentId;
494  l_u8 bytesRemaining = 0U;
495  l_ifc_handle iii = g_lin_hardware_ifc[instance];
496  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
497  lin_protocol_state_t * prot_state_ptr = &g_lin_protocol_state_array[iii];
498 #if (SUPPORT_MASTER_MODE == 1U)
499  lin_master_data_t * master_data_ptr;
500 #endif /* End (SUPPORT_MASTER_MODE == 1U) */
501 
502  switch (linCurrentState->currentEventId)
503  {
504  case LIN_PID_OK:
505  event_id = LIN_LLD_PID_OK;
506  CALLBACK_HANDLER(iii, (event_id), (id));
507  break;
508  case LIN_PID_ERROR:
509  event_id = LIN_LLD_PID_ERR;
510  CALLBACK_HANDLER(iii, (event_id), (id));
511  break;
512  case LIN_TX_COMPLETED:
513  event_id = LIN_LLD_TX_COMPLETED;
514  CALLBACK_HANDLER(iii, (event_id), (id));
515  break;
516  case LIN_RX_COMPLETED:
517  event_id = LIN_LLD_RX_COMPLETED;
518  CALLBACK_HANDLER(iii, (event_id), (id));
519  break;
520  case LIN_CHECKSUM_ERROR:
521  event_id = LIN_LLD_CHECKSUM_ERR;
522  CALLBACK_HANDLER(iii, (event_id), (id));
523  break;
524  case LIN_READBACK_ERROR:
525  event_id = LIN_LLD_READBACK_ERR;
526  CALLBACK_HANDLER(iii, (event_id), (id));
527  break;
528  case LIN_FRAME_ERROR:
529  event_id = LIN_LLD_FRAME_ERR;
530  CALLBACK_HANDLER(iii, (event_id), (id));
531  break;
532  case LIN_RECV_BREAK_FIELD_OK:
533  prot_state_ptr->frame_timeout_cnt = (l_u16)(s_lin_max_frame_res_timeout_val[iii][7U] + s_lin_max_header_timeout[iii]);
534  break;
535  case LIN_SYNC_ERROR:
536  #if (SUPPORT_PROTOCOL_J2602 == 1U)
537  if (prot_user_config_ptr->protocol_version == LIN_PROTOCOL_J2602)
538  {
539  event_id = LIN_LLD_READBACK_ERR;
540  CALLBACK_HANDLER(iii, (event_id), (id));
541  }
542  #endif /* End of (SUPPORT_PROTOCOL_J2602 == 1U) */
543 
544  prot_state_ptr->frame_timeout_cnt = s_lin_max_frame_res_timeout_val[iii][7U];
545  break;
546  case LIN_BAUDRATE_ADJUSTED:
547  /* Set s_baudrate_adjusted_flg[iii] */
548  s_baudrate_adjusted_flg[iii] = true;
549  break;
550  case LIN_NO_EVENT:
551  if (linCurrentState->timeoutCounterFlag == (bool)1U)
552  {
553  #if (SUPPORT_MASTER_MODE == 1U)
554  if ((prot_state_ptr->diagnostic_mode == DIAG_INTERLEAVE_MODE) && (prot_state_ptr->current_id == 0x3DU))
555  {
556  master_data_ptr = &g_lin_master_data_array[g_lin_protocol_user_cfg_array[iii].master_ifc_handle];
557  master_data_ptr->active_schedule_id = master_data_ptr->previous_schedule_id;
558  master_data_ptr->schedule_start_entry_ptr[master_data_ptr->active_schedule_id] = 0U;
559  }
560  else
561  #endif /* End (SUPPORT_MASTER_MODE == 1U) */
562  {
563  (void)LIN_DRV_GetReceiveStatus(instance, &bytesRemaining);
564  /* Received part of data but not completed */
565  if (linCurrentState->rxSize > bytesRemaining)
566  {
567  event_id = LIN_LLD_NODATA_TIMEOUT;
568  CALLBACK_HANDLER(iii, (event_id), (id));
569  }
570  }
571  }
572  break;
573  default:
574  /* do nothing */
575  break;
576  }
577 
578  prot_state_ptr->idle_timeout_cnt = prot_user_config_ptr->max_idle_timeout_cnt;
579  /*
580  * @violates MISRA 2012 Advisory Rule 8.13 Advisory
581  */
582 }

lin_lld_timeout_service
void lin_lld_timeout_service(l_ifc_handle iii)
Callback function for Timer Interrupt Handler In timer IRQ handler, call this function. Used to check if frame timeout has occurred during frame data transmission and reception, to check for N_As and N_Cr timeout for LIN 2.1 and above. This function also check if there is no LIN bus communication (no headers and no frame data transferring) for Idle timeout (s), then put LIN node to Sleep mode. Users may initialize a timer (for example FTM)with period of Timeout unit (default: 500 micro seconds) to call lin_lld_timeout_service(). For an interface iii, Idle timeout (s) = max_idle_timeout_cnt * Timeout unit (us) frame timeout (us) = frame_timeout_cnt * Timeout unit (us) N_As timeout (us) = N_As_timeout * Timeout unit (us) N_Cr timeout (us) = N_Cr_timeout * Timeout unit (us)

LIN_DRV_SendFrameData
status_t LIN_DRV_SendFrameData(uint32_t instance, const uint8_t *txBuff, uint8_t txSize)
Sends frame data out through the LIN Hardware Interface using non-blocking method. This enables an a-sync method for transmitting data. Non-blocking means that the function returns immediately. The application has to get the transmit status to know when the transmit is complete. This function will calculate the checksum byte and send it with the frame data. The function will return immediately after calling this function. If txSize is equal to 0 or greater than 8 or node's current state is in SLEEP mode then the function will return STATUS_ERROR. If isBusBusy is currently true then the function will return LIN_BUS_BUSY.
