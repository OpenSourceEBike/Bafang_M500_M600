123 status_t EDMA_DRV_Init(edma_state_t *edmaState,
124  const edma_user_config_t *userConfig,
125  edma_chn_state_t * const chnStateArray[],
126  const edma_channel_config_t * const chnConfigArray[],
127  uint32_t chnCount)
128 {
129  uint32_t index = 0U;
130  DMA_Type *edmaRegBase = NULL;
131  IRQn_Type irqNumber = NotAvail_IRQn;
132  status_t edmaStatus = STATUS_SUCCESS;
133  status_t chnInitStatus = STATUS_SUCCESS;
134 #if defined (CUSTOM_DEVASSERT) || defined (DEV_ERROR_DETECT)
135  uint32_t freq = 0U;
136  status_t clockManagerStatus = STATUS_SUCCESS;
137 #endif
138 
139  /* Check the state and configuration structure pointers are valid */
140  DEV_ASSERT((edmaState != NULL) && (userConfig != NULL));
141 
142  /* Check the module has not already been initialized */
143  DEV_ASSERT(s_virtEdmaState == NULL);
144 
145 #if defined (CUSTOM_DEVASSERT) || defined (DEV_ERROR_DETECT)
146  /* Check that eDMA and DMAMUX modules are clock gated on */
147  for (index = 0U; index < (uint32_t)DMA_INSTANCE_COUNT; index++)
148  {
149  clockManagerStatus = CLOCK_SYS_GetFreq(s_edmaClockNames[index], &freq);
150  DEV_ASSERT(clockManagerStatus == STATUS_SUCCESS);
151  }
152 #ifdef FEATURE_DMAMUX_AVAILABLE
153  for (index = 0U; index < (uint32_t)DMAMUX_INSTANCE_COUNT; index++)
154  {
155  clockManagerStatus = CLOCK_SYS_GetFreq(s_dmamuxClockNames[index], &freq);
156  DEV_ASSERT(clockManagerStatus == STATUS_SUCCESS);
157  }
158 #endif /* FEATURE_DMAMUX_AVAILABLE */
159 #endif /* (CUSTOM_DEVASSERT) || defined (DEV_ERROR_DETECT) */
160 
161  /* Save the runtime state structure for the driver */
162  s_virtEdmaState = edmaState;
163 
164  /* Clear the state structure. */
165  EDMA_DRV_ClearStructure((uint8_t *)s_virtEdmaState, sizeof(edma_state_t));
166 
167  /* Init all DMA instances */
168  for(index = 0U; index < (uint32_t)DMA_INSTANCE_COUNT; index++)
169  {
170  edmaRegBase = s_edmaBase[index];
171 
172  /* Init eDMA module on hardware level. */
173  EDMA_Init(edmaRegBase);
174 
175 #ifdef FEATURE_DMA_HWV3
176  /* Set arbitration mode */
177  EDMA_SetChannelArbitrationMode(edmaRegBase, userConfig->chnArbitration);
178 #else /* FEATURE_DMA_HWV3 */
179  /* Set arbitration mode */
180  EDMA_SetChannelArbitrationMode(edmaRegBase, userConfig->chnArbitration);
181 #if (FEATURE_DMA_CHANNEL_GROUP_COUNT > 0x1U)
182  EDMA_SetGroupArbitrationMode(edmaRegBase, userConfig->groupArbitration);
183  EDMA_SetGroupPriority(edmaRegBase, userConfig->groupPriority);
184 #endif /* (FEATURE_DMA_CHANNEL_GROUP_COUNT > 0x1U) */
185 #endif /* FEATURE_DMA_HWV3 */
186  /* Set 'Halt on error' configuration */
187  EDMA_SetHaltOnErrorCmd(edmaRegBase, userConfig->haltOnError);
188  }
189 
190 #if defined FEATURE_DMA_HAS_ERROR_IRQ
191  /* Enable the error interrupts for eDMA module. */
192  for (index = 0U; index < (uint32_t)FEATURE_DMA_VIRTUAL_ERROR_INTERRUPT_LINES; index++)
193  {
194  /* Enable channel interrupt ID. */
195  irqNumber = s_edmaErrIrqId[index];
196  INT_SYS_EnableIRQ(irqNumber);
197  }
198 #endif
199 
200  /* Register all edma channel interrupt handlers into vector table. */
201  for (index = 0U; index < (uint32_t)FEATURE_DMA_VIRTUAL_CHANNELS_INTERRUPT_LINES; index++)
202  {
203  /* Enable channel interrupt ID. */
204  irqNumber = s_edmaIrqId[index];
205  INT_SYS_EnableIRQ(irqNumber);
206  }
207 
208 #ifdef FEATURE_DMAMUX_AVAILABLE
209  /* Initialize all DMAMUX instances */
210  for (index = 0U; index < (uint32_t)DMAMUX_INSTANCE_COUNT; index++)
211  {
212  DMAMUX_Init(s_dmaMuxBase[index]);
213  }
214 #endif
215 
216  /* Initialize the channels based on configuration list */
217  if ((chnStateArray != NULL) && (chnConfigArray != NULL))
218  {
219  for (index = 0U; index < chnCount; index++)
220  {
221  chnInitStatus = EDMA_DRV_ChannelInit(chnStateArray[index], chnConfigArray[index]);
222  if (chnInitStatus != STATUS_SUCCESS)
223  {
224  edmaStatus = chnInitStatus;
225  }
226  }
227  }
228 
229  return edmaStatus;
230 }

287 status_t EDMA_DRV_ChannelInit(edma_chn_state_t *edmaChannelState,
288  const edma_channel_config_t *edmaChannelConfig)
289 {
290  /* Check the state and configuration structure pointers are valid */
291  DEV_ASSERT((edmaChannelState != NULL) && (edmaChannelConfig != NULL));
292 
293  /* Check if the module is initialized */
294  DEV_ASSERT(s_virtEdmaState != NULL);
295 
296  /* Check the channel has not already been allocated */
297  DEV_ASSERT(s_virtEdmaState->virtChnState[edmaChannelConfig->virtChnConfig] == NULL);
298 
299  /* Check if the channel defined by user in the channel configuration structure is valid */
300  DEV_ASSERT(edmaChannelConfig->virtChnConfig < FEATURE_DMA_VIRTUAL_CHANNELS);
301 
302  /* Get DMA instance from virtual channel */
303  uint8_t dmaInstance = (uint8_t)FEATURE_DMA_VCH_TO_INSTANCE(edmaChannelConfig->virtChnConfig);
304 
305  /* Get DMA channel from virtual channel */
306  uint8_t dmaChannel = (uint8_t)FEATURE_DMA_VCH_TO_CH(edmaChannelConfig->virtChnConfig);
307 
308  /* Get virtual channel value */
309  uint8_t virtualChannel = edmaChannelConfig->virtChnConfig;
310 
311  /* Get status */
312  status_t retStatus = STATUS_SUCCESS;
313 
314  /* Load corresponding DMA instance pointer */
315  DMA_Type *edmaRegBase = s_edmaBase[dmaInstance];
316 
317  /* Reset the channel state structure to default value. */
318  EDMA_DRV_ClearStructure((uint8_t *)edmaChannelState, sizeof(edma_chn_state_t));
319 
320 #ifdef FEATURE_DMAMUX_AVAILABLE
321  retStatus = EDMA_DRV_SetChannelRequestAndTrigger(edmaChannelConfig->virtChnConfig, (uint8_t)edmaChannelConfig->source, edmaChannelConfig->enableTrigger);
322 #endif
323 
324  /* Clear the TCD registers for this channel */
325  EDMA_TCDClearReg(edmaRegBase, dmaChannel);
326 
327 #ifdef FEATURE_DMAMUX_AVAILABLE
328  if (retStatus == STATUS_SUCCESS)
329 #endif
330  {
331  /* Set virtual channel state */
332  s_virtEdmaState->virtChnState[virtualChannel] = edmaChannelState;
333  /* Set virtual channel value */
334  s_virtEdmaState->virtChnState[virtualChannel]->virtChn = virtualChannel;
335  /* Set virtual channel status to normal */
336  s_virtEdmaState->virtChnState[virtualChannel]->status = EDMA_CHN_NORMAL;
337 
338  /* Enable error interrupt for this channel */
339  EDMA_SetErrorIntCmd(edmaRegBase, dmaChannel, true);
340 #ifdef FEATURE_DMA_HWV3
341  /* Put the channel in a priority group, as defined in configuration */
342  EDMA_SetChannelPriorityGroup(edmaRegBase, dmaChannel, edmaChannelConfig->groupPriority);
343 #endif
344 
345  /* Set the channel priority, as defined in the configuration, only if fixed arbitration mode is selected */
346  if ((EDMA_GetChannelArbitrationMode(edmaRegBase) == EDMA_ARBITRATION_FIXED_PRIORITY) &&
347  (edmaChannelConfig->channelPriority != EDMA_CHN_DEFAULT_PRIORITY))
348  {
349  EDMA_SetChannelPriority(edmaRegBase, dmaChannel, edmaChannelConfig->channelPriority);
350  }
351  /* Install the user callback */
352  retStatus = EDMA_DRV_InstallCallback(edmaChannelConfig->virtChnConfig, edmaChannelConfig->callback, edmaChannelConfig->callbackParam);
353  }
354 
355  return retStatus;
356 }

502 /*FUNCTION**********************************************************************
503  *
504  * Function Name : EDMA_DRV_ConfigSingleBlockTransfer
505  * Description : Configures a DMA single block transfer.
506  *
507  * Implements : EDMA_DRV_ConfigSingleBlockTransfer_Activity
508  *END**************************************************************************/
509 status_t EDMA_DRV_ConfigSingleBlockTransfer(uint8_t virtualChannel,
510  edma_transfer_type_t type,
511  uint32_t srcAddr,
512  uint32_t destAddr,
513  edma_transfer_size_t transferSize,
514  uint32_t dataBufferSize)
515 {
516  /* Check that virtual channel number is valid */
517  DEV_ASSERT(virtualChannel < FEATURE_DMA_VIRTUAL_CHANNELS);
518 
519  /* Check that eDMA module is initialized */
520  DEV_ASSERT(s_virtEdmaState != NULL);
521 
522  /* Check that virtual channel is initialized */
523  DEV_ASSERT(s_virtEdmaState->virtChnState[virtualChannel] != NULL);
524 
525 #if defined (CUSTOM_DEVASSERT) || defined (DEV_ERROR_DETECT)
526  /* Check if the value passed for 'transferSize' is valid */
527  DEV_ASSERT(EDMA_DRV_ValidTransferSize(transferSize));
528 #endif
529 
530  /* Get DMA instance from virtual channel */
531  uint8_t dmaInstance = (uint8_t)FEATURE_DMA_VCH_TO_INSTANCE(virtualChannel);
532 
533  /* Get DMA channel from virtual channel*/
534  uint8_t dmaChannel = (uint8_t)FEATURE_DMA_VCH_TO_CH(virtualChannel);
535 
536  DMA_Type *edmaRegBase = s_edmaBase[dmaInstance];
537  status_t retStatus = STATUS_SUCCESS;
538 
539  /* Compute the transfer offset, based on transfer size.
540  * The number of bytes transferred in each source read/destination write
541  * is obtained with the following formula:
542  * source_read_size = 2^SSIZE
543  * destination_write_size = 2^DSIZE
544  */
545  uint8_t transferOffset = (uint8_t) (1U << ((uint8_t)transferSize));
546 
547  /* Check that source address and destination address are valid */
548  DEV_ASSERT((srcAddr % transferOffset) == 0U);
549  DEV_ASSERT((destAddr % transferOffset) == 0U);
550 
551  /* The number of bytes to be transferred (buffer size) must
552  * be a multiple of the source read/destination write size
553  */
554  if ((dataBufferSize % transferOffset) != 0U)
555  {
556  retStatus = STATUS_ERROR;
557  }
558 
559  if (retStatus == STATUS_SUCCESS)
560  {
561  /* Clear transfer control descriptor for the current channel */
562  EDMA_TCDClearReg(edmaRegBase, dmaChannel);
563 
564 #ifdef FEATURE_DMA_ENGINE_STALL
565  /* Configure the DMA Engine to stall for a number of cycles after each R/W */
566  EDMA_TCDSetEngineStall(edmaRegBase, dmaChannel, EDMA_ENGINE_STALL_4_CYCLES);
567 #endif
568 
569 #ifdef FEATURE_DMA_HWV3
570  EDMA_SetMinorLoopMappingCmd(edmaRegBase, dmaChannel, false);
571 #else
572  EDMA_SetMinorLoopMappingCmd(edmaRegBase, true);
573 #endif
574 
575  /* Configure source and destination addresses */
576  EDMA_TCDSetSrcAddr(edmaRegBase, dmaChannel, srcAddr);
577  EDMA_TCDSetDestAddr(edmaRegBase, dmaChannel, destAddr);
578 
579  /* Set transfer size (1B/2B/4B/16B/32B) */
580  EDMA_TCDSetAttribute(edmaRegBase, dmaChannel, EDMA_MODULO_OFF, EDMA_MODULO_OFF, transferSize, transferSize);
581 
582  /* Configure source/destination offset. */
583  switch (type)
584  {
585  case EDMA_TRANSFER_PERIPH2MEM:
586  EDMA_TCDSetSrcOffset(edmaRegBase, dmaChannel, 0);
587  EDMA_TCDSetDestOffset(edmaRegBase, dmaChannel, (int8_t) transferOffset);
588  break;
589  case EDMA_TRANSFER_MEM2PERIPH:
590  EDMA_TCDSetSrcOffset(edmaRegBase, dmaChannel, (int8_t) transferOffset);
591  EDMA_TCDSetDestOffset(edmaRegBase, dmaChannel, 0);
592  break;
593  case EDMA_TRANSFER_MEM2MEM:
594  EDMA_TCDSetSrcOffset(edmaRegBase, dmaChannel, (int8_t) transferOffset);
595  EDMA_TCDSetDestOffset(edmaRegBase, dmaChannel, (int8_t) transferOffset);
596  break;
597  case EDMA_TRANSFER_PERIPH2PERIPH:
598  EDMA_TCDSetSrcOffset(edmaRegBase, dmaChannel, 0);
599  EDMA_TCDSetDestOffset(edmaRegBase, dmaChannel, 0);
600  break;
601  default:
602  /* This should never be reached - all the possible values have been handled. */
603  break;
604  }
605 
606  /* Set the total number of bytes to be transfered */
607  EDMA_TCDSetNbytes(edmaRegBase, dmaChannel, dataBufferSize);
608 
609  /* Set major iteration count to 1 (single block mode) */
610  EDMA_TCDSetMajorCount(edmaRegBase, dmaChannel, 1U);
611 
612  /* Enable interrupt when the transfer completes */
613  EDMA_TCDSetMajorCompleteIntCmd(edmaRegBase, dmaChannel, true);
614 
615  /* Set virtual channel status to normal */
616  s_virtEdmaState->virtChnState[virtualChannel]->status = EDMA_CHN_NORMAL;
617  }
618 
619  return retStatus;
620 }

755  * Description : Configure eDMA for scatter/gather operation
756  *

759 status_t EDMA_DRV_ConfigScatterGatherTransfer(uint8_t virtualChannel,

761  edma_transfer_size_t transferSize,
762  uint32_t bytesOnEachRequest,
763  const edma_scatter_gather_list_t *srcList,

777  DEV_ASSERT((stcd != NULL) && (srcList != NULL) && (destList != NULL));
778 
779 #if defined (CUSTOM_DEVASSERT) || defined (DEV_ERROR_DETECT)
780  /* Check if the value passed for 'transferSize' is valid */
781  DEV_ASSERT(EDMA_DRV_ValidTransferSize(transferSize));
782 #endif
783 
784  uint8_t i = 0U;
785  uint16_t transferOffset = 0U;
786  uint32_t stcdAlignedAddr = STCD_ADDR(stcd);
787  edma_software_tcd_t *edmaSwTcdAddr = (edma_software_tcd_t *)stcdAlignedAddr;
788  edma_loop_transfer_config_t edmaLoopConfig;
789  edma_transfer_config_t edmaTransferConfig;
790  status_t retStatus = STATUS_SUCCESS;
791 
792  /* Set virtual channel status to normal */
793  s_virtEdmaState->virtChnState[virtualChannel]->status = EDMA_CHN_NORMAL;
794 
795  /* Compute the transfer offset, based on transfer size.
796  * The number of bytes transferred in each source read/destination write
797  * is obtained with the following formula:
798  * source_read_size = 2^SSIZE
799  * destination_write_size = 2^DSIZE
800  */
801  transferOffset = (uint16_t) (1UL << ((uint16_t)transferSize));
802 
803  /* The number of bytes to be transferred on each request must
804  * be a multiple of the source read/destination write size
805  */
806  if ((bytesOnEachRequest % transferOffset) != 0U)
807  {
808  retStatus = STATUS_ERROR;
809  }
810 
811  /* Clear the configuration structures before initializing them. */
812  EDMA_DRV_ClearStructure((uint8_t *)(&edmaTransferConfig), sizeof(edma_transfer_config_t));
813  EDMA_DRV_ClearStructure((uint8_t *)(&edmaLoopConfig), sizeof(edma_loop_transfer_config_t));
814 
815  /* Configure the transfer for scatter/gather mode. */
816  edmaTransferConfig.srcLastAddrAdjust = 0;
817  edmaTransferConfig.destLastAddrAdjust = 0;
818  edmaTransferConfig.srcModulo = EDMA_MODULO_OFF;
819  edmaTransferConfig.destModulo = EDMA_MODULO_OFF;
820  edmaTransferConfig.srcTransferSize = transferSize;
821  edmaTransferConfig.destTransferSize = transferSize;
822  edmaTransferConfig.minorByteTransferCount = bytesOnEachRequest;
823  edmaTransferConfig.interruptEnable = true;
824  edmaTransferConfig.scatterGatherEnable = true;
825  edmaTransferConfig.loopTransferConfig = &edmaLoopConfig;
826  edmaTransferConfig.loopTransferConfig->srcOffsetEnable = false;
827  edmaTransferConfig.loopTransferConfig->dstOffsetEnable = false;
828  edmaTransferConfig.loopTransferConfig->minorLoopChnLinkEnable = false;
829  edmaTransferConfig.loopTransferConfig->majorLoopChnLinkEnable = false;
830 
831  /* Copy scatter/gather lists to transfer configuration*/
832  for (i = 0U; (i < tcdCount) && (retStatus == STATUS_SUCCESS); i++)
833  {
834  edmaTransferConfig.srcAddr = srcList[i].address;
835  edmaTransferConfig.destAddr = destList[i].address;
836  if ((srcList[i].length != destList[i].length) || (srcList[i].type != destList[i].type))
837  {
838  retStatus = STATUS_ERROR;
839  break;
840  }
841  edmaTransferConfig.loopTransferConfig->majorLoopIterationCount = srcList[i].length/bytesOnEachRequest;
842 
843  switch (srcList[i].type)
844  {
845  case EDMA_TRANSFER_PERIPH2MEM:
846  /* Configure Source Read. */
847  edmaTransferConfig.srcOffset = 0;
848  /* Configure Dest Write. */
849  edmaTransferConfig.destOffset = (int16_t) transferOffset;
850  break;
851  case EDMA_TRANSFER_MEM2PERIPH:
852  /* Configure Source Read. */
853  edmaTransferConfig.srcOffset = (int16_t) transferOffset;
854  /* Configure Dest Write. */
855  edmaTransferConfig.destOffset = 0;
856  break;
857  case EDMA_TRANSFER_MEM2MEM:
858  /* Configure Source Read. */
859  edmaTransferConfig.srcOffset = (int16_t) transferOffset;
860  /* Configure Dest Write. */
861  edmaTransferConfig.destOffset = (int16_t) transferOffset;
862  break;
863  case EDMA_TRANSFER_PERIPH2PERIPH:
864  /* Configure Source Read. */
865  edmaTransferConfig.srcOffset = 0;
866  /* Configure Dest Write. */
867  edmaTransferConfig.destOffset = 0;
868  break;
869  default:
870  /* This should never be reached - all the possible values have been handled. */
871  break;
872  }
873 
874  /* Configure the pointer to next software TCD structure; for the last one, this address should be 0 */
875  if (i == ((uint8_t)(tcdCount - 1U)))
876  {
877  edmaTransferConfig.scatterGatherNextDescAddr = 0U;
878  }
879  else
880  {
881  edma_software_tcd_t * ptNextAddr = &edmaSwTcdAddr[i];
882  edmaTransferConfig.scatterGatherNextDescAddr = ((uint32_t) ptNextAddr);
883  }
884 
885  if (i == 0U)
886  {
887  /* Push the configuration for the first descriptor to registers */
888  EDMA_DRV_PushConfigToReg(virtualChannel, &edmaTransferConfig);
889  }
890  else
891  {
892  /* Copy configuration to software TCD structure */
893  EDMA_DRV_PushConfigToSTCD(&edmaTransferConfig, &edmaSwTcdAddr[i - 1U]);
894  }
895  }
896 
897  return retStatus;
898 }
