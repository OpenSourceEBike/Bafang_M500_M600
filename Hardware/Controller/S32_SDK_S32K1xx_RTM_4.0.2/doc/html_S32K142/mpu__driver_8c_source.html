49 /*FUNCTION**********************************************************************
50  *
51  * Function Name : MPU_DRV_Init
52  * Description : Initializes the memory protection unit by setting the access
53  * configurations of all available masters, process identifier and the memory
54  * location for the given regions; and activate module finally.
55  * Please note that access rights for region 0 will always be configured and
56  * regionCnt takes values between 1 and the maximum region count supported by
57  * the hardware. E.g. In S32K144 the number of supported regions is 8.
58  * The user must make sure that the clock is enabled.
59  *
60  * Implements : MPU_DRV_Init_Activity
61  *END**************************************************************************/
62 status_t MPU_DRV_Init(uint32_t instance,
63  uint8_t regionCnt,
64  const mpu_user_config_t * userConfigArr)
65 {
66  DEV_ASSERT(instance < MPU_INSTANCE_COUNT);
67  DEV_ASSERT((regionCnt > 0U) && (regionCnt <= MPU_RGD_COUNT));
68  DEV_ASSERT(userConfigArr != NULL);
69 
70  MPU_Type * base = s_mpuBase[instance];
71  status_t retStatus = STATUS_SUCCESS;
72  uint8_t regionNum = 0U;
73 
74  /* De-initializes all region descriptor */
75  MPU_DRV_Deinit(instance);
76 
77  for (regionNum = 0U; regionNum < regionCnt; regionNum++)
78  {
79  /* Sets region configuration */
80  retStatus = MPU_DRV_SetRegionConfig(instance, regionNum, &userConfigArr[regionNum]);
81 
82  if (retStatus != STATUS_SUCCESS)
83  {
84  break;
85  }
86  }
87 
88  if (retStatus == STATUS_SUCCESS)
89  {
90  /* Enables the MPU module operation */
91  MPU_Enable(base, true);
92  }
93 
94  return retStatus;
95 }

97 /*FUNCTION**********************************************************************
98  *
99  * Function Name : MPU_DRV_Deinit
100  * Description : De-initializes the memory protection unit by reseting
101  * all regions to default and disable module.
102  *
103  * Implements : MPU_DRV_Deinit_Activity
104  *END**************************************************************************/
105 void MPU_DRV_Deinit(uint32_t instance)
106 {
107  DEV_ASSERT(instance < MPU_INSTANCE_COUNT);
108 
109  MPU_Type * base = s_mpuBase[instance];
110  uint8_t regionNum = 0U;
111  uint8_t slaveNum = 0U;
112 
113  /* Disables the entire MPU module */
114  MPU_Enable(base, false);
115 
116  for (slaveNum = 0U; slaveNum < FEATURE_MPU_SLAVE_COUNT; slaveNum++)
117  {
118  /* Clears slave error flag */
119  MPU_ClearErrorFlag(base, slaveNum);
120  }
121 
122  /* De-initializes all region descriptors */
123  for (regionNum = 0U; regionNum < MPU_RGD_COUNT; regionNum++)
124  {
125  /* Resets the region configuration to default */
126  MPU_InitRegion(base, regionNum);
127  }
128 }

155 /*FUNCTION**********************************************************************
156  *
157  * Function Name : MPU_DRV_SetRegionConfig
158  * Description : Sets the region configuration.
159  * Updates the access configuration of all available masters,
160  * process identifier and memory location in a given region.
161  *
162  * Implements : MPU_DRV_SetRegionConfig_Activity
163  *END**************************************************************************/
164 status_t MPU_DRV_SetRegionConfig(uint32_t instance,
165  uint8_t regionNum,
166  const mpu_user_config_t * userConfigPtr)
167 {
168  DEV_ASSERT(instance < MPU_INSTANCE_COUNT);
169  DEV_ASSERT(regionNum < MPU_RGD_COUNT);
170  DEV_ASSERT(userConfigPtr != NULL);
171  DEV_ASSERT(userConfigPtr->startAddr <= userConfigPtr->endAddr);
172 
173  MPU_Type * base = s_mpuBase[instance];
174  uint8_t masterIdx = 0U;
175  status_t returnCode = STATUS_SUCCESS;
176 
177  /* Sets access right for masters */
178  for (masterIdx = 0U; masterIdx < FEATURE_MPU_MASTER_COUNT; masterIdx++)
179  {
180  if (userConfigPtr->masterAccRight[masterIdx].masterNum <= FEATURE_MPU_MAX_HIGH_MASTER_NUMBER)
181  {
182  MPU_SetMasterAccessRight(base,
183  regionNum,
184  &userConfigPtr->masterAccRight[masterIdx]);
185  }
186  else
187  {
188  returnCode = STATUS_ERROR;
189  break;
190  }
191  }
192 
193  /* Configures address, process identifier except region 0 */
194  if (regionNum > 0U)
195  {
196  if (returnCode == STATUS_SUCCESS)
197  {
198  /* Sets a region's start and end addresses */
199  MPU_SetRegionAddr(base, regionNum, userConfigPtr->startAddr, userConfigPtr->endAddr);
200 
201  #if FEATURE_MPU_HAS_PROCESS_IDENTIFIER
202  /* Sets process identifier */
203  MPU_SetProcessIdentifier(base, regionNum, userConfigPtr->processIdentifier);
204 
205  /* Sets process identifier mask */
206  MPU_SetProcessIdentifierMask(base, regionNum, userConfigPtr->processIdMask);
207  #endif
208 
209  /* Enables the region descriptor valid bit */
210  MPU_SetRegionValidCmd(base, regionNum, true);
211  }
212  }
213 
214  return returnCode;
215 }

248 /*FUNCTION**********************************************************************
249  *
250  * Function Name : MPU_DRV_GetDetailErrorAccessInfo
251  * Description : Checks and gets the MPU access error detail information
252  * for a slave port. Clears bus error flag if an error occurs.
253  *
254  * Implements : MPU_DRV_GetDetailErrorAccessInfo_Activity
255  *END**************************************************************************/
256 bool MPU_DRV_GetDetailErrorAccessInfo(uint32_t instance,
257  uint8_t slavePortNum,
258  mpu_access_err_info_t * errInfoPtr)
259 {
260  DEV_ASSERT(instance < MPU_INSTANCE_COUNT);
261  DEV_ASSERT(slavePortNum < FEATURE_MPU_SLAVE_COUNT);
262  DEV_ASSERT(errInfoPtr != NULL);
263 
264  MPU_Type * base = s_mpuBase[instance];
265  bool errorStatus = false;
266  uint8_t slaveNum = slavePortNum;
267 
268 #ifdef ERRATA_E11109
269  if (slavePortNum == FEATURE_MPU_SLAVE_QUADSPI)
270  {
271  /* The access violation will be captured on slave port 3 */
272  slaveNum = 3U;
273  }
274 #endif
275 
276  /* Gets slaver port error status */
277  errorStatus = MPU_GetErrorStatus(base, slaveNum);
278 
279  /* Check if there is access violation in the slave port */
280  if (errorStatus)
281  {
282  /* Get the slave port detail error */
283  MPU_GetErrorInfo(base, slaveNum, errInfoPtr);
284 
285  /* Clears slave port error flag */
286  MPU_ClearErrorFlag(base, slaveNum);
287 
288  #ifdef ERRATA_E11109
289  if (slavePortNum == FEATURE_MPU_SLAVE_QUADSPI)
290  {
291  /* Clears slave port error flag */
292  MPU_ClearErrorFlag(base, slavePortNum);
293  }
294  #endif
295  }
296 
297  return errorStatus;
298 }

300 /*FUNCTION**********************************************************************
301  *
302  * Function Name : MPU_DRV_GetDefaultRegionConfig
303  * Description : Gets default region configuration. Grants all access rights
304  * for masters and disables PID.
305  *
306  * Implements : MPU_DRV_GetDefaultRegionConfig_Activity
307  *END**************************************************************************/
308 mpu_user_config_t MPU_DRV_GetDefaultRegionConfig(mpu_master_access_right_t * masterAccRight)
309 {
310  DEV_ASSERT(masterAccRight != NULL);
311 
312  uint8_t masterIdx = 0U;
313  uint8_t masterNum[FEATURE_MPU_MASTER_COUNT] = FEATURE_MPU_MASTER;
314  mpu_user_config_t regionConfig;
315 
316  /* Allocates entire memory */
317  regionConfig.startAddr = 0x0U;
318  regionConfig.endAddr = 0xFFFFFFFFU;
319 
320  /* All access rights of masters are allowed */
321  for (masterIdx = 0U; masterIdx < FEATURE_MPU_MASTER_COUNT; masterIdx++)
322  {
323  MPU_GetDefaultMasterAccRight(masterNum[masterIdx], &masterAccRight[masterIdx]);
324  }
325  regionConfig.masterAccRight = masterAccRight;
326 
327 #if FEATURE_MPU_HAS_PROCESS_IDENTIFIER
328  /* Disables process identifier */
329  regionConfig.processIdentifier = 0U;
330  regionConfig.processIdMask = 0U;
331 #endif
332 
333  return regionConfig;
334 }
