49 /*FUNCTION*********************************************************************
50  *
51  * Function Name : PDB_DRV_Init
52  * Description : This function initializes the PDB counter, input triggers
53  * and general pre-trigger settings.
54  * It resets PDB registers and enables the clock for PDB. So it should be
55  * called before any operation to PDB module. After initialized, the PDB can
56  * act as a triggered timer, which lays the foundation for other features in
57  * PDB module.
58  *
59  * Implements : PDB_DRV_Init_Activity
60  *END*************************************************************************/
61 void PDB_DRV_Init(const uint32_t instance,
62  const pdb_timer_config_t * const userConfigPtr)
63 {
64  DEV_ASSERT(instance < PDB_INSTANCE_COUNT);
65  DEV_ASSERT(userConfigPtr != NULL);
66  /* Table of PDB clock names as defined in clock manager. */
67 #if (PDB_INSTANCE_COUNT == 1U)
68  static const clock_names_t s_pdbClkNames[PDB_INSTANCE_COUNT] = {PDB0_CLK};
69 #elif (PDB_INSTANCE_COUNT == 2U)
70  static const clock_names_t s_pdbClkNames[PDB_INSTANCE_COUNT] = {PDB0_CLK, PDB1_CLK};
71 #endif
72  status_t clkStatus;
73 
74  PDB_Type * base = s_pdbBase[instance];
75 
76  clock_names_t instanceClkName = s_pdbClkNames[instance];
77 
78  /* Get the PDB clock as configured in the clock manager */
79  clkStatus = CLOCK_SYS_GetFreq(instanceClkName, NULL);
80 
81  /* Exit if clock is not enabled for current instance. */
82  DEV_ASSERT(clkStatus == STATUS_SUCCESS);
83  (void)clkStatus;
84 
85  /* Reset the registers for PDB module to reset state. */
86  PDB_Init(base);
87  PDB_ConfigTimer(base, userConfigPtr);
88 
89 #if FEATURE_PDB_HAS_INSTANCE_BACKTOBACK
90  /* Note: this configuration is common between PDB instances 0 and 1 (configures the same register even if configured for either PDB instance)
91  Enabling when configuring PDB0, then disabling when configuring PDB1, will result in being disabled. */
92  if (userConfigPtr->instanceBackToBackEnable)
93  {
94  PDB_SIM_EnableInstanceBackToBack();
95  }
96  else
97  {
98  PDB_SIM_DisableInstanceBackToBack();
99  }
100 #endif /* FEATURE_PDB_HAS_INSTANCE_BACKTOBACK */
101 
102 #if FEATURE_PDB_HAS_INTERCHANNEL_BACKTOBACK
103  if (userConfigPtr->interchannelBackToBackEnable)
104  {
105  PDB_SIM_EnableInterchannelBackToBack(instance);
106  }
107  else
108  {
109  PDB_SIM_DisableInterchannelBackToBack(instance);
110  }
111 #endif /* FEATURE_PDB_HAS_INTERCHANNEL_BACKTOBACK */
112 
113  /* Configure NVIC. */
114  if (userConfigPtr->intEnable || userConfigPtr->seqErrIntEnable)
115  {
116  INT_SYS_EnableIRQ(s_pdbIrqId[instance]); /* Enable PDB interrupt in NVIC level.*/
117  }
118  else
119  {
120  INT_SYS_DisableIRQ(s_pdbIrqId[instance]); /* Disable PDB interrupt in NVIC level.*/
121  }
122 }

124 /*FUNCTION*********************************************************************
125  *
126  * Function Name : PDB_DRV_Deinit
127  * Description : De-initialize the PDB module.
128  * When the PDB module is not used. Calling this function would shut down the
129  * PDB module and reduce the power consumption.
130  *
131  * Note: instance back to back configuration is common between PDB instances 0 and 1 (configures the same register even if configured for either PDB instance)
132  * This function disables it, so affects all other instances.
133  *
134  * Implements : PDB_DRV_Deinit_Activity
135  *END*************************************************************************/
136 void PDB_DRV_Deinit(const uint32_t instance)
137 {
138  DEV_ASSERT(instance < PDB_INSTANCE_COUNT);
139  PDB_Type * base = s_pdbBase[instance];
140 
141  INT_SYS_DisableIRQ(s_pdbIrqId[instance]);
142 
143 #if FEATURE_PDB_HAS_INSTANCE_BACKTOBACK
144  /* Note: instance back to back configuration is common between PDB instances 0 and 1 (configures the same register even if configured for either PDB instance)
145  Enabling when configuring PDB0, then disabling when configuring PDB1, will result in being disabled. */
146  PDB_SIM_DisableInstanceBackToBack();
147 #endif /* FEATURE_PDB_HAS_INSTANCE_BACKTOBACK */
148 
149 #if FEATURE_PDB_HAS_INTERCHANNEL_BACKTOBACK
150  PDB_SIM_DisableInterchannelBackToBack(instance);
151 #endif /* FEATURE_PDB_HAS_INTERCHANNEL_BACKTOBACK */
152 
153  PDB_Disable(base);
154 }

156 /*FUNCTION*********************************************************************
157  *
158  * Function Name : PDB_DRV_GetDefaultConfig
159  * Description : Gets the default configuration structure of PDB with default settings.
160  * This function is used to initialize the hardware configuration structure to default values.
161  *
162  * Implements : PDB_DRV_GetDefaultConfig_Activity
163  *END*************************************************************************/
164 void PDB_DRV_GetDefaultConfig(pdb_timer_config_t * const config)
165 {
166  DEV_ASSERT(config != NULL);
167  /* Load immediately after load operation. */
168  config->loadValueMode = PDB_LOAD_VAL_IMMEDIATELY;
169  /* Sequence error interrupt disable */
170  config->seqErrIntEnable = false;
171  /* Prescaler divider select by MULT */
172  config->clkPreDiv = PDB_CLK_PREDIV_BY_1;
173  /* Select multiplication source mode */
174  config->clkPreMultFactor = PDB_CLK_PREMULT_FACT_AS_1;
175  /* Trigger input source selection */
176  config->triggerInput = PDB_TRIGGER_IN0;
177  /* Run in One-shot mode */
178  config->continuousModeEnable = false;
179  /* Disable DMA */
180  config->dmaEnable = false;
181  /* Disable interrupt */
182  config->intEnable = false;
183 #if FEATURE_PDB_HAS_INSTANCE_BACKTOBACK
184  /* Disable instanceBackToBackEnable */
185  config->instanceBackToBackEnable = false;
186 #endif /* FEATURE_PDB_HAS_INSTANCE_BACKTOBACK */
187 #if FEATURE_PDB_HAS_INTERCHANNEL_BACKTOBACK
188  /* Disable interchannelBackToBackEnable */
189  config->interchannelBackToBackEnable = false;
190 #endif /* FEATURE_PDB_HAS_INTERCHANNEL_BACKTOBACK */
191 }
