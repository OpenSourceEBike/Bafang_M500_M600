118 /*FUNCTION**********************************************************************
119  *
120  * Function Name : LPIT_DRV_Init
121  * Description : Initializes LPIT module.
122  * This function resets LPIT module, enables the LPIT module, configures LPIT
123  * module operation in Debug and DOZE mode. The LPIT configuration structure shall
124  * be passed as arguments.
125  * This configuration structure affects all timer channels.
126  * This function should be called before calling any other LPIT driver function.
127  *
128  * Implements : LPIT_DRV_Init_Activity
129  *END**************************************************************************/
130 void LPIT_DRV_Init(uint32_t instance,
131  const lpit_user_config_t *userConfig)
132 {
133  LPIT_Type * base;
134  status_t clkErr;
135 
136  DEV_ASSERT(instance < LPIT_INSTANCE_COUNT);
137  DEV_ASSERT(userConfig != NULL);
138 
139  /* Gets current functional clock frequency of LPIT instance */
140  clkErr = CLOCK_SYS_GetFreq(s_lpitClkNames[instance], &s_lpitSourceClockFrequency[instance]);
141  /* Checks the functional clock of LPIT module */
142  (void)clkErr;
143  DEV_ASSERT(clkErr == STATUS_SUCCESS);
144  DEV_ASSERT(s_lpitSourceClockFrequency[instance] > 0U);
145  /* When resetting the LPIT module, a delay of 4 peripheral clock cycles
146  must be ensured. This peripheral clock and the core clock running the
147  code could be very different, two distinct cases are identified:
148  - core_clk > peripheral_clk. This requires a delay loop to be implemented,
149  and the delay value based on the ratio between the two frequencies.
150  - core_clk <= peripheral_clk. This requires a short delay, which is usually
151  below the delay caused naturally by the read-modify-write operation.
152  */
153  uint32_t core_freq = 0u;
154  clkErr = CLOCK_SYS_GetFreq(CORE_CLK, &core_freq);
155  (void)clkErr;
156  uint32_t lpit_freq = s_lpitSourceClockFrequency[instance];
157  uint32_t core_to_per_clock_ratio = (core_freq + (lpit_freq >> 1u)) / lpit_freq;
158  base = s_lpitBase[instance];
159  /* Resets LPIT module */
160  LPIT_Reset(base, core_to_per_clock_ratio);
161  /* Enables functional clock of LPIT module*/
162  LPIT_Enable(base, core_to_per_clock_ratio);
163  /* Sets LPIT operation in Debug and DOZE mode*/
164  LPIT_SetTimerRunInDebugCmd(base, userConfig->enableRunInDebug);
165  LPIT_SetTimerRunInDozeCmd(base, userConfig->enableRunInDoze);
166 }

188 /*FUNCTION**********************************************************************
189  *
190  * Function Name : LPIT_DRV_InitChannel
191  * Description : Initializes LPIT channel.
192  * This function initializes the LPIT timers by using a channel, this function
193  * configures timer channel chaining, timer channel mode, timer channel period,
194  * interrupt generation, trigger source, trigger select, reload on trigger,
195  * stop on interrupt and start on trigger.
196  * The timer channel number and its configuration structure shall be passed as arguments.
197  * Timer channels do not start counting by default after calling this function.
198  * The function LPIT_DRV_StartTimerChannels must be called to start the timer channel counting.
199  * In order to re-configures the period, call the LPIT_DRV_SetTimerPeriodByUs or
200  * LPIT_DRV_SetTimerPeriodByCount.
201  *
202  * Implements : LPIT_DRV_InitChannel_Activity
203  *END**************************************************************************/
204 status_t LPIT_DRV_InitChannel(uint32_t instance,
205  uint32_t channel,
206  const lpit_user_channel_config_t * userChannelConfig)
207 {
208  LPIT_Type * base;
209  status_t reVal = STATUS_SUCCESS;
210  const IRQn_Type lpitIrqId[] = LPIT_IRQS;
211 
212  DEV_ASSERT(instance < LPIT_INSTANCE_COUNT);
213  DEV_ASSERT(userChannelConfig != NULL);
214  DEV_ASSERT(channel < LPIT_TMR_COUNT);
215 
216  base = s_lpitBase[instance];
217 
218  if ((channel == 0U) && (userChannelConfig->chainChannel))
219  {
220  reVal = STATUS_ERROR;
221  }
222  else
223  {
224  /* Setups the timer channel chaining */
225  LPIT_SetTimerChannelChainCmd(base, channel, userChannelConfig->chainChannel);
226  /* Setups the timer channel operation mode */
227  LPIT_SetTimerChannelModeCmd(base, channel, userChannelConfig->timerMode);
228  if (userChannelConfig->periodUnits == LPIT_PERIOD_UNITS_MICROSECONDS)
229  {
230  /* Setups timer channel period in microsecond unit */
231  reVal = LPIT_DRV_SetTimerPeriodByUs(instance, channel, userChannelConfig->period);
232  }
233  else
234  {
235  /* Setups timer channel period in count unit */
236  LPIT_DRV_SetTimerPeriodByCount(instance, channel, userChannelConfig->period);
237  }
238 
239  if (reVal == STATUS_SUCCESS)
240  {
241  /* Setups the timer channel trigger source, trigger select, reload on trigger,
242  stop on timeout, start on trigger and channel chaining */
243  LPIT_SetTriggerSourceCmd(base, channel, userChannelConfig->triggerSource);
244  LPIT_SetTriggerSelectCmd(base, channel, userChannelConfig->triggerSelect);
245  LPIT_SetReloadOnTriggerCmd(base, channel, userChannelConfig->enableReloadOnTrigger);
246  LPIT_SetStopOnInterruptCmd(base, channel, userChannelConfig->enableStopOnInterrupt);
247  LPIT_SetStartOnTriggerCmd(base, channel, userChannelConfig->enableStartOnTrigger);
248  /* Setups interrupt generation for timer channel */
249  if (userChannelConfig->isInterruptEnabled)
250  {
251  /* Enables interrupt generation */
252  LPIT_EnableInterruptTimerChannels(base, (uint32_t)1U << channel);
253  INT_SYS_EnableIRQ(lpitIrqId[channel]);
254  }
255  else
256  {
257  /* Disables interrupt generation */
258  LPIT_DisableInterruptTimerChannels(base, (uint32_t)1U << channel);
259  /* Only disable channel interrupt globally if each channel has a separate interrupt line */
260 #if (FEATURE_LPIT_HAS_NUM_IRQS_CHANS == LPIT_TMR_COUNT)
261  INT_SYS_DisableIRQ(lpitIrqId[channel]);
262 #endif
263  }
264  }
265  }
266  return reVal;
267 }

316 /*FUNCTION**********************************************************************
317  *
318  * Function Name : LPIT_DRV_SetTimerPeriodByUs
319  * Description : Sets timer channel period in microseconds unit.
320  * This function sets the timer channel period in microseconds
321  * when timer channel mode is 32 bit periodic or dual 16 bit counter mode.
322  * The period range depends on the frequency of the LPIT functional clock and
323  * operation mode of timer channel.
324  * If the required period is out of range, use the suitable mode if applicable.
325  * This function is only valid for one single channel.
326  *
327  * Implements : LPIT_DRV_SetTimerPeriodByUs_Activity
328  *END**************************************************************************/
329 status_t LPIT_DRV_SetTimerPeriodByUs(uint32_t instance,
330  uint32_t channel,
331  uint32_t periodUs)
332 {
333  LPIT_Type * base;
334  lpit_timer_modes_t timerMode;
335  status_t clkErr;
336  status_t reVal = STATUS_SUCCESS;
337  uint64_t count;
338 
339  DEV_ASSERT(instance < LPIT_INSTANCE_COUNT);
340  DEV_ASSERT(channel < LPIT_TMR_COUNT);
341 
342  /* Gets current functional clock frequency of LPIT instance */
343  clkErr = CLOCK_SYS_GetFreq(s_lpitClkNames[instance], &s_lpitSourceClockFrequency[instance]);
344  /* Checks the functional clock of LPIT module */
345  (void)clkErr;
346  DEV_ASSERT(clkErr == STATUS_SUCCESS);
347  DEV_ASSERT(s_lpitSourceClockFrequency[instance] > 0U);
348 
349  base = s_lpitBase[instance];
350  /* Calculates the count value, assign it to timer channel counter register.*/
351  count = ((uint64_t)periodUs) * s_lpitSourceClockFrequency[instance];
352  count = (count / 1000000U) - 1U;
353  /* Gets current timer channel operation mode */
354  timerMode = LPIT_GetTimerChannelModeCmd(base, channel);
355  /* Checks whether the count is valid with timer channel operation mode */
356  if (count <= MAX_PERIOD_COUNT)
357  {
358  if (timerMode == LPIT_DUAL_PERIODIC_COUNTER)
359  {
360  if (count > MAX_PERIOD_COUNT_IN_DUAL_16BIT_MODE)
361  {
362  reVal = STATUS_ERROR;
363  }
364  else
365  {
366  if (count > MAX_PERIOD_COUNT_16_BIT)
367  {
368  /* Calculates the count value for dual 16 bit periodic counter mode */
369  count = ((count - (MAX_PERIOD_COUNT_16_BIT + 1U)) << 16U)
370  | (MAX_PERIOD_COUNT_16_BIT);
371  }
372  }
373  }
374  }
375  else
376  {
377  reVal = STATUS_ERROR;
378  }
379  if (reVal == STATUS_SUCCESS)
380  {
381  /* Sets the timer channel period in count unit */
382  LPIT_SetTimerPeriodByCount(base, channel, (uint32_t)count);
383  }
384  return reVal;
385 }

387 /*FUNCTION**********************************************************************
388  *
389  * Function Name : LPIT_DRV_SetTimerPeriodInDual16ModeByUs
390  * Description : Sets timer channel period in microseconds unit.
391  * This function sets the timer channel period in microseconds
392  * when timer channel mode is dual 16 bit periodic counter mode.
393  * The period range depends on the frequency of the LPIT functional clock and
394  * operation mode of timer channel.
395  * If the required period is out of range, use the suitable mode if applicable.
396  * This function is only valid for one single channel.
397  *
398  * Implements : LPIT_DRV_SetTimerPeriodInDual16ModeByUs_Activity
399  *END**************************************************************************/
400 status_t LPIT_DRV_SetTimerPeriodInDual16ModeByUs(uint32_t instance,
401  uint32_t channel,
402  uint16_t periodHigh,
403  uint16_t periodLow)
404 {
405  LPIT_Type * base;
406  status_t reVal = STATUS_SUCCESS;
407  status_t clkErr;
408  uint64_t periodHighCount;
409  uint64_t periodLowCount;
410  uint64_t periodCount;
411 
412  DEV_ASSERT(instance < LPIT_INSTANCE_COUNT);
413  DEV_ASSERT(channel < LPIT_TMR_COUNT);
414 
415  /* Gets current functional clock frequency of LPIT instance */
416  clkErr = CLOCK_SYS_GetFreq(s_lpitClkNames[instance], &s_lpitSourceClockFrequency[instance]);
417  /* Checks the functional clock of LPIT module */
418  (void)clkErr;
419  DEV_ASSERT(clkErr == STATUS_SUCCESS);
420  DEV_ASSERT(s_lpitSourceClockFrequency[instance] > 0U);
421 
422  base = s_lpitBase[instance];
423  /* Calculates the count value of 16 bit higher period.*/
424  periodHighCount = ((uint64_t)periodHigh) * s_lpitSourceClockFrequency[instance];
425  periodHighCount = (periodHighCount / 1000000U) - 1U;
426 
427  /* Calculates the count value of 16 bit lower period.*/
428  periodLowCount = ((uint64_t)periodLow) * s_lpitSourceClockFrequency[instance];
429  periodLowCount = (periodLowCount / 1000000U) - 1U;
430  /* Checks whether the count is valid */
431  if ((periodHighCount > MAX_PERIOD_COUNT_16_BIT) || (periodLowCount > MAX_PERIOD_COUNT_16_BIT))
432  {
433  reVal = STATUS_ERROR;
434  }
435  else
436  {
437  periodCount = (periodHighCount << 16U) | periodLowCount;
438  LPIT_SetTimerPeriodByCount(base, channel, (uint32_t)periodCount);
439  }
440 
441  return reVal;
442 }

444 /*FUNCTION**********************************************************************
445  *
446  * Function Name : LPIT_DRV_GetTimerPeriodByUs
447  * Description : Gets the timer channel period in microseconds.
448  * The returned period here makes sense if the operation mode of timer channel
449  * is 32 bit periodic counter or dual 16 bit periodic counter.
450  *
451  * Implements : LPIT_DRV_GetTimerPeriodByUs_Activity
452  *END**************************************************************************/
453 uint64_t LPIT_DRV_GetTimerPeriodByUs(uint32_t instance,
454  uint32_t channel)
455 {
456  const LPIT_Type * base;
457  status_t clkErr;
458  lpit_timer_modes_t timerMode;
459  uint64_t currentPeriod;
460 
461  DEV_ASSERT(instance < LPIT_INSTANCE_COUNT);
462  DEV_ASSERT(channel < LPIT_TMR_COUNT);
463 
464  /* Gets current functional clock frequency of LPIT instance */
465  clkErr = CLOCK_SYS_GetFreq(s_lpitClkNames[instance], &s_lpitSourceClockFrequency[instance]);
466  /* Checks the functional clock of LPIT module */
467  (void)clkErr;
468  DEV_ASSERT(clkErr == STATUS_SUCCESS);
469  DEV_ASSERT(s_lpitSourceClockFrequency[instance] > 0U);
470 
471  base = s_lpitBase[instance];
472  /* Gets current timer channel period in count.*/
473  currentPeriod = LPIT_GetTimerPeriodByCount(base, channel);
474  /* Gets current timer channel operation mode */
475  timerMode = LPIT_GetTimerChannelModeCmd(base, channel);
476 
477  if (timerMode == LPIT_DUAL_PERIODIC_COUNTER)
478  {
479  if (currentPeriod > MAX_PERIOD_COUNT_16_BIT)
480  {
481  currentPeriod = (((currentPeriod >> 16U) + (currentPeriod & MAX_PERIOD_COUNT_16_BIT) + 2U) * 1000000U)
482  / s_lpitSourceClockFrequency[instance];
483  }
484  else
485  {
486  /* Converts period from count unit to microseconds unit for other modes */
487  currentPeriod = ((currentPeriod + 1U) * 1000000U) / s_lpitSourceClockFrequency[instance];
488  }
489  }
490  else
491  {
492  /* Converts period from count unit to microseconds unit for other modes */
493  currentPeriod = ((currentPeriod + 1U) * 1000000U) / s_lpitSourceClockFrequency[instance];
494  }
495  return currentPeriod;
496 }

498 /*FUNCTION**********************************************************************
499  *
500  * Function Name : LPIT_DRV_GetCurrentTimerUs
501  * Description : Gets current timer channel counting value in microseconds unit.
502  * This function returns an absolute time stamp in microseconds.
503  * One common use of this function is to measure the running time of a part of
504  * code. Call this function at both the beginning and end of code. The time
505  * difference between these two time stamps is the running time.
506  * The return counting value here makes sense if the operation mode of timer channel
507  * is 32 bit periodic counter or dual 16 bit periodic counter or 32-bit trigger input capture.
508  * Need to make sure the running time will not exceed the timer channel period.
509  *
510  * Implements : LPIT_DRV_GetCurrentTimerUs_Activity
511  *END**************************************************************************/
512 uint64_t LPIT_DRV_GetCurrentTimerUs(uint32_t instance,
513  uint32_t channel)
514 {
515  const LPIT_Type * base;
516  status_t clkErr;
517  lpit_timer_modes_t timerMode;
518  uint64_t currentTime = 0U;
519 
520  DEV_ASSERT(instance < LPIT_INSTANCE_COUNT);
521  DEV_ASSERT(channel < LPIT_TMR_COUNT);
522 
523  /* Gets current functional clock frequency of LPIT instance */
524  clkErr = CLOCK_SYS_GetFreq(s_lpitClkNames[instance], &s_lpitSourceClockFrequency[instance]);
525  (void)clkErr;
526  DEV_ASSERT(s_lpitSourceClockFrequency[instance] > 0U);
527 
528  base = s_lpitBase[instance];
529  /* Gets current timer channel counting value */
530  currentTime = LPIT_GetCurrentTimerCount(base, channel);
531  /* Gets current timer channel operation mode */
532  timerMode = LPIT_GetTimerChannelModeCmd(base, channel);
533 
534  if (timerMode == LPIT_DUAL_PERIODIC_COUNTER)
535  {
536  currentTime = (((currentTime >> 16U) + (currentTime & MAX_PERIOD_COUNT_16_BIT)) * 1000000U)
537  / s_lpitSourceClockFrequency[instance];
538  }
539  else
540  {
541  /* Converts counting value to microseconds unit for other modes */
542  currentTime = (currentTime * 1000000U) / s_lpitSourceClockFrequency[instance];
543  }
544 
545  return currentTime;
546 }
