86 status_t FLEXIO_DRV_InitDevice(uint32_t instance, flexio_device_state_t *deviceState)
87 {
88  FLEXIO_Type *baseAddr;
89  uint8_t count;
90  status_t osifStat = STATUS_SUCCESS;
91 
92  if (g_flexioDeviceStatePtr[instance] == NULL)
93  {
94  baseAddr = g_flexioBase[instance];
95  /* Create resource allocation mutex */
96  osifStat = OSIF_MutexCreate(&(deviceState->resourceLock));
97  DEV_ASSERT(osifStat == STATUS_SUCCESS);
98  /* Reset FlexIO module */
99  FLEXIO_Init(baseAddr);
100  /* Enable FLEXIO interrupt in the interrupt manager */
101  INT_SYS_EnableIRQ(g_flexioIrqId[instance]);
102  /* Enable module */
103  FLEXIO_SetEnable(baseAddr, true);
104  /* Initialize device structure */
105  deviceState->resourceAllocation = 0U;
106  /* Clear state structures array */
107  for (count = 0U; count < FEATURE_FLEXIO_MAX_SHIFTER_COUNT; count++)
108  {
109  deviceState->flexioStatePtr[count] = NULL;
110  }
111  g_flexioDeviceStatePtr[instance] = deviceState;
112  }
113 
114  (void)osifStat;
115  return STATUS_SUCCESS;
116 }

161 /*FUNCTION**********************************************************************
162  *
163  * Function Name : FLEXIO_DRV_Allocate
164  * Description : Allocate timers and shifters for a new driver instance
165  *
166  *END**************************************************************************/
167 static status_t FLEXIO_DRV_Allocate(uint32_t instance, flexio_common_state_t *driver)
168 {
169  uint8_t count; /* used to iterate through resources (shifters/timers) */
170  uint8_t step; /* keeps track of how many resources are needed for this driver */
171  uint8_t size; /* total number of resources */
172  uint8_t mask; /* bit-mask corresponding to current resources */
173  uint8_t resourceAllocation; /* current resource allocation map */
174  const FLEXIO_Type *baseAddr; /* base address of the FlexIO module */
175  status_t status = STATUS_ERROR; /* assume the worst: no resources found for this driver */
176 
177  /* Find free resources for a new driver. Drivers may need one or two adjacent shifters and timers */
178  baseAddr = g_flexioBase[instance];
179  resourceAllocation = g_flexioDeviceStatePtr[instance]->resourceAllocation;
180  step = driver->resourceCount;
181  size = FLEXIO_GetShifterNum(baseAddr);
182  count = 0U;
183  mask = (uint8_t)((1U << step) - 1U);
184  (void)OSIF_MutexLock(&(g_flexioDeviceStatePtr[instance]->resourceLock), OSIF_WAIT_FOREVER);
185 
186  /* find available shifters and timers for the driver */
187  while ((status == STATUS_ERROR) && (((uint8_t)(count + step))<= size))
188  {
189  if ((resourceAllocation & mask) == 0U)
190  {
191  /* resources found */
192  driver->resourceIndex = count;
193  /* mark resources as busy */
194  g_flexioDeviceStatePtr[instance]->resourceAllocation |= mask;
195  status = STATUS_SUCCESS;
196  }
197  /* continue searching */
198  count += 1U;
199  mask <<= 1;
200  }
201 
202  (void)OSIF_MutexUnlock(&(g_flexioDeviceStatePtr[instance]->resourceLock));
203 
204  return status;
205 }

208 /*FUNCTION**********************************************************************
209  *
210  * Function Name : FLEXIO_DRV_Deallocate
211  * Description : De-allocate timers and shifters for a driver instance
212  *
213  *END**************************************************************************/
214 static void FLEXIO_DRV_Deallocate(uint32_t instance, const flexio_common_state_t *driver)
215 {
216  uint8_t mask;
217 
218  mask = (uint8_t)((1U << driver->resourceCount) - 1U);
219  mask <<= driver->resourceIndex;
220  (void)OSIF_MutexLock(&(g_flexioDeviceStatePtr[instance]->resourceLock), OSIF_WAIT_FOREVER);
221  g_flexioDeviceStatePtr[instance]->resourceAllocation &= (uint8_t)~mask;
222  (void)OSIF_MutexUnlock(&(g_flexioDeviceStatePtr[instance]->resourceLock));
223 }

226 /*FUNCTION**********************************************************************
227  *
228  * Function Name : FLEXIO_DRV_ResourcesInit
229  * Description : Initializes the resources for the current driver
230  *
231  *END**************************************************************************/
232 static void FLEXIO_DRV_ResourcesInit(uint32_t instance, const flexio_common_state_t *driver)
233 {
234  uint8_t resource;
235  uint8_t resourceCount;
236  uint8_t resourceIndex;
237  FLEXIO_Type *baseAddr;
238 
239  baseAddr = g_flexioBase[instance];
240  resourceCount = driver->resourceCount;
241  resourceIndex = driver->resourceIndex;
242  for (resource = resourceIndex; resource < (resourceIndex + resourceCount); resource++)
243  {
244  /* Ensure all shifters/timers are disabled */
245  FLEXIO_SetShifterMode(baseAddr, resource, FLEXIO_SHIFTER_MODE_DISABLED);
246  FLEXIO_SetTimerMode(baseAddr, resource, FLEXIO_TIMER_MODE_DISABLED);
247  /* Ensure all interrupts and DMA requests are disabled */
248  FLEXIO_SetShifterInterrupt(baseAddr, (uint8_t)(1U << resource), false);
249  FLEXIO_SetShifterErrorInterrupt(baseAddr, (uint8_t)(1U << resource), false);
250  FLEXIO_SetTimerInterrupt(baseAddr, (uint8_t)(1U << resource), false);
251  FLEXIO_SetShifterDMARequest(baseAddr, (uint8_t)(1U << resource), false);
252  /* Clear any leftover flags */
253  FLEXIO_ClearShifterStatus(baseAddr, resource);
254  FLEXIO_ClearShifterErrorStatus(baseAddr, resource);
255  FLEXIO_ClearTimerStatus(baseAddr, resource);
256  }
257 }

260 /*FUNCTION**********************************************************************
261  *
262  * Function Name : FLEXIO_DRV_InitDriver
263  * Description : Initializes an instance of FlexIO driver
264  *
265  *END**************************************************************************/
266 status_t FLEXIO_DRV_InitDriver(uint32_t instance, flexio_common_state_t *driver)
267 {
268  uint16_t count;
269  status_t retCode;
270  flexio_common_state_t **flexioStatePtr;
271 
272  /* allocate times and shifters for the driver */
273  retCode = FLEXIO_DRV_Allocate(instance, driver);
274  if (retCode != STATUS_SUCCESS)
275  { /* no more resources available */
276  return retCode;
277  }
278 
279  /* get driver list for this device instance */
280  flexioStatePtr = g_flexioDeviceStatePtr[instance]->flexioStatePtr;
281  /* find an empty state structure slot for the driver */
282  for (count = 0U; count < FEATURE_FLEXIO_MAX_SHIFTER_COUNT; count++)
283  {
284  if (flexioStatePtr[count] == NULL)
285  {
286  /* found it, place the new driver struct here */
287  flexioStatePtr[count] = driver;
288  break;
289  }
290  }
291  /* no need to check if there was room since allocation was successful */
292  DEV_ASSERT(count < FEATURE_FLEXIO_MAX_SHIFTER_COUNT);
293 
294  /* initialize the allocated resources */
295  FLEXIO_DRV_ResourcesInit(instance, driver);
296 
297  /* init the rest of the common state structure */
298  driver->isr = NULL;
299  driver->instance = instance;
300 
301  return STATUS_SUCCESS;
302 }

341  *
342  * Function Name : FLEXIO_IRQHandler
343  * Description : Implementation of FLEXIO interrupt handler named in start-up code
344  *
345  *END**************************************************************************/
346 void FLEXIO_IRQHandler(void)
347 {
348  uint16_t count;
349  uint32_t instance;
350  uint32_t resourceMask;
351  uint32_t shifterEvents;
352  uint32_t shifterErrorEvents;
353  uint32_t timerEvents;
354  uint32_t enabledInterrupts;
355  const FLEXIO_Type *baseAddr;
356  flexio_common_state_t * const *driverList;
357  flexio_common_state_t *driverState;
358 
359  instance = 0U;
360  baseAddr = g_flexioBase[instance];
361  /* get masks of FlexIO events */
362  /* read enabled interrupts in a separate instruction to avoid MISRA violation */
363  enabledInterrupts = FLEXIO_GetAllShifterInterrupt(baseAddr);
364  shifterEvents = FLEXIO_GetAllShifterStatus(baseAddr) & enabledInterrupts;
365  enabledInterrupts = FLEXIO_GetAllShifterErrorInterrupt(baseAddr);
366  shifterErrorEvents = FLEXIO_GetAllShifterErrorStatus(baseAddr) & enabledInterrupts;
367  enabledInterrupts = FLEXIO_GetAllTimerInterrupt(baseAddr);
368  timerEvents = FLEXIO_GetAllTimerStatus(baseAddr) & enabledInterrupts;
369  /* get driver list for this device instance */
370  driverList = g_flexioDeviceStatePtr[instance]->flexioStatePtr;
371  /* check which driver instances need to be serviced */
372  for (count = 0U; count < FEATURE_FLEXIO_MAX_SHIFTER_COUNT; count++)
373  {
374  driverState = driverList[count];
375  /* check if driver is initialized and uses interrupts */
376  if ((driverState != NULL) && (driverState->isr != NULL))
377  {
378  /* compute mask of shifters/timers used by this driver */
379  resourceMask = ((1UL << driverState->resourceCount) - 1U) << driverState->resourceIndex;
380  /* check if this instance has any pending events */
381  if (((shifterEvents & resourceMask) != 0U) ||
382  ((shifterErrorEvents & resourceMask) != 0U) ||
383  ((timerEvents & resourceMask) != 0U))
384  {
385  /* there is an event for the current instance - call the isr */
386  (driverState->isr)(driverState);
387  }
388  }
389  }
390 }
