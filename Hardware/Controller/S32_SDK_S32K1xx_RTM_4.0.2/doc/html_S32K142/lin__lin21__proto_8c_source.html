41  /* Check MASTER function */
42  if ((frame_index != 0xFFU) && ((bool)LIN_MASTER == prot_user_config_ptr->function))
43  {
44  /* Find collision resolver table */
45  /* Swap schedule table */
46  /* Get active_schedule_id */

52 
53  master_data_ptr->active_schedule_id = ptr->coll_resolv_schd;

60 /*FUNCTION**********************************************************************

62  * Function Name : lin_update_word_status_lin21
63  * Description : Update node status flags.

65  * Implements : lin_update_word_status_lin21_Activity
66  *END**************************************************************************/
67 void lin_update_word_status_lin21(l_ifc_handle iii,
68  lin_lld_event_id_t event_id)
69 {
70  l_u8 err_in_res;
71  l_u8 suc_in_tras;
72  l_u8 save_config;
73  l_u8 overrun_flag;
74  lin_word_status_str_t * word_status_ptr;
75  lin_protocol_state_t * prot_state_ptr = &g_lin_protocol_state_array[iii];
76 #if (SUPPORT_MASTER_MODE == 1U)
77  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
78  const lin_master_data_t * master_data_ptr;
79  l_u8 active_schedule_id;
80 #endif /* End (SUPPORT_MASTER_MODE == 1U) */
81 
82  err_in_res = prot_state_ptr->error_in_response;
83  prot_state_ptr->error_in_response = 0;
84  suc_in_tras = prot_state_ptr->successful_transfer;
85  /* use of ternary operator below is for being MISRA compliant */
86  save_config = (l_u8)(prot_state_ptr->save_config_flg ? 1U : 0U);
87  overrun_flag = prot_state_ptr->overrun_flg;
88 
89  word_status_ptr = &prot_state_ptr->word_status;
90  word_status_ptr->overrun = overrun_flag;
91 
92  /* values for all processed frames */
93  word_status_ptr->error_in_res |= err_in_res;
94  word_status_ptr->successful_transfer |= suc_in_tras;
95  word_status_ptr->last_pid = prot_state_ptr->last_pid;
96 
97  if(prot_state_ptr->go_to_sleep_flg == true)
98  {
99  /* Set go to sleep flag for word status*/
100  word_status_ptr->go_to_sleep_flg = 1;
101  }
102 
103  if(LIN_LLD_BUS_ACTIVITY_TIMEOUT != event_id)
104  {
105  /* Set bus activity timeout */
106  word_status_ptr->bus_activity = 1U;
107  }
108 
109 #if (SUPPORT_MASTER_MODE == 1U)
110  if ((bool)LIN_MASTER == prot_user_config_ptr->function)
111  {
112  master_data_ptr = &g_lin_master_data_array[g_lin_protocol_user_cfg_array[iii].master_ifc_handle];
113  /* If is Master node, value always equal 0 */
114  word_status_ptr->save_config_flg = 0U;
115 
116 
117  active_schedule_id = master_data_ptr->active_schedule_id;
118  /* If collision schedule table is executing, set event_trigger_collision_flag bit field*/
119  if (prot_user_config_ptr->schedule_tbl[active_schedule_id].sch_tbl_type == LIN_SCH_TBL_COLL_RESOLV)
120  {
121  word_status_ptr->event_trigger_collision_flg = 1;
122  }
123  else
124  {
125  word_status_ptr->event_trigger_collision_flg = 0;
126  }
127  }
128  else
129 #endif /* End (SUPPORT_MASTER_MODE == 1U) */
130  {
131  /* For Slave node - using ternary operator is a workaround for MISRA */
132  word_status_ptr->event_trigger_collision_flg = 0U;
133  /* Updated save configuration value to word status */
134  word_status_ptr->save_config_flg = save_config;
135  word_status_ptr->reserved = 0U;
136  }
137 }

147 void lin_update_err_signal(l_ifc_handle iii,
148  l_u8 frm_id)
149 {
150  l_u8 i;
151  const l_u16* byte_offset_temp_ptr;
152  const l_u8* bit_offset_temp_ptr;
153  l_u8 err_in_res = g_lin_protocol_state_array[iii].error_in_response;
154  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
155  const lin_node_attribute_t * node_attr_ptr;
156  node_attr_ptr = &g_lin_node_attribute_array[prot_user_config_ptr->slave_ifc_handle];
157 
158  if (frm_id < LIN_NUM_OF_FRMS)
159  {
160  /* Set error signal equal to error in response */
161  if (LIN_FRM_EVNT != prot_user_config_ptr->frame_tbl_ptr[frm_id].frm_type)
162  {
163  for (i = 0; i < node_attr_ptr->num_frame_have_esignal; i++)
164  {
165  byte_offset_temp_ptr = &node_attr_ptr->response_error_byte_offset_ptr[i];
166  bit_offset_temp_ptr = &node_attr_ptr->response_error_bit_offset_ptr[i];
167  g_lin_frame_data_buffer[*byte_offset_temp_ptr] = (l_u8)((g_lin_frame_data_buffer[*byte_offset_temp_ptr] & (~(1U << (*bit_offset_temp_ptr)))) |
168  ((l_u8)(err_in_res << (*bit_offset_temp_ptr))));
169  }
170  }
171  }
172 }

176 /*FUNCTION**********************************************************************
177  *
178  * Function Name : lin_update_rx_evnt_frame
179  * Description : The function updates the receive flags associated with signals/frames
180  * in case receive an event trigger frame.
181  *
182  * Implements : lin_update_rx_evnt_frame_Activity
183  *END**************************************************************************/
184 void lin_update_rx_evnt_frame(l_ifc_handle iii,
185  l_u8 pid)
186 {
187  const lin_associate_frame_t * ptr;
188  l_u8 frame_index, i, associate_id;
189  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
190 
191  /* Get frame index of the event trigger frame */
192  frame_index = lin_get_frame_index(iii, pid);
193 
194  if (0xFFU != frame_index) /* This PID doesn't belong to this node */
195  {
196  /* calculate PID that associated with ETF */
197  associate_id = lin_process_parity(g_lin_protocol_state_array[iii].response_buffer_ptr[0], CHECK_PARITY);
198  ptr = prot_user_config_ptr->frame_tbl_ptr[frame_index].frame_data_ptr;
199 
200  for (i = 0U; i < ptr->num_of_associated_uncond_frames; i++)
201  {
202  /* First byte of response for event trigger is the PID of the associate frame */
203  if (prot_user_config_ptr->list_identifiers_RAM_ptr[((l_u8)ptr->associated_uncond_frame_ptr[i] - prot_user_config_ptr->frame_start) + 1U] == associate_id)
204  {
205  /* Update unconditional frame */
206  lin_update_uncd_frame(iii, associate_id);
207  break;
208  }
209  }
210  }
211 }
