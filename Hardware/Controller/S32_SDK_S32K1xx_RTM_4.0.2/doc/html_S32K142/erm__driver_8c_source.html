48  * Function Name : ERM_DRV_Init
49  * Description : This function initializes ERM driver based on user configuration input,

53  *END**************************************************************************/
54 void ERM_DRV_Init(uint32_t instance,
55  uint8_t channelCnt,
56  const erm_user_config_t * userConfigArr)
57 {
58  DEV_ASSERT(instance < ERM_INSTANCE_COUNT);
59  DEV_ASSERT(userConfigArr != NULL);
60  DEV_ASSERT(channelCnt > 0U);
61  DEV_ASSERT(channelCnt <= ERM_EARn_COUNT);
62  ERM_Type * base = s_ermBase[instance];
63  uint8_t i;
64 
65  /* Initializes the module */
66  ERM_Init(base);
67 
68  /* Sets interrupt notification from user configuration input */
69  for (i = 0U; i < channelCnt; i++)
70  {
71  ERM_DRV_SetInterruptConfig(instance, userConfigArr[i].channel, *userConfigArr[i].interruptCfg);
72  }
73 }

169  * Description : This function gets the address of the last ECC event in Memory n

173  *END**************************************************************************/
174 erm_ecc_event_t ERM_DRV_GetErrorDetail(uint32_t instance,
175  uint8_t channel,
176  uint32_t * addressPtr)
177 {
178  DEV_ASSERT(instance < ERM_INSTANCE_COUNT);
179  DEV_ASSERT(channel < ERM_EARn_COUNT);
180  DEV_ASSERT(addressPtr != NULL);
181  const ERM_Type * base = s_ermBase[instance];
182  erm_ecc_event_t eccEvent;
183 
184  /* Single-bit correction ECC events */
185  if (ERM_IsEventDetected(base, channel, ERM_EVENT_SINGLE_BIT) != false)
186  {
187  eccEvent = ERM_EVENT_SINGLE_BIT;
188  }
189  /* Non-correctable ECC events */
190  else if (ERM_IsEventDetected(base, channel, ERM_EVENT_NON_CORRECTABLE) != false)
191  {
192  eccEvent = ERM_EVENT_NON_CORRECTABLE;
193  }
194  /* None ECC events */
195  else
196  {
197  eccEvent = ERM_EVENT_NONE;
198  }
199 
200  /* Gets address */
201  *addressPtr = ERM_GetLastErrorAddress(base, channel);
202 
203  return eccEvent;
204 }
