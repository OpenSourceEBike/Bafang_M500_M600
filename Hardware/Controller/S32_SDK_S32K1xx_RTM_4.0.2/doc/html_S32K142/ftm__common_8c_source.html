106 /*FUNCTION**********************************************************************
107  *
108  * Function Name : FTM_DRV_Init
109  * Description : Initializes the FTM driver and get the clock frequency value
110  * which select one of three possible clock sources for the FTM counter.
111  *
112  * Implements : FTM_DRV_Init_Activity
113  *END**************************************************************************/
114 status_t FTM_DRV_Init(uint32_t instance,
115  const ftm_user_config_t * info,
116  ftm_state_t * state)
117 {
118  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
119  DEV_ASSERT(info != NULL);
120  DEV_ASSERT(state != NULL);
121  DEV_ASSERT((info->ftmClockSource == FTM_CLOCK_SOURCE_SYSTEMCLK) || \
122  (info->ftmClockSource == FTM_CLOCK_SOURCE_FIXEDCLK) || \
123  (info->ftmClockSource == FTM_CLOCK_SOURCE_EXTERNALCLK));
124  FTM_Type * ftmBase = g_ftmBase[instance];
125  status_t status = STATUS_SUCCESS;
126  uint8_t index = 0U;
127 
128  /* Check if this instance is already initialized */
129  if (ftmStatePtr[instance] != NULL)
130  {
131  status = STATUS_ERROR;
132  }
133  else
134  {
135  /* Configure state structure. */
136  state->ftmClockSource = info->ftmClockSource;
137  state->ftmMode = FTM_MODE_NOT_INITIALIZED;
138  state->ftmPeriod = 0U;
139  ftmStatePtr[instance] = state;
140 
141  for (index = 0U; index < FEATURE_FTM_CHANNEL_COUNT; index++)
142  {
143  state->measurementResults[index] = 0U;
144  state->channelsCallbacksParams[index] = NULL;
145  state->channelsCallbacks[index] = NULL;
146  state->enableNotification[index] = false;
147  }
148 
149  /* The reset operation doesn't care about write protection. FTM_DRV_Reset will
150  * disable this protection.*/
151  FTM_DRV_Reset(ftmBase);
152  FTM_DRV_InitModule(ftmBase, info->ftmPrescaler);
153  /* Get clock name used to configure the FlexTimer module */
154  state->ftmSourceClockFrequency = FTM_DRV_GetFrequency(instance);
155  /* Check the functional clock is selected for FTM */
156  DEV_ASSERT(state->ftmSourceClockFrequency > 0U);
157  }
158 
159  if (STATUS_SUCCESS == status)
160  {
161  /* Check if the mode operation in PWM mode */
162  if ((FTM_MODE_EDGE_ALIGNED_PWM == info->ftmMode) || (FTM_MODE_CEN_ALIGNED_PWM == info->ftmMode) || (FTM_MODE_OUTPUT_COMPARE == info->ftmMode) || (FTM_MODE_EDGE_ALIGNED_PWM_AND_INPUT_CAPTURE == info->ftmMode))
163  {
164  /* Configure sync for between registers and buffers */
165  status = FTM_DRV_SetSync(instance, &(info->syncMethod));
166  }
167 
168  /* Enable the generation of initialization trigger on chip module */
169  FTM_DRV_SetInitTriggerCmd(ftmBase, info->enableInitializationTrigger);
170  FTM_DRV_SetBdmMode(ftmBase, info->BDMMode);
171 
172  /* Check if enable interrupt in counter mode */
173  if (info->isTofIsrEnabled)
174  {
175  FTM_DRV_SetTimerOverflowInt(ftmBase, true);
176  INT_SYS_EnableIRQ(g_ftmOverflowIrqId[instance]);
177  }
178  else
179  {
180  FTM_DRV_SetTimerOverflowInt(ftmBase, false);
181  INT_SYS_DisableIRQ(g_ftmOverflowIrqId[instance]);
182  }
183  }
184 
185  return status;
186 }

438 /*FUNCTION**********************************************************************
439  *
440  * Function Name : FTM_DRV_SetSync
441  * Description : This function configure the synchronization for PWM register
442  * (CnV, MOD, CINT, HCR, OUTMASK).If this function is used whit wrong parameters
443  * it's possible to generate wrong waveform. Registers synchronization need to
444  * be configured for PWM and output compare mode.
445  *
446  * Implements : FTM_DRV_SetSync_Activity
447  *END**************************************************************************/
448 status_t FTM_DRV_SetSync(uint32_t instance,
449  const ftm_pwm_sync_t * param)
450 {
451  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
452  DEV_ASSERT(param != NULL);
453  FTM_Type * ftmBase = g_ftmBase[instance];
454  status_t retStatus = STATUS_SUCCESS;
455  bool hardwareSync = param->hardwareSync0 || param->hardwareSync1 || param->hardwareSync2;
456 
457  /* Software and hardware triggers are not allowed in the same time */
458  if ((param->softwareSync && hardwareSync) || (true != (param->softwareSync || hardwareSync)))
459  {
460  retStatus = STATUS_ERROR;
461  }
462  else if (param->softwareSync)
463  {
464  /* Configure sync for OUTMASK register */
465  FTM_DRV_SetOutmaskSoftwareSyncModeCmd(ftmBase, true);
466  /* Configure sync for INVCTRL register */
467  FTM_DRV_SetInvctrlSoftwareSyncModeCmd(ftmBase, true);
468  /* Configure sync for SWOCTRL register */
469  FTM_DRV_SetSwoctrlSoftwareSyncModeCmd(ftmBase, true);
470  /* Configure sync for MOD, HCR, CNTIN, and CnV registers */
471  FTM_DRV_SetModCntinCvSoftwareSyncModeCmd(ftmBase, true);
472  /* Configure synchronization method (waiting next loading point or now) */
473  FTM_DRV_SetCounterSoftwareSyncModeCmd(ftmBase, param->syncPoint);
474  }
475  else
476  {
477  /* Configure sync for OUTMASK register */
478  FTM_DRV_SetOutmaskHardwareSyncModeCmd(ftmBase, true);
479  /* Configure sync for INVCTRL register */
480  FTM_DRV_SetInvctrlHardwareSyncModeCmd(ftmBase, true);
481  /* Configure sync for SWOCTRL register */
482  FTM_DRV_SetSwoctrlHardwareSyncModeCmd(ftmBase, true);
483  /* Configure sync for MOD, HCR, CNTIN, and CnV registers */
484  FTM_DRV_SetModCntinCvHardwareSyncModeCmd(ftmBase, true);
485  /* Configure synchronization method (waiting next loading point or now) */
486  FTM_DRV_SetCounterHardwareSyncModeCmd(ftmBase, (bool)param->syncPoint);
487  }
488 
489  if (STATUS_SUCCESS == retStatus)
490  {
491  /* Enhanced PWM sync is used */
492  FTM_DRV_SetPwmSyncModeCmd(ftmBase, true);
493  /* Configure trigger source for sync */
494  FTM_DRV_SetHardwareSyncTriggerSrc(ftmBase, 0U, param->hardwareSync0);
495  FTM_DRV_SetHardwareSyncTriggerSrc(ftmBase, 1U, param->hardwareSync1);
496  FTM_DRV_SetHardwareSyncTriggerSrc(ftmBase, 2U, param->hardwareSync2);
497  /* Configure loading points */
498  FTM_DRV_SetMaxLoadingCmd(ftmBase, param->maxLoadingPoint);
499  FTM_DRV_SetMinLoadingCmd(ftmBase, param->minLoadingPoint);
500  /* Configure sync for OUTMASK register */
501  FTM_DRV_SetOutmaskPwmSyncModeCmd(ftmBase, (bool)param->maskRegSync);
502  /* Configure sync for INVCTRL register */
503  FTM_DRV_SetInvctrlPwmSyncModeCmd(ftmBase, param->inverterSync);
504  /* Configure sync for SWOCTRL register */
505  FTM_DRV_SetSwoctrlPwmSyncModeCmd(ftmBase, param->outRegSync);
506  /* Configure sync for MOD, HCR, CNTIN, and CnV registers */
507  FTM_DRV_SetCntinPwmSyncModeCmd(ftmBase, param->initCounterSync);
508  /* Configure if FTM clears TRIGj (j=0,1,2) when the hardware trigger j is detected. */
509  FTM_DRV_SetHwTriggerSyncModeCmd(ftmBase, param->autoClearTrigger);
510  }
511 
512  return retStatus;
513 }

534 /*FUNCTION**********************************************************************
535  *
536  * Function Name : FTM_DRV_EnableInterrupts
537  * Description : This function will enable the generation a list of interrupts.
538  * It includes the FTM overflow interrupts, the reload point interrupt, the fault
539  * interrupt and the channel (n) interrupt.
540  *
541  * Implements : FTM_DRV_EnableInterrupts_Activity
542  *END**************************************************************************/
543 status_t FTM_DRV_EnableInterrupts(uint32_t instance,
544  uint32_t interruptMask)
545 {
546  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
547  FTM_Type * ftmBase = g_ftmBase[instance];
548  uint32_t chnlInts = (interruptMask & 0x000000FFU);
549  uint8_t channel = 0U;
550 
551  /* Enable the timer overflow interrupt */
552  if ((interruptMask & (uint32_t)FTM_TIME_OVER_FLOW_INT_ENABLE) != 0x0U)
553  {
554  FTM_DRV_SetTimerOverflowInt(ftmBase, true);
555  INT_SYS_EnableIRQ(g_ftmOverflowIrqId[instance]);
556  }
557 
558  /* Enable the fault interrupt */
559  if ((interruptMask & (uint32_t)FTM_FAULT_INT_ENABLE) != 0x0U)
560  {
561  FTM_DRV_SetFaultInt(ftmBase, true);
562  INT_SYS_EnableIRQ(g_ftmFaultIrqId[instance]);
563  }
564 
565  /* Enable the reload interrupt */
566  if ((interruptMask & (uint32_t)FTM_RELOAD_INT_ENABLE) != 0x0U)
567  {
568  FTM_DRV_SetReIntEnabledCmd(ftmBase, true);
569  INT_SYS_EnableIRQ(g_ftmReloadIrqId[instance]);
570  }
571 
572  /* Enable the channel interrupts */
573  while (chnlInts != 0U)
574  {
575  if ((chnlInts & 0x1U) != 0x0U)
576  {
577  FTM_DRV_EnableChnInt(ftmBase, channel);
578  INT_SYS_EnableIRQ(g_ftmIrqId[instance][channel]);
579  }
580  channel++;
581  chnlInts = chnlInts >> 1U;
582  }
583 
584  return STATUS_SUCCESS;
585 }

632  chnlInts = chnlInts >> 1U;
633  }
634 }

637  *

639  * Description : This function will get the enabled FTM interrupts.

641  * Implements : FTM_DRV_GetEnabledInterrupts_Activity

651  /* Check if timer overflow interrupt is enabled */
652  if (FTM_DRV_IsOverflowIntEnabled(ftmBase) == true)
653  {
654  enabledInterrupts |= (uint32_t)FTM_TIME_OVER_FLOW_INT_ENABLE;
655  }
656 
657  /* Check if fault interrupt is enabled */
658  if (FTM_DRV_IsFaultIntEnabled(ftmBase) == true)
659  {
660  enabledInterrupts |= (uint32_t)FTM_FAULT_INT_ENABLE;
661  }
662 
663  /* Check if the reload interrupt is enabled */
664  if (FTM_DRV_IsReloadIntEnabled(ftmBase) == true)
665  {
666  enabledInterrupts |= (uint32_t)FTM_RELOAD_INT_ENABLE;
667  }
668 
669  /* Check if the channel interrupts are enabled */
670  while (channel > 0U)
671  {
672  channel--;
673  if (FTM_DRV_IsChnIntEnabled(ftmBase, channel) == true)
674  {
675  enabledInterrupts |= (1UL << (uint32_t)channel);
676  }
677  }
678 
679  return enabledInterrupts;
680 }

682 /*FUNCTION**********************************************************************
683  *
684  * Function Name : FTM_DRV_GetStatusFlags
685  * Description : This function will get the FTM status flags.
686  *
687  * Implements : FTM_DRV_GetStatusFlags_Activity
688  *END**************************************************************************/
689 uint32_t FTM_DRV_GetStatusFlags(uint32_t instance)
690 {
691  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
692  FTM_Type const * ftmBase = g_ftmBase[instance];
693  uint8_t channel = 0U;
694  uint32_t statusFlags = 0U;
695 
696  /* Check the timer flag */
697  if (FTM_DRV_HasTimerOverflowed(ftmBase) == true)
698  {
699  statusFlags |= (uint32_t)FTM_TIME_OVER_FLOW_FLAG;
700  }
701 
702  /* Check fault flag */
703  if (FTM_DRV_GetDetectedFaultInput(ftmBase) == true)
704  {
705  statusFlags |= (uint32_t)FTM_FAULT_FLAG;
706  }
707 
708  /* Check reload flag */
709  if (FTM_DRV_GetReloadFlag(ftmBase) == true)
710  {
711  statusFlags |= (uint32_t)FTM_RELOAD_FLAG;
712  }
713 
714  /* Check channel trigger flag */
715  if (FTM_DRV_IsChnTriggerGenerated(ftmBase) == true)
716  {
717  statusFlags |= (uint32_t)FTM_CHANNEL_TRIGGER_FLAG;
718  }
719 
720  /* Lower 8 bits contain the channel status flags */
721  for (channel = 0U; channel < FEATURE_FTM_CHANNEL_COUNT; channel++)
722  {
723  if (FTM_DRV_HasChnEventOccurred(ftmBase, channel) == true)
724  {
725  statusFlags |= (1UL << (uint32_t)channel);
726  }
727  }
728 
729  return statusFlags;
730 }

732 /*FUNCTION**********************************************************************
733  *
734  * Function Name : FTM_DRV_ClearStatusFlags
735  * Description : This function is used to clear the FTM status flags.
736  *
737  * Implements : FTM_DRV_ClearStatusFlags_Activity
738  *END**************************************************************************/
739 void FTM_DRV_ClearStatusFlags(uint32_t instance,
740  uint32_t flagMask)
741 {
742  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
743  FTM_Type * ftmBase = g_ftmBase[instance];
744  uint32_t chnlMask = (flagMask & 0x000000FFU);
745  uint8_t channel = 0U;
746 
747  /* Clear the timer overflow flag by writing a 0 to the bit while it is set */
748  if ((flagMask & (uint32_t)FTM_TIME_OVER_FLOW_FLAG) != 0x0U)
749  {
750  FTM_DRV_ClearTimerOverflow(ftmBase);
751  }
752 
753  /* Clear fault flag by writing a 0 to the bit while it is set */
754  if ((flagMask & (uint32_t)FTM_FAULT_FLAG) != 0x0U)
755  {
756  FTM_DRV_ClearFaultsIsr(ftmBase);
757  }
758 
759  /* Check reload flag by writing a 0 to the bit while it is set */
760  if ((flagMask & (uint32_t)FTM_RELOAD_FLAG) != 0x0U)
761  {
762  FTM_DRV_ClearReloadFlag(ftmBase);
763  }
764 
765  /* Clear channel trigger flag */
766  if ((flagMask & (uint32_t)FTM_CHANNEL_TRIGGER_FLAG) != 0x0U)
767  {
768  FTM_DRV_ClearChnTriggerFlag(ftmBase);
769  }
770 
771  /* Clear the channel status flags by writing a 0 to the bit */
772  for (channel = 0U; channel < FEATURE_FTM_CHANNEL_COUNT; channel++)
773  {
774  if ((chnlMask & 0x00000001U) != 0x0U)
775  {
776  FTM_DRV_ClearChnEventStatus(ftmBase, channel);
777  }
778  chnlMask = chnlMask >> 1U;
779  }
780 }

782 /*FUNCTION**********************************************************************
783  *
784  * Function Name : FTM_DRV_GetFrequency
785  * Description : Retrieves the frequency of the clock source feeding the FTM counter.
786  * Function will return a 0 if no clock source is selected and the FTM counter is disabled.
787  * The returned value is clock sources for the FTM counter.
788  *
789  * Implements : FTM_DRV_GetFrequency_Activity
790  *END**************************************************************************/
791 uint32_t FTM_DRV_GetFrequency(uint32_t instance)
792 {
793  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
794  FTM_Type const * ftmBase = g_ftmBase[instance];
795  status_t returnCode = STATUS_SUCCESS;
796  uint8_t clkPs;
797  uint32_t frequency = 0U;
798  const ftm_state_t * state = ftmStatePtr[instance];
799  DEV_ASSERT(state != NULL);
800  clkPs = (uint8_t)(1U << FTM_DRV_GetClockPs(ftmBase));
801 
802  switch (state->ftmClockSource)
803  {
804  case FTM_CLOCK_SOURCE_EXTERNALCLK:
805  returnCode = CLOCK_SYS_GetFreq(g_ftmExtClockSel[instance], &frequency);
806  break;
807  case FTM_CLOCK_SOURCE_FIXEDCLK:
808  /* Get the clock frequency value */
809  returnCode = CLOCK_SYS_GetFreq(SIM_RTCCLK_CLK, &frequency);
810  break;
811  case FTM_CLOCK_SOURCE_SYSTEMCLK:
812  /* Get the clock frequency value */
813  returnCode = CLOCK_SYS_GetFreq(CORE_CLK, &frequency);
814  break;
815  default:
816  /* Nothing to do */
817  break;
818  }
819 
820  /* Checks the functional clock of FTM module */
821  (void)returnCode;
822  DEV_ASSERT(returnCode == STATUS_SUCCESS);
823 
824  return (uint32_t)(frequency / clkPs);
825 }
