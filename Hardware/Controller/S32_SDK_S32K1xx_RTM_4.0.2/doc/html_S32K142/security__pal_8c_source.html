67 /*FUNCTION**********************************************************************
68  *
69  * Function Name : SecurityAllocateInstance
70  * Description : Allocate state.
71  *
72  *END**************************************************************************/
73 static status_t SecurityAllocateInstance(bool * isAllocated,
74  security_instance_t * instList,
75  security_instance_t instId,
76  uint32_t numberOfInst,
77  uint32_t * retInstId)
78 {
79  uint32_t counter = 0U;
80 
81  for (counter = 0U; counter < numberOfInst; counter++)
82  {
83  if (false == isAllocated[counter])
84  {
85  isAllocated[counter] = true;
86  instList[counter] = instId;
87  *retInstId = counter;
88  return STATUS_SUCCESS;
89  }
90  }
91 
92  return STATUS_ERROR;
93 }

101 static void SecurityFreeInstance(bool * isAllocated,
102  const security_instance_t * instList,
103  security_instance_t instId,
104  uint32_t numberOfInst)
105 {
106  uint32_t counter = 0U;
107 
108  for (counter = 0U; counter < numberOfInst; counter++)
109  {
110  if (instId == instList[counter])
111  {
112  isAllocated[counter] = false;
113  break;
114  }
115  }
116 }

133 /*FUNCTION**********************************************************************
134  *
135  * Function Name : SECURITY_Init
136  * Description : Initializes the state structure, enables interrupt,
137  * installs interrupt handler.
138  * Implements : SECURITY_Init_Activity
139  *
140  *END**************************************************************************/
141 status_t SECURITY_Init(security_instance_t instance,
142  const security_user_config_t * config)
143 {
144  DEV_ASSERT(config != NULL);
145  status_t status = STATUS_SUCCESS;
146  uint32_t stateInsts = 0U;
147 #if (defined (SECURITY_OVER_CSEC))
148  (void)instance;
149  static csec_state_t s_tCsecState[NO_OF_INSTS_FOR_SECURITY_PAL];
150 
151  status = SecurityAllocateInstance(g_bCsecStateIsAllocated, g_tCsecInstance, SECURITY_INSTANCE0, NO_OF_INSTS_FOR_SECURITY_PAL, &stateInsts);
152  if (STATUS_SUCCESS != status)
153  {
154  return STATUS_ERROR;
155  }
156  (void)CSEC_DRV_Init(&s_tCsecState[stateInsts]);
157  (void)CSEC_DRV_InstallCallback(config->callback, config->callbackParam);
158 #elif (defined (SECURITY_OVER_HSM))
159  (void)instance;
160  static hsm_state_t s_tHsmState[NO_OF_INSTS_FOR_SECURITY_PAL];
161 
162  status = SecurityAllocateInstance(g_bHsmStateIsAllocated, g_tHsmInstance, SECURITY_INSTANCE0, NO_OF_INSTS_FOR_SECURITY_PAL, &stateInsts);
163  if (STATUS_SUCCESS != status)
164  {
165  return STATUS_ERROR;
166  }
167  status = HSM_DRV_Init(&s_tHsmState[stateInsts]);
168  if (STATUS_SUCCESS != status)
169  {
170  return STATUS_ERROR;
171  }
172  (void)HSM_DRV_InstallCallback(config->callback, config->callbackParam);
173 #else
174  DEV_ASSERT((uint32_t)NO_OF_INSTS_FOR_SECURITY_PAL > ((uint32_t)instance));
175 #endif
176 
177  return status;
178 }
