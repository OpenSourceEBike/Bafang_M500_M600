144 {
145  uint8_t i;
146  /* Allocate one of the I2C state structure for this instance */
147  for (i = 0;i < numberOfinstances;i++)
148  {
149  if (isAllocated[i] == false)
150  {
151  instanceMapping[i] = instance->instIdx;
152  isAllocated[i] = true;
153  break;
154  }
155  }
156 
157  return i;

214 
215  /* Define I2C PAL over I2C */
216  #if (defined (I2C_OVER_LPI2C))
217  if(instance->instType == I2C_INST_TYPE_LPI2C)
218  {
219  lpi2c_master_user_config_t lpi2cConfig;
220  lpi2cConfig.slaveAddress = config->slaveAddress;
221  lpi2cConfig.is10bitAddr = config->is10bitAddr;
222  switch(config->transferType)
223  {
224  case I2C_PAL_USING_DMA: lpi2cConfig.transferType = LPI2C_USING_DMA; break;
225  case I2C_PAL_USING_INTERRUPTS : lpi2cConfig.transferType = LPI2C_USING_INTERRUPTS; break;
226  default:
227  /* Transfer type not available for PAL */
228  break;
229  }
230  lpi2cConfig.masterCallback = config->callback;
231  lpi2cConfig.callbackParam = config->callbackParam;
232 
233  switch(config->operatingMode)
234  {
235  case I2C_PAL_STANDARD_MODE: lpi2cConfig.operatingMode = LPI2C_STANDARD_MODE; break;
236  case I2C_PAL_FAST_MODE: lpi2cConfig.operatingMode = LPI2C_FAST_MODE; break;
237  #if (LPI2C_HAS_FAST_PLUS_MODE)
238  case I2C_PAL_FASTPLUS_MODE:
239  lpi2cConfig.operatingMode = LPI2C_FASTPLUS_MODE;
240  break;
241  #endif
242  #if (LPI2C_HAS_HIGH_SPEED_MODE)
243  case I2C_PAL_HIGHSPEED_MODE:
244  lpi2cConfig.operatingMode = LPI2C_HIGHSPEED_MODE;
245  break;
246  #endif
247  #if (LPI2C_HAS_ULTRA_FAST_MODE)
248  case I2C_PAL_ULTRAFAST_MODE:
249  lpi2cConfig.operatingMode = LPI2C_ULTRAFAST_MODE;
250  break;
251  #endif
252  default:
253  /* Operating mode not available for PAL */
254  DEV_ASSERT(0);
255  break;
256  }
257 
258  lpi2cConfig.baudRate = config->baudRate;
259  /* DMA channel */
260  lpi2cConfig.dmaChannel = config->dmaChannel1;
261  /* Allocate one of the LPI2C state structure for this instance */
262  index = I2CAllocateState(Lpi2cStateIsAllocated, Lpi2cStateInstanceMapping, instance, (uint8_t)NO_OF_LPI2C_INSTS_FOR_I2C);
263  status = LPI2C_DRV_MasterInit((uint32_t)instance->instIdx, &lpi2cConfig, &Lpi2cMasterState[index]);
264  }
265  #endif
266 
267  #if(defined (I2C_OVER_I2C))
268  if(instance->instType == I2C_INST_TYPE_I2C)
269  {
270  i2c_master_user_config_t i2cConfig;
271  i2cConfig.slaveAddress = config->slaveAddress;
272  switch(config->transferType)
273  {
274  case I2C_PAL_USING_DMA: i2cConfig.transferType = I2C_USING_DMA; break;
275  case I2C_PAL_USING_INTERRUPTS : i2cConfig.transferType = I2C_USING_INTERRUPTS; break;
276  default:
277  /* Transfer type not available for PAL */
278  break;
279  }
280  i2cConfig.masterCallback = config->callback;
281  i2cConfig.callbackParam = (uint8_t *)instance->instIdx;
282  i2cConfig.dmaChannel = config->dmaChannel1;
283  i2cConfig.baudRate = config->baudRate;
284  /*Allocate one of the I2C state structure for this instance */
285  index = I2CAllocateState(I2CStateIsAllocated, I2CStateInstanceMapping, instance, NO_OF_I2C_INSTS_FOR_I2C);
286  status = I2C_DRV_MasterInit((uint8_t)instance->instIdx, &i2cConfig, &I2CMasterState[index]);
287  }
288  #endif
289 
290 #if (defined(I2C_OVER_SWI2C))
291  if(instance->instType == I2C_INST_TYPE_SWI2C)
292  {
293  swi2c_master_user_config_t swi2cConfig;
294  swi2cConfig.slaveAddress = config->slaveAddress;
295  swi2cConfig. baudRate = config->baudRate;
296 
297  /* Initialize pins */
298  swi2cConfig.sclPin = ((extension_swi2c_for_i2c_t *) config->extension)->sclPin;
299  swi2cConfig.sdaPin = ((extension_swi2c_for_i2c_t *) config->extension)->sdaPin;
300  swi2cConfig.sclReadPin = ((extension_swi2c_for_i2c_t *) config->extension)->sclReadPin;
301  swi2cConfig.sdaReadPin = ((extension_swi2c_for_i2c_t *) config->extension)->sdaReadPin;
302 
303  status = SWI2C_DRV_MasterInit(&SWI2CMasterState[instance->instIdx], &swi2cConfig);
304 
305  /* Cast to avoid compiler warnings */
306  (void) index;
307  }
308 #endif
309 
310  #if(defined (I2C_OVER_FLEXIO))
311  if(instance->instType == I2C_INST_TYPE_FLEXIO)
312  {
313  flexio_i2c_master_user_config_t flexioI2CConfig;
314  flexioI2CConfig.slaveAddress = config->slaveAddress;
315  switch(config->transferType)
316  {
317  case I2C_PAL_USING_DMA: flexioI2CConfig.driverType = FLEXIO_DRIVER_TYPE_DMA; break;
318  case I2C_PAL_USING_INTERRUPTS : flexioI2CConfig.driverType = FLEXIO_DRIVER_TYPE_INTERRUPTS; break;
319  default:
320  /* Transfer type not available for PAL */
321  break;
322  }
323  flexioI2CConfig.sdaPin = ((extension_flexio_for_i2c_t*)(config->extension))->sdaPin;
324  flexioI2CConfig.sclPin = ((extension_flexio_for_i2c_t*)(config->extension))->sclPin;
325  flexioI2CConfig.callback = config->callback;
326  flexioI2CConfig.callbackParam = config->callbackParam;
327  flexioI2CConfig.baudRate = config->baudRate;
328  flexioI2CConfig.rxDMAChannel = config -> dmaChannel2;
329  flexioI2CConfig.txDMAChannel = config -> dmaChannel1;
330  status = FLEXIO_DRV_InitDevice(0U, &FlexioState);
331 
332  if(status == STATUS_SUCCESS)
333  {
334  /* Allocate one of the Flexio state structure for this instance */
335  index = I2CAllocateState(FlexioI2CStateIsAllocated, FlexioI2CStateInstanceMapping, instance, NO_OF_FLEXIO_INSTS_FOR_I2C);
336  status = FLEXIO_I2C_DRV_MasterInit(0U, &flexioI2CConfig, (flexio_i2c_master_state_t*)&FlexioI2CState[index]);
337  }
338  }
339  #endif
340 
341  return status;
342 }
