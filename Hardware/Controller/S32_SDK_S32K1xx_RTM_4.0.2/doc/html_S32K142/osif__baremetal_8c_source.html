99 static inline void osif_UpdateTickConfig(void)
100 {
101  uint32_t core_freq = 0u;
102  static bool first_init = true;
103  /* Get the correct name of the core clock */
104  clock_names_t coreclk = CORE_CLK;
105  status_t clk_status = CLOCK_SYS_GetFreq(coreclk, &core_freq);
106  DEV_ASSERT(clk_status == STATUS_SUCCESS);
107  DEV_ASSERT(core_freq > 0u);
108  (void)clk_status;
109 
110  /* For Cortex-M0 devices the systick counter is initialized with an undefined
111  value, so make sure to initialize it to 0 before starting */
112  S32_SysTick->CSR = S32_SysTick_CSR_ENABLE(0u);
113  S32_SysTick->RVR = S32_SysTick_RVR_RELOAD(core_freq / 1000u);
114  if (first_init)
115  {
116  /* only initialize CVR on the first entry, to not cause time drift */
117  S32_SysTick->CVR = S32_SysTick_CVR_CURRENT(0U);
118  first_init = false;
119  }
120  S32_SysTick->CSR = S32_SysTick_CSR_ENABLE(1u) | S32_SysTick_CSR_TICKINT(1u) | S32_SysTick_CSR_CLKSOURCE(1u);
121 }

144 static inline void osif_UpdateTickConfig(void)
145 {
146  uint32_t tick_freq = 0u;
147  PIT_Type * base = OSIF_PIT;
148 
149  /* get the clock frequency for the timer and compute ticks for 1 ms */
150  status_t clk_status = CLOCK_SYS_GetFreq(PITRTI0_CLK, &tick_freq);
151  DEV_ASSERT(clk_status == STATUS_SUCCESS);
152  DEV_ASSERT(tick_freq > 0u);
153  (void)clk_status;
154  uint32_t tick_1ms = tick_freq / 1000u;
155 
156  /* setup timer and enable interrupt */
157  base->MCR &= ~PIT_MCR_MDIS(1u); /* make sure module is started */
158  base->MCR |= PIT_MCR_FRZ(1u); /* stop in Debug */
159  base->TIMER[OSIF_PIT_CHAN_ID].LDVAL = tick_1ms;
160  base->TIMER[OSIF_PIT_CHAN_ID].TCTRL |= PIT_TCTRL_TEN(1u) | PIT_TCTRL_TIE(1u);
161  static const IRQn_Type pitIrqId[PIT_INSTANCE_COUNT][PIT_IRQS_CH_COUNT] = PIT_IRQS;
162 
163  INT_SYS_InstallHandler(pitIrqId[0U][OSIF_PIT_CHAN_ID], OSIF_PIT_IRQHandler, NULL);
164  INT_SYS_EnableIRQ(pitIrqId[0U][OSIF_PIT_CHAN_ID]);
165 }

201 /*FUNCTION**********************************************************************
202  *
203  * Function Name : OSIF_TimeDelay
204  * Description : This function blocks execution for a number of milliseconds.
205  *
206  * Implements : OSIF_TimeDelay_baremetal_Activity
207  *END**************************************************************************/
208 void OSIF_TimeDelay(const uint32_t delay)
209 {
210  osif_UpdateTickConfig();
211  uint32_t start = osif_GetCurrentTickCount();
212  uint32_t crt_ticks = osif_GetCurrentTickCount();
213  uint32_t delta = crt_ticks - start;
214  uint32_t delay_ticks = MSEC_TO_TICK(delay);
215  while (delta < delay_ticks)
216  {
217  crt_ticks = osif_GetCurrentTickCount();
218  delta = crt_ticks - start;
219  }
220 }

222 /*FUNCTION**********************************************************************
223  *
224  * Function Name : OSIF_GetMilliseconds
225  * Description : This function returns the number of miliseconds elapsed since
226  * starting the internal timer. To initialize the internal timer
227  * (Systick) in bare-metal, call either OSIF_TimeDelay or
228  * OSIF_SemaWait functions. Calling OSIF_TimeDelay(0) will initialize

231  * Implements : OSIF_GetMilliseconds_baremetal_Activity

239  * have been called, the timer is already initialized.
240  */
241  return osif_GetCurrentTickCount(); /* This assumes that 1 tick = 1 millisecond */

302 /*FUNCTION**********************************************************************
303  *
304  * Function Name : OSIF_SemaWait
305  * Description : This function performs the 'wait' (decrement) operation on a semaphore.
306  * When timeout value is 0, it's the equivalent of TryWait - try to decrement but return
307  * immediately if it fails (counter is 0).
308  *
309  * Implements : OSIF_SemaWait_baremetal_Activity
310  *END**************************************************************************/
311 status_t OSIF_SemaWait(semaphore_t * const pSem,
312  const uint32_t timeout)
313 {
314  DEV_ASSERT(pSem != NULL);
315 
316  status_t osif_ret_code = STATUS_SUCCESS;
317 
318  osif_UpdateTickConfig();
319  if ((timeout == 0u))
320  {
321  /* when the timeout is 0 the wait operation is the equivalent of try_wait,
322  meaning that if the semaphore is 0 return immediately with an error code
323  */
324  if (*pSem == 0u)
325  {
326  osif_ret_code = STATUS_TIMEOUT;
327  }
328  }
329  else
330  {
331  /* timeout is not 0 */
332  uint32_t timeoutTicks;
333  if (timeout == OSIF_WAIT_FOREVER)
334  {
335  timeoutTicks = OSIF_WAIT_FOREVER;
336  }
337  else
338  {
339  /* Convert timeout from milliseconds to ticks. */
340  timeoutTicks = MSEC_TO_TICK(timeout);
341  }
342  uint32_t start = osif_GetCurrentTickCount();
343  uint32_t end = (uint32_t)(start + timeoutTicks);
344  uint32_t max = end - start;
345  while (*pSem == 0u)
346  {
347  uint32_t crt_ticks = osif_GetCurrentTickCount();
348  uint32_t delta = crt_ticks - start;
349  if ((timeoutTicks != OSIF_WAIT_FOREVER) && (delta > max))
350  {
351  /* Timeout occured, stop waiting and return fail code */
352  osif_ret_code = STATUS_TIMEOUT;
353  break;
354  }
355  }
356  }
357 
358  if (osif_ret_code == STATUS_SUCCESS)
359  {
360  osif_DisableIrqGlobal();
361  --(*pSem);
362  osif_EnableIrqGlobal();
363  }
364 
365  return osif_ret_code;
366 }
