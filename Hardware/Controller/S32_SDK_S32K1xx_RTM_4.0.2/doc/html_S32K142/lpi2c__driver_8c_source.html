186 /*FUNCTION**********************************************************************
187  *
188  * Function Name : LPI2C_DRV_MasterQueueCmd
189  * Description : queues a command in the hardware FIFO or in the master software queue
190  *
191  *END**************************************************************************/
192 static inline void LPI2C_DRV_MasterQueueCmd(LPI2C_Type *baseAddr,
193  lpi2c_master_state_t * master,
194  lpi2c_master_command_t cmd,
195  uint8_t data)
196 {
197  DEV_ASSERT(master != NULL);
198  DEV_ASSERT(baseAddr != NULL);
199 
200  uint16_t txFIFOCount = LPI2C_Get_MasterTxFIFOCount(baseAddr);
201  uint16_t txFIFOSize = LPI2C_Get_MasterTxFIFOSize(baseAddr);
202 
203  /* Check if there is room in the hardware FIFO */
204  if (txFIFOCount < txFIFOSize)
205  {
206  LPI2C_Cmd_MasterTransmit(baseAddr, cmd, data);
207  }
208  else
209  {
210  /* Hardware FIFO full, use software FIFO */
211  DEV_ASSERT(master->cmdQueue.writeIdx < LPI2C_MASTER_CMD_QUEUE_SIZE);
212 
213  master->cmdQueue.cmd[master->cmdQueue.writeIdx] = cmd;
214  master->cmdQueue.data[master->cmdQueue.writeIdx] = data;
215  master->cmdQueue.writeIdx++;
216  }
217 }

221  *
222  * Function Name : LPI2C_DRV_MasterSendQueuedCmd
223  * Description : transfers commands from the master software queue to the hardware FIFO
224  *
225  *END**************************************************************************/
226 static inline void LPI2C_DRV_MasterSendQueuedCmd(LPI2C_Type *baseAddr, lpi2c_master_state_t * master)
227 {
228  DEV_ASSERT(master != NULL);
229  DEV_ASSERT(baseAddr != NULL);
230 
231  uint16_t txFIFOCount = LPI2C_Get_MasterTxFIFOCount(baseAddr);
232  uint16_t txFifoSize = LPI2C_Get_MasterTxFIFOSize(baseAddr);
233 
234  while ((!LPI2C_DRV_MasterCmdQueueEmpty(master)) && (txFIFOCount < txFifoSize))
235  {
236  LPI2C_Cmd_MasterTransmit(baseAddr,
237  master->cmdQueue.cmd[master->cmdQueue.readIdx],
238  master->cmdQueue.data[master->cmdQueue.readIdx]);
239  master->cmdQueue.readIdx++;
240 
241  txFIFOCount = LPI2C_Get_MasterTxFIFOCount(baseAddr);
242  }
243 
244  if (LPI2C_DRV_MasterCmdQueueEmpty(master))
245  {
246  /* Reset queue */
247  LPI2C_DRV_MasterResetQueue(master);
248  }
249 }

252 /*FUNCTION**********************************************************************
253  *
254  * Function Name : LPI2C_DRV_MasterSendAddress
255  * Description : send start event and slave address
256  * parameter receive specifies the direction of the transfer
257  *
258  *END**************************************************************************/
259 static void LPI2C_DRV_MasterSendAddress(LPI2C_Type *baseAddr,
260  lpi2c_master_state_t * master,
261  bool receive)
262 {
263  uint8_t addrByte;
264  lpi2c_master_command_t startCommand;
265 
266  DEV_ASSERT(master != NULL);
267  DEV_ASSERT(baseAddr != NULL);
268 
269 #if(LPI2C_HAS_HIGH_SPEED_MODE)
270  if ((master->operatingMode == LPI2C_HIGHSPEED_MODE) && (master->highSpeedInProgress == false))
271  {
272  /* Initiating High-speed mode - send master code first */
273  LPI2C_DRV_MasterQueueCmd(baseAddr, master, LPI2C_MASTER_COMMAND_START_NACK, master->masterCode);
274  master->highSpeedInProgress = true;
275  }
276 
277  if (master->highSpeedInProgress == true)
278  {
279  /* Use high-speed settings after start event in High Speed mode */
280  startCommand = LPI2C_MASTER_COMMAND_START_HS;
281  }
282  else
283 #endif
284  {
285  /* Normal START command */
286  startCommand = LPI2C_MASTER_COMMAND_START;
287  }
288 
289  if (master->is10bitAddr)
290  {
291  /* 10-bit addressing */
292  /* First address byte: 1111 0XXD, where XX are bits 10 and 9 of address, and D = 0(transmit) */
293  addrByte = (uint8_t)(0xF0U + ((master->slaveAddress >> 7U) & 0x6U) + (uint8_t)0U);
294  LPI2C_DRV_MasterQueueCmd(baseAddr, master, startCommand, addrByte);
295  /* Second address byte: Remaining 8 bits of address */
296  addrByte = (uint8_t)(master->slaveAddress & 0xFFU);
297  LPI2C_DRV_MasterQueueCmd(baseAddr, master, LPI2C_MASTER_COMMAND_TRANSMIT, addrByte);
298  if (receive == true)
299  {
300  /* Receiving from 10-bit slave - must send repeated start and resend first address byte */
301  /* First address byte: 1111 0XXD, where XX are bits 10 and 9 of address, and D = 1 (receive) */
302  addrByte = (uint8_t)(0xF0U + ((master->slaveAddress >> 7U) & 0x6U) + (uint8_t)1U);
303  LPI2C_DRV_MasterQueueCmd(baseAddr, master, startCommand, addrByte);
304  }
305  }
306  else
307  {
308  /* 7-bit addressing */
309  /* Address byte: slave 7-bit address + D = 0(transmit) or 1 (receive) */
310  addrByte = (uint8_t)((master->slaveAddress << 1U) + (uint8_t)receive);
311  LPI2C_DRV_MasterQueueCmd(baseAddr, master, startCommand, addrByte);
312  }
313 }

348  * Function Name : LPI2C_DRV_MasterEndTransfer
349  * Description : ends current transmission or reception
350  *
351  *END**************************************************************************/
352 static void LPI2C_DRV_MasterEndTransfer(LPI2C_Type *baseAddr,
353  lpi2c_master_state_t * master,
354  bool sendStop,
355  bool resetFIFO)
356 {
357  DEV_ASSERT(master != NULL);
358  DEV_ASSERT(baseAddr != NULL);
359 
360  /* Disable all events */
361  LPI2C_Set_MasterInt(baseAddr, LPI2C_MASTER_FIFO_ERROR_INT |
362  LPI2C_MASTER_ARBITRATION_LOST_INT |
363  LPI2C_MASTER_NACK_DETECT_INT |
364  LPI2C_MASTER_TRANSMIT_DATA_INT |
365  LPI2C_MASTER_RECEIVE_DATA_INT,
366  false);
367 
368  if (master->transferType == LPI2C_USING_DMA)
369  {
370  /* Stop DMA channel */
371  (void)EDMA_DRV_StopChannel(master->dmaChannel);
372 
373  /* Disable LPI2C DMA request. */
374  if (master->rxSize != (uint16_t)0)
375  {
376  (void)LPI2C_Set_MasterRxDMA(baseAddr, false);
377  }
378  else
379  {
380  (void)LPI2C_Set_MasterTxDMA(baseAddr, false);
381  }
382  }
383 
384  if (resetFIFO == true)
385  {
386  /* Reset FIFOs if requested */
387  LPI2C_Reset_MasterTxFIFOCmd(baseAddr);
388  LPI2C_Reset_MasterRxFIFOCmd(baseAddr);
389  }
390 
391  /* Queue STOP command if requested */
392  if (sendStop == true)
393  {
394  LPI2C_Cmd_MasterTransmit(baseAddr, LPI2C_MASTER_COMMAND_STOP, 0U);
395 #if(LPI2C_HAS_HIGH_SPEED_MODE)
396  master->highSpeedInProgress = false; /* High-speed transfers end at STOP condition */
397 #endif
398  }
399 
400  master->txBuff = NULL;
401  master->txSize = 0;
402  master->rxBuff = NULL;
403  master->rxSize = 0;
404  master->i2cIdle = true;
405 }

571  * Function Name : LPI2C_DRV_MasterStartDmaTransfer
572  * Description : starts the DMA transfer for master
573  *
574  *END**************************************************************************/
575 static void LPI2C_DRV_MasterStartDmaTransfer(uint32_t instance)
576 {
577  LPI2C_Type *baseAddr = g_lpi2cBase[instance];
578  lpi2c_master_state_t *master = g_lpi2cMasterStatePtr[instance];
579  lpi2c_dma_transfer_params_t dmaTransParams;
580  bool receive = false;
581 
582  dmaTransParams.dmaChannel = master->dmaChannel;
583  if (master->txSize > 0U)
584  {
585  /* Configure watermarks for transmit DMA for master */
586  uint32_t txBytes = LPI2C_Get_MasterTxFIFOSize(baseAddr);
587  if (txBytes > master->txSize)
588  {
589  txBytes = master->txSize;
590  }
591  LPI2C_Set_MasterTxFIFOWatermark(baseAddr, (uint16_t)(txBytes - 1U));
592 
593  dmaTransParams.dmaTransferType = EDMA_TRANSFER_MEM2PERIPH;
594  dmaTransParams.i2cDataRegAddr = (uint32_t)(&(baseAddr->MTDR));
595  dmaTransParams.bufferTransfer = (uint8_t *)master->txBuff;
596  dmaTransParams.transferDirection = LPI2C_TX_REQ;
597  dmaTransParams.transferSize = master->txSize;
598 
599  }
600  else
601  {
602  /* Configure watermarks for receive DMA for master */
603  LPI2C_Set_MasterRxFIFOWatermark(baseAddr, 0U);
604 
605  receive = true;
606 
607  dmaTransParams.dmaTransferType = EDMA_TRANSFER_PERIPH2MEM;
608  dmaTransParams.i2cDataRegAddr = (uint32_t)(&(baseAddr->MRDR));
609  dmaTransParams.bufferTransfer = master->rxBuff;
610  dmaTransParams.transferDirection = LPI2C_RX_REQ;
611  dmaTransParams.transferSize = master->rxSize;
612  }
613 
614  (void)LPI2C_DRV_ConfigureDmaTransfer(instance, &dmaTransParams);
615 
616  /* Disable DMA requests for channel when transfer is done */
617  EDMA_DRV_DisableRequestsOnTransferComplete(dmaTransParams.dmaChannel, true);
618 
619  /* Call callback function when all the bytes were transfered. */
620  (void)EDMA_DRV_InstallCallback(dmaTransParams.dmaChannel, (LPI2C_DRV_MasterCompleteDMATransfer), (void*)(instance));
621 
622  /* Start channel */
623  (void)EDMA_DRV_StartChannel(dmaTransParams.dmaChannel);
624 
625  LPI2C_DRV_MasterSendAddress(baseAddr, master, receive);
626 
627  /* Enable transmit/receive DMA requests */
628  if (master->txSize > (uint32_t)0U)
629  {
630  (void)LPI2C_Set_MasterTxDMA(baseAddr, true);
631  }
632  else
633  {
634  LPI2C_DRV_MasterQueueCmd(baseAddr, master, LPI2C_MASTER_COMMAND_RECEIVE, (uint8_t)(master->rxSize - 1U));
635  (void)LPI2C_Set_MasterRxDMA(baseAddr, true);
636  }
637 }

792  * Description : waits for the end of a blocking transfer
793  *
794  *END**************************************************************************/

799  lpi2c_master_state_t *master;

806  osifError = OSIF_SemaWait(&(master->idleSemaphore), timeout);
807

809  {
810  /* If master is sending data transfer is aborted now in case timeout occurred */
811  if(master->txSize != 0U)
812  {
813  LPI2C_DRV_MasterEndTransfer(baseAddr, master, false, true);
814 
815  master->status = STATUS_TIMEOUT;
816  }

821  /* Stop DMA channel and activate interrupts */

823  }

826  LPI2C_Set_MasterInt(baseAddr, (uint32_t)LPI2C_MASTER_RECEIVE_DATA_INT, false);

829  rxFifoFill = LPI2C_Get_MasterRxFIFOCount(baseAddr);

837  master->blocking = false;

844  {
845  master->abortedTransfer = true;
846  LPI2C_Set_MasterRxFIFOWatermark(baseAddr, rxFifoFill);

848  master->status = STATUS_TIMEOUT;

858  }
859 
860  master->abortedTransfer = false;
861  }
862  }

865  master->blocking = false;

867 }

1284  *END**************************************************************************/

1286 {

1288  const lpi2c_master_state_t *master;

1303  DEV_ASSERT(retVal == STATUS_SUCCESS);
1304  DEV_ASSERT(inputClock > 0U);
1305 
1306  /* Ignoring the glitch filter, the baud rate formula is:
1307  SCL_freq = Input_freq / (2^PRESCALER * (CLKLO + CLKHI + 2))

1309  prescaler = (uint32_t)LPI2C_Get_MasterPrescaler(baseAddr);

1311  clkLo = (uint32_t)LPI2C_Get_MasterClockLowPeriod(baseAddr);

1330 /*FUNCTION**********************************************************************
1331  *
1332  * Function Name : LPI2C_DRV_MasterSetBaudRate
1333  * Description : set the baud rate for any subsequent I2C communication
1334  *
1335  * Implements : LPI2C_DRV_MasterSetBaudRate_Activity
1336  *END**************************************************************************/
1337 status_t LPI2C_DRV_MasterSetBaudRate(uint32_t instance,
1338  const lpi2c_mode_t operatingMode,
1339  const lpi2c_baud_rate_params_t baudRate)
1340 {
1341  LPI2C_Type *baseAddr;
1342  const lpi2c_master_state_t * master;
1343  status_t retVal;
1344  uint32_t inputClock;
1345  uint32_t minPrescaler = 0U;
1346  uint32_t prescaler;
1347  uint32_t clkTotal;
1348  uint32_t clkLo;
1349  uint32_t clkHi;
1350  uint32_t setHold;
1351  uint32_t dataVd;
1352 
1353  DEV_ASSERT(instance < LPI2C_INSTANCE_COUNT);
1354 
1355  baseAddr = g_lpi2cBase[instance];
1356  master = g_lpi2cMasterStatePtr[instance];
1357  DEV_ASSERT(master != NULL);
1358 
1359  /* Check if driver is busy */
1360  if(!master->i2cIdle)
1361  {
1362  return STATUS_BUSY;
1363  }
1364 
1365  /* Get the protocol clock frequency */
1366  retVal = CLOCK_SYS_GetFreq(g_lpi2cClock[instance], &inputClock);
1367  DEV_ASSERT(retVal == STATUS_SUCCESS);
1368  DEV_ASSERT(inputClock > 0U);
1369 
1370  /* Disable master */
1371  LPI2C_Set_MasterEnable(baseAddr, false);
1372 
1373  /* Ignoring the glitch filter, the baud rate formula is:
1374  SCL_freq = Input_freq / (2^PRESCALER * (CLKLO + CLKHI + 2))
1375  Assume CLKLO = CLKHI, SETHOLD = CLKHI, DATAVD = CLKHI/2
1376  */
1377 
1378  if (baudRate.baudRate != 0U)
1379  {
1380  /* Compute minimum prescaler for which CLKLO and CLKHI values are in valid range. Always round up. */
1381  minPrescaler = ((inputClock - 1U) / ((baudRate.baudRate) * (CLKHI_MAX_VALUE + CLKLO_MAX_VALUE + 2U))) + (uint32_t)1U;
1382  for (prescaler = 0U; prescaler < 7U; prescaler++)
1383  {
1384  if (((uint32_t)1U << prescaler) >= minPrescaler)
1385  {
1386  break;
1387  }
1388  }
1389 
1390  /* Compute CLKLO and CLKHI values for this prescaler. Round to nearest integer. */
1391  clkTotal = (inputClock + ((baudRate.baudRate << prescaler) >> 1U)) / (baudRate.baudRate << prescaler);
1392  }
1393  else
1394  {
1395  prescaler = 7U;
1396  clkTotal = (CLKHI_MAX_VALUE + CLKLO_MAX_VALUE + 2U);
1397  }
1398 
1399  if (clkTotal > (CLKHI_MAX_VALUE + CLKLO_MAX_VALUE + 2U))
1400  {
1401  clkTotal = (CLKHI_MAX_VALUE + CLKLO_MAX_VALUE + 2U);
1402  }
1403 
1404  /*
1405  * If we try to compute clk high and low values using clkTotal equal with 0
1406  * (this is the case when the baudrate is 0), we will get negative values for
1407  * them, so we set them to 0 for this case.
1408  */
1409  if (clkTotal <= 1U)
1410  {
1411  clkHi = 0U;
1412  clkLo = 0U;
1413  }
1414  else
1415  {
1416  clkHi = (clkTotal - 2U) / 2U;
1417  clkLo = clkTotal - 2U - clkHi;
1418  }
1419 
1420  if (clkHi < CLKHI_MIN_VALUE)
1421  {
1422  clkHi = CLKHI_MIN_VALUE;
1423  }
1424  if (clkLo < CLKLO_MIN_VALUE)
1425  {
1426  clkLo = CLKLO_MIN_VALUE;
1427  }
1428 
1429  /* Compute DATAVD and SETHOLD */
1430  setHold = clkHi;
1431  dataVd = clkHi >> 1U;
1432  if (setHold < SETHOLD_MIN_VALUE)
1433  {
1434  setHold = SETHOLD_MIN_VALUE;
1435  }
1436  if (dataVd < DATAVD_MIN_VALUE)
1437  {
1438  dataVd = DATAVD_MIN_VALUE;
1439  }
1440 
1441  /* Apply settings */
1442  LPI2C_Set_MasterPrescaler(baseAddr, (lpi2c_master_prescaler_t)prescaler);
1443  LPI2C_Set_MasterDataValidDelay(baseAddr, (uint8_t)dataVd);
1444  LPI2C_Set_MasterSetupHoldDelay(baseAddr, (uint8_t)setHold);
1445  LPI2C_Set_MasterClockHighPeriod(baseAddr, (uint8_t)clkHi);
1446  LPI2C_Set_MasterClockLowPeriod(baseAddr, (uint8_t)clkLo);
1447 
1448 #if(LPI2C_HAS_HIGH_SPEED_MODE)
1449  if (operatingMode == LPI2C_HIGHSPEED_MODE)
1450  {
1451  /* Compute settings for High-speed baud rate */
1452  /* Compute High-speed CLKLO and CLKHI values for the same prescaler. Round to nearest integer. */
1453  clkTotal = (inputClock + ((baudRate.baudRateHS << prescaler) >> 1U)) / (baudRate.baudRateHS << prescaler);
1454  if (clkTotal > (CLKHI_MAX_VALUE + CLKLO_MAX_VALUE + 2U))
1455  {
1456  clkTotal = (CLKHI_MAX_VALUE + CLKLO_MAX_VALUE + 2U);
1457  }
1458 
1459  clkHi = (clkTotal - 2U) / 3U;
1460  clkLo = clkTotal - 2U - clkHi;
1461  if (clkHi < CLKHI_MIN_VALUE)
1462  {
1463  clkHi = CLKHI_MIN_VALUE;
1464  }
1465  if (clkLo < CLKLO_MIN_VALUE)
1466  {
1467  clkLo = CLKLO_MIN_VALUE;
1468  }
1469 
1470  /* Compute High-speed DATAVD and SETHOLD */
1471  setHold = clkHi;
1472  dataVd = clkHi >> 1U;
1473  if (setHold < SETHOLD_MIN_VALUE)
1474  {
1475  setHold = SETHOLD_MIN_VALUE;
1476  }
1477  if (dataVd < DATAVD_MIN_VALUE)
1478  {
1479  dataVd = DATAVD_MIN_VALUE;
1480  }
1481 
1482  /* Apply High-speed settings */
1483  LPI2C_Set_MasterDataValidDelayHS(baseAddr, (uint8_t)dataVd);
1484  LPI2C_Set_MasterSetupHoldDelayHS(baseAddr, (uint8_t)setHold);
1485  LPI2C_Set_MasterClockHighPeriodHS(baseAddr, (uint8_t)clkHi);
1486  LPI2C_Set_MasterClockLowPeriodHS(baseAddr, (uint8_t)clkLo);
1487  }
1488 #endif
1489 
1490  /* Perform other settings related to the chosen operating mode */
1491  LPI2C_DRV_MasterSetOperatingMode(instance, operatingMode);
1492 
1493  /* Re-enable master */
1494  LPI2C_Set_MasterEnable(baseAddr, true);
1495 
1496  (void) minPrescaler;
1497  (void)master;
1498  (void)retVal;
1499 
1500  return STATUS_SUCCESS;
1501 }
