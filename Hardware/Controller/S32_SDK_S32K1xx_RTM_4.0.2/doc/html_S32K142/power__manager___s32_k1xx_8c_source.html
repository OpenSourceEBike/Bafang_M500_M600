90 /*******************************************************************************
91  *
92  * It is expected that prior to the POWER_SYS_Init() call the write-once protection
93  * register was configured appropriately allowing entry to all required low power
94  * modes.
95  * The following is an example of how to set up two power modes and one
96  * callback, and initialize the Power manager with structures containing their settings.
97  * The example shows two possible ways the configuration structures can be stored
98  * (ROM or RAM), although it is expected that they will be placed in the read-only
99  * memory to save the RAM space. (Note: In the example it is assumed that the programmed chip
100  * doesn't support any optional power options described in the power_manager_user_config_t)
101  * :
102  * @code
103  *
104  * power_manager_user_config_t vlprConfig = { vlprConfig power mode configuration
105  * .powerMode = POWER_MANAGER_VLPR,
106  * .sleepOnExitValue = false,
107  * };
108  *
109  * power_manager_user_config_t stopConfig = { stopConfig power mode configuration
110  * .powerMode = POWER_MANAGER_STOP,
111  * .sleepOnExitValue = false,
112  * };
113  *
114  * power_manager_user_config_t const * powerConfigsArr[] = { Power mode configurations array
115  * &vlprConfig,
116  * &stopConfig
117  * };
118  *
119  * power_manager_callback_user_config_t callbackCfg0 = { Callback configuration structure callbackCfg0
120  * .callbackFunction = &callback0,
121  * .callbackType = POWER_MANAGER_CALLBACK_BEFORE_AFTER,
122  * .callbackData = (void *)0,
123  * };
124  *
125  * power_manager_callback_user_config_t const * callbacksConfigsArr[] = { Callback configuration structures array
126  * &callbackCfg0
127  * };
128  *
129  * status_t callback0(power_manager_notify_struct_t * notify, Definition of power manager callback
130  * power_manager_callback_data_t * dataPtr)
131  * {
132  * status_t ret = STATUS_SUCCESS;
133  * ...
134  * return ret;
135  * }
136  *
137  * int main(void) Main function
138  * {
139  * status_t ret = STATUS_SUCCESS;
140  *
141  * Calling of init method
142  * POWER_SYS_Init(&powerConfigsArr, 2U, &powerStaticCallbacksConfigsArr, 1U);
143  *
144  * Switch to VLPR mode
145  * ret = POWER_SYS_SetMode(MODE_VLPR,POWER_MANAGER_POLICY_AGREEMENT);
146  *
147  * if (ret != STATUS_SUCCESS)
148  * {
149  * return -1;
150  * }
151  * return 0;
152  * }
153  *
154  * @endcode
155  *
156  *END**************************************************************************/
157 status_t POWER_SYS_DoInit(void)
158 {
159  uint8_t k = 0U;
160 
161  smc_power_mode_protection_config_t powerModeProtConfig;
162 #if FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE
163  powerModeProtConfig.hsrunProt = false;
164 #endif /* #if FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE */
165  powerModeProtConfig.vlpProt = false;
166 
167  for (k = 0;k < gPowerManagerState.configsNumber;k++)
168  {
169  const power_manager_user_config_t * const config = (*gPowerManagerState.configs)[k];
170 #if FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE
171  if (config->powerMode == POWER_MANAGER_HSRUN)
172  {
173  powerModeProtConfig.hsrunProt = true; /* High speed mode is allowed. */
174  }
175 #endif /* #if FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE */
176  if ((config->powerMode == POWER_MANAGER_VLPR) || (config->powerMode == POWER_MANAGER_VLPS))
177  {
178  powerModeProtConfig.vlpProt = true; /* Very low power mode is allowed. */
179  }
180  }
181 
182  /* Biasing disabled, core logic can run in full performance */
183  PMC_DisableBiasen(PMC);
184 
185  /* Very low power modes and high speed mode are not protected. */
186  SMC_SetProtectionMode(SMC, &powerModeProtConfig);
187  /* Get all clock source were enabled. This one was used for update initialize clock when CPU
188  came back RUN mode from very low power mode */
189  POWER_DRV_GetEnableClockSrc();
190  return STATUS_SUCCESS;
191 }

217 status_t POWER_SYS_DoSetMode(const power_manager_user_config_t * const configPtr)
218 {
219  status_t returnCode; /* Function return */
220 
221  /* Check whether the power mode is a sleeping or a running power mode */
222  if (configPtr->powerMode <= POWER_MANAGER_VLPR)
223  {
224  /* Switch to a running power mode */
225  returnCode = POWER_SYS_SwitchToRunningPowerMode(configPtr);
226  }
227  else
228  {
229  /* Switch to a sleeping power mode */
230  returnCode = POWER_SYS_SwitchToSleepingPowerMode(configPtr);
231  }
232 
233  return returnCode;
234 }

236 /*FUNCTION**********************************************************************
237  *
238  * Function Name : POWER_SYS_GetCurrentMode
239  * Description : Returns currently running power mode.
240  *
241  * Implements POWER_SYS_GetCurrentMode_Activity
242  *
243  *END**************************************************************************/
244 power_manager_modes_t POWER_SYS_GetCurrentMode(void)
245 {
246  power_manager_modes_t retVal;
247 
248  switch (SMC_GetPowerModeStatus(SMC))
249  {
250 #if FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE
251  /* High speed run mode */
252  case STAT_HSRUN:
253  retVal = POWER_MANAGER_HSRUN;
254  break;
255 #endif
256  /* Run mode */
257  case STAT_RUN:
258  retVal = POWER_MANAGER_RUN;
259  break;
260  /* Very low power run mode */
261  case STAT_VLPR:
262  retVal = POWER_MANAGER_VLPR;
263  break;
264  /* This should never happen - core has to be in some run mode to execute code */
265  default:
266  retVal = POWER_MANAGER_MAX;
267  break;
268  }
269 
270  return retVal;
271 }

273 /*FUNCTION**********************************************************************
274  *
275  * Function Name : POWER_SYS_SwitchToRunningPowerMode
276  * Description :Internal function used by POWER_SYS_SetMode function to switch to a running power mode
277  * configPtr pointer to the requested user-defined power mode configuration.
278  * System clock source must be SIRC or SOSC in Run mode before transition very low power run mode.
279  * Update initialization or default clock source in run mode when came back from very low power run mode.
280  *
281  *END**************************************************************************/
282 static status_t POWER_SYS_SwitchToRunningPowerMode(const power_manager_user_config_t * const configPtr)
283 {
284  smc_power_mode_config_t modeConfig; /* SMC hardware layer configuration structure */
285  power_mode_stat_t currentMode = SMC_GetPowerModeStatus(SMC);
286  status_t returnCode = STATUS_SUCCESS;
287 
288  /* Configure the running mode */
289  switch (configPtr->powerMode)
290  {
291 #if FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE
292  /* High speed run mode */
293  case POWER_MANAGER_HSRUN:
294  /* High speed run mode can be entered only from Run mode */
295  if (currentMode != STAT_HSRUN)
296  {
297  if (currentMode != STAT_RUN)
298  {
299  modeConfig.powerModeName = POWER_MANAGER_RUN;
300  /* Switch the mode */
301  returnCode = SMC_SetPowerMode(SMC, &modeConfig);
302  }
303  if (returnCode == STATUS_SUCCESS)
304  {
305  returnCode = POWER_SYS_EnterHsrunMode();
306  }
307  }
308  else
309  {
310  returnCode = STATUS_SUCCESS;
311  }
312  break;
313 #endif /* if FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE */
314  /* Run mode */
315  case POWER_MANAGER_RUN:
316  if (currentMode != STAT_RUN)
317  {
318  modeConfig.powerModeName = POWER_MANAGER_RUN;
319  /* Switch the mode */
320  returnCode = SMC_SetPowerMode(SMC, &modeConfig);
321  }
322  if ((returnCode == STATUS_SUCCESS) && changeClkVlp)
323  {
324  /* Enable all clock source */
325  POWER_DRV_EnableVlpClockSrc();
326  /* Update initialize clock configuration */
327  returnCode = POWER_DRV_UpdateInitClk(&sysClkConfig);
328  if (returnCode == STATUS_SUCCESS)
329  {
330  changeClkVlp = false;
331  }
332  }
333 
334  break;
335  /* Very low power run mode */
336  case POWER_MANAGER_VLPR:
337 
338  if (currentMode != STAT_VLPR)
339  {
340  /* Very low power run mode can be entered only from Run mode */
341  if (SMC_GetPowerModeStatus(SMC) != STAT_RUN)
342  {
343  modeConfig.powerModeName = POWER_MANAGER_RUN;
344  /* Switch the mode */
345  returnCode = SMC_SetPowerMode(SMC, &modeConfig);
346  }
347  if (STATUS_SUCCESS == returnCode)
348  {
349  if (!changeClkVlp)
350  {
351  CLOCK_DRV_GetSystemClockSource(&sysClkConfig);
352  }
353  returnCode = POWER_DRV_SwitchVlprClk(&sysClkConfig);
354  if (STATUS_SUCCESS == returnCode)
355  {
356  changeClkVlp = true;
357  modeConfig.powerModeName = POWER_MANAGER_VLPR;
358  /* Disable all clock source except SIRC */
359  POWER_DRV_DisableVlpClockSrc();
360  /* Switch the mode */
361  returnCode = SMC_SetPowerMode(SMC, &modeConfig);
362  }
363  }
364  }
365  else
366  {
367  returnCode = STATUS_SUCCESS;
368  }
369 
370  break;
371  /* Wait mode */
372  default:
373  /* invalid power mode */
374  returnCode = STATUS_UNSUPPORTED;
375  modeConfig.powerModeName = POWER_MANAGER_MAX;
376  break;
377  }
378 
379  return returnCode;
380 }

382 /*FUNCTION**********************************************************************
383  *
384  * Function Name : POWER_SYS_SwitchToSleepingPowerMode
385  * Description :Internal function used by POWER_SYS_SetMode function to switch to a sleeping power mode
386  * configPtr pointer to the requested user-defined power mode configuration
387  *
388  *END**************************************************************************/
389 static status_t POWER_SYS_SwitchToSleepingPowerMode(const power_manager_user_config_t * const configPtr)
390 {
391  smc_power_mode_config_t modeConfig; /* SMC hardware layer configuration structure */
392  status_t returnCode = STATUS_SUCCESS; /* return value */
393  power_mode_stat_t pwrModeStat = SMC_GetPowerModeStatus(SMC); /* power mode stat */
394 
395  /* Configure the hardware layer */
396  switch (configPtr->powerMode)
397  {
398 #if FEATURE_SMC_HAS_WAIT_VLPW
399  /* Wait mode */
400  case POWER_MANAGER_WAIT:
401  /* Wait mode can be entered only from Run mode */
402  if (pwrModeStat != STAT_RUN)
403  {
404  modeConfig.powerModeName = POWER_MANAGER_RUN;
405  /* Switch the mode */
406  returnCode = SMC_SetPowerMode(SMC, &modeConfig);
407  }
408 
409  modeConfig.powerModeName = POWER_MANAGER_WAIT;
410  break;
411  /* Very low power wait mode */
412  case POWER_MANAGER_VLPW:
413  /* Very low power wait mode can be entered only from Very low power run mode */
414  if (pwrModeStat != STAT_VLPR)
415  {
416  modeConfig.powerModeName = POWER_MANAGER_VLPR;
417  /* Switch the mode */
418  returnCode = SMC_SetPowerMode(SMC, &modeConfig);
419  }
420 
421  modeConfig.powerModeName = POWER_MANAGER_VLPW;
422  break;
423 #endif /* if FEATURE_SMC_HAS_WAIT_VLPW */
424 #if FEATURE_SMC_HAS_PSTOPO
425  /* Partial stop modes */
426  case POWER_MANAGER_PSTOP1:
427  /* fall-through */
428  case POWER_MANAGER_PSTOP2:
429  /* fall-through */
430 #endif
431 #if FEATURE_SMC_HAS_STOPO
432  /* Stop modes */
433  case POWER_MANAGER_STOP1:
434  /* fall-through */
435  case POWER_MANAGER_STOP2:
436  /* Stop1 and Stop2 mode can be entered only from Run mode */
437  if (pwrModeStat != STAT_RUN)
438  {
439  modeConfig.powerModeName = POWER_MANAGER_RUN;
440  /* Switch the mode */
441  returnCode = SMC_SetPowerMode(SMC, &modeConfig);
442  }
443 
444  modeConfig.powerModeName = configPtr->powerMode;
445 #endif /* #if FEATURE_SMC_HAS_STOPO */
446 #if FEATURE_SMC_HAS_PSTOPO
447  modeConfig.pstopOption = true;
448  /* Set the partial stop option value */
449  if (POWER_MANAGER_PSTOP1 == configPtr->powerMode)
450  {
451  modeConfig.pstopOptionValue = SMC_PSTOP_STOP1;
452  }
453  else if (POWER_MANAGER_PSTOP2 == configPtr->powerMode)
454  {
455  modeConfig.pstopOptionValue = SMC_PSTOP_STOP2;
456  }
457  else
458  {
459  modeConfig.pstopOptionValue = SMC_PSTOP_STOP;
460  }
461 
462 #endif /* if FEATURE_SMC_HAS_PSTOPO */
463 #if FEATURE_SMC_HAS_STOPO
464  /* Set the stop option value */
465  if (POWER_MANAGER_STOP1 == configPtr->powerMode)
466  {
467  modeConfig.stopOptionValue = SMC_STOP1;
468  }
469  else
470  {
471  modeConfig.stopOptionValue = SMC_STOP2;
472  }
473 #endif /* if FEATURE_SMC_HAS_STOPO */
474  break;
475  /* Very low power stop mode */
476  case POWER_MANAGER_VLPS:
477  /* Very low power stop mode can be entered only from Run mode or Very low power run mode*/
478  if ((pwrModeStat != STAT_RUN) && (pwrModeStat != STAT_VLPR))
479  {
480  modeConfig.powerModeName = POWER_MANAGER_RUN;
481  returnCode = SMC_SetPowerMode(SMC, &modeConfig);
482  }
483 
484  if (returnCode == STATUS_SUCCESS)
485  {
486  if (POWER_SYS_GetCurrentMode() == POWER_MANAGER_RUN)
487  {
488  /* Get current source clock */
489  if (!changeClkVlp)
490  {
491  CLOCK_DRV_GetSystemClockSource(&sysClkConfig);
492  }
493  returnCode = POWER_DRV_SwitchVlprClk(&sysClkConfig);
494  if (STATUS_SUCCESS == returnCode)
495  {
496  changeClkVlp = true;
497  modeConfig.powerModeName = POWER_MANAGER_VLPS;
498  /* Disable all clock source except SIRC */
499  POWER_DRV_DisableVlpClockSrc();
500  }
501  }
502  }
503  modeConfig.powerModeName = POWER_MANAGER_VLPS;
504  break;
505  default:
506  /* invalid power mode */
507  returnCode = STATUS_UNSUPPORTED;
508  modeConfig.powerModeName = POWER_MANAGER_MAX;
509  break;
510  }
511 
512  if (STATUS_SUCCESS == returnCode)
513  {
514  /* Configure ARM core what to do after interrupt invoked in (deep) sleep state */
515  if (configPtr->sleepOnExitValue)
516  {
517  /* Go back to (deep) sleep state on ISR exit */
518  S32_SCB->SCR |= S32_SCB_SCR_SLEEPONEXIT_MASK;
519  }
520  else
521  {
522  /* Do not re-enter (deep) sleep state on ISR exit */
523  S32_SCB->SCR &= ~(S32_SCB_SCR_SLEEPONEXIT_MASK);
524  }
525 
526  /* Switch the mode */
527  if (SMC_SetPowerMode(SMC, &modeConfig) != STATUS_SUCCESS)
528  {
529  returnCode = STATUS_MCU_TRANSITION_FAILED;
530  }
531  }
532 
533  return returnCode;
534 }

536 /*FUNCTION**********************************************************************
537  *
538  * Function Name : POWER_DRV_SwitchVlprClk
539  * Description : This function will change system clock in run mode before MCU enter very low power run mode.
540  *
541  *
542  *END**************************************************************************/
543 static status_t POWER_DRV_SwitchVlprClk(const sys_clk_config_t * const sysClock)
544 {
545  status_t retCode = STATUS_SUCCESS;
546  sys_clk_config_t sysClkVlprConfig;
547  clock_names_t currentSystemClockSource = sysClock->src;
548 
549  if (currentSystemClockSource != SIRC_CLK)
550  {
551  /* Set SIRC the system clock source */
552  sysClkVlprConfig.src = SIRC_CLK;
553  sysClkVlprConfig.dividers[0U] = (uint16_t)(SCG_Get_VCCR_DIVCORE(SCG) + 1U); /* Core clock divider, do not divide */
554  sysClkVlprConfig.dividers[1U] = (uint16_t)(SCG_Get_VCCR_DIVBUS(SCG) + 1U); /* Bus clock divider, do not divide */
555  sysClkVlprConfig.dividers[2U] = (uint16_t)(SCG_Get_VCCR_DIVSLOW(SCG) + 1U); /* Bus clock divider, do not divide */
556 
557  retCode = CLOCK_DRV_SetSystemClock(NULL,&sysClkVlprConfig);
558  }
559 
560  return retCode;
561 }
