121 /*TIMER MODE CONFIGURATION******************************************************
122  *
123  * Timer Mode - Prescaler settings calculations
124  * --------------------------------------------
125  *
126  * Timer Mode configuration takes a period (timeout) value expressed in
127  * micro-seconds. To convert this to LPTMR prescaler (and compare value)
128  * settings, the closest match must be found.
129  * For best precision, the lowest prescaler that allows the corresponding
130  * compare value to fit in the 16-bit register will be chosen.
131  *
132  * Algorithm for choosing prescaler and compare values:
133  * =============================================================================
134  * In: tper_us (period in microseconds), fclk (input clock frequency in Hertz)
135  * Out: nticks (timer ticks), p (prescaler coefficient, 2^p = prescaler value)
136  * ---
137  * 1) Compute nn = tper_us * fclk / 1000000
138  * 2) for p = 0..16
139  * 2.1) nticks = nn / 2^p
140  * 2.2) if nticks < 0x10000
141  * 2.2.1) STOP, found nticks and p
142  * 3) nticks = 0xFFFF, p = 16
143  * =============================================================================
144  *
145  * A few names used throughout the static functions affecting Timer mode:
146  * nn - total number of timer ticks (undivided, unprescaled) that is necessary
147  * for a particular timeout.
148  * nn = (tper_us * fclk) / 1000000 = nticks * npresc
149  *
150  * tper_us - a period (or timeout) expressed in microsecond units. In most
151  * functions will be denoted as 'us' for microseconds.
152  *
153  * nticks - number of timer ticks that is necessary for a particular timeout,
154  * after prescaling
155  *
156  * npresc - prescaler value (1, 2, 4 ... 65536)
157  *
158  * p - prescaler coefficient, 2^p = npresc
159  *
160  * fclk - input clock frequency, in Hertz. In most function will be denoted as
161  * 'clkfreq'.
162  *END**************************************************************************/

164 /*FUNCTION**********************************************************************
165  *
166  * Function Name : lptmr_cfg2p
167  * Description : Transform prescaler settings (bypass on/off, prescaler value)
168  * to prescaler coefficient value (2's power), p.
169  * Return: the value of p.
170  *END**************************************************************************/
171 static inline uint8_t lptmr_cfg2p(
172  const lptmr_prescaler_t prescval,
173  const bool bypass
174  )
175 {
176  uint8_t p = 0u;
177 
178  if (!bypass)
179  {
180  p = (uint8_t)(((uint8_t)prescval) + 1u);
181  }
182 
183  return p;
184 }

186 /*FUNCTION**********************************************************************
187  *
188  * Function Name : lptmr_us2nn
189  * Description : Transform microseconds to undivided (unprescaled) timer units,
190  * nn.
191  * Return: the value of nn.
192  *END**************************************************************************/
193 static inline uint64_t lptmr_us2nn(
194  const uint32_t clkfreq,
195  const uint32_t us
196  )
197 {
198  /* Approximate the timeout in undivided (unprescaled) timer ticks.
199  - us is the timeout in microseconds (1/10^6 seconds)
200  - clkfreq is the frequency in Hertz
201  Operation:
202  nn = (us/1000000) * clkfreq
203  In C:
204  For better precision, first to the multiplication (us * clkfreq)
205  To overcome the truncation of the div operator in C, add half of the
206  denominator before the division. Hence:
207  nn = (us * clkfreq + 500000) / 1000000
208  */
209  /* There is no risk of overflow since us is 32-bit wide and clkfreq can be
210  a theoretical maximum of ~100 MHz (platform maximum), which is over the
211  maximum input of the LPTMR anyway
212  */
213  uint64_t nn = (uint64_t)( (uint64_t)us * (uint64_t)clkfreq );
214  nn = (nn + 500000u) / 1000000u;
215  return nn;
216 }

218 /*FUNCTION**********************************************************************
219  *
220  * Function Name : lptmr_compute_nticks
221  * Description : Compute total number of divided (prescaled) timer ticks,
222  * nticks.
223  * Return: the value of nticks.
224  *END**************************************************************************/
225 static inline uint64_t lptmr_compute_nticks(
226  uint64_t nn,
227  uint8_t p
228  )
229 {
230  uint64_t npresc = (uint64_t) 1u << p;
231  DEV_ASSERT(npresc != 0u);
232 
233  /* integer division */
234  uint64_t nticks = ((nn + (npresc >> 1u)) / npresc);
235 
236  return nticks;
237 }

239 /*FUNCTION**********************************************************************
240  *
241  * Function Name : nticks2compare_ticks
242  * Description : Transform the value of divided (prescaled) timer ticks, nticks
243  * to a 16-bit value to be written to the hardware register. Cap or underflow
244  * cause an error.
245  * Return: the success state.
246  * - true: no underflow or overflow detected
247  * - false: value written was capped, underflow or overflow detected
248  *
249  *END**************************************************************************/
250 static inline bool nticks2compare_ticks(
251  uint64_t nticks,
252  uint16_t* ticks
253  )
254 {
255  bool success = true;
256 
257  /* if nticks fits, write the value to ticks */
258  if (nticks <= LPTMR_MAX_CMR_NTICKS)
259  {
260  if (nticks == 0u)
261  {
262  /* timeout period (us) too low for prescaler settings */
263  *ticks = 0u;
264  success = false;
265  }
266  else{
267  /* According to RM, the LPTMR compare events take place when "the CNR equals the value of the CMR and increments".
268  * The additional increment is compensated here by decrementing the calculated compare value with 1, before being written to CMR. */
269  *ticks = (uint16_t)(nticks - 1u);
270  }
271  }
272  else {
273  /* timeout period (us) too high for prescaler settings */
274  *ticks = LPTMR_CMR_COMPARE_MASK;
275  success = false;
276  }
277 
278  return success;
279 }

281 /*FUNCTION**********************************************************************
282  *
283  * Function Name : lptmr_GetClkFreq
284  * Description : Get the clock frequency for the selected clock source. If the
285  * selected clock source is not enabled, a frequency of 0 is returned.
286  * Return values:
287  * - the clock frequency or 0 if the clock is invalid.
288  *
289  *END**************************************************************************/
290 static uint32_t lptmr_GetClkFreq(const lptmr_clocksource_t clkSrc,
291  const uint32_t instance)
292 {
293  /* LPTMR PCC clock source names, for getting the input clock frequency */
294  static const clock_names_t lptmrPccClockName[LPTMR_INSTANCE_COUNT] = {LPTMR0_CLK};
295  clock_names_t inputClockName = SIRC_CLK;
296  uint32_t clkFreq;
297  status_t clkStatus;
298 
299  /* Get input clock name */
300  switch(clkSrc)
301  {
302  case LPTMR_CLOCKSOURCE_SIRCDIV2:
303  inputClockName = SIRC_CLK;
304  break;
305  case LPTMR_CLOCKSOURCE_1KHZ_LPO:
306  inputClockName = SIM_LPO_1K_CLK;
307  break;
308  case LPTMR_CLOCKSOURCE_RTC:
309  inputClockName = SIM_RTCCLK_CLK;
310  break;
311  case LPTMR_CLOCKSOURCE_PCC:
312  inputClockName = lptmrPccClockName[instance];
313  break;
314  default:
315  /* Invalid clock source */
316  DEV_ASSERT(false);
317  break;
318  }
319 
320  /* Get input clock frequency */
321  if (inputClockName == SIRC_CLK)
322  {
323  clkStatus = CLOCK_SYS_GetFreq(SIRCDIV2_CLK, &clkFreq);
324  DEV_ASSERT(clkStatus == STATUS_SUCCESS);
325  (void) clkStatus;
326  DEV_ASSERT(clkFreq != 0u); /* If the GetFreq functions fails, clkfreq will be 0 */
327  }
328  else
329  {
330  clkStatus = CLOCK_SYS_GetFreq(inputClockName, &clkFreq);
331  DEV_ASSERT(clkStatus == STATUS_SUCCESS);
332  (void) clkStatus;
333  DEV_ASSERT(clkFreq != 0u); /* If the GetFreq functions fails, clkfreq will be 0 */
334  }
335 
336  return clkFreq;
337 }

339 /*FUNCTION**********************************************************************
340  *
341  * Function Name : lptmr_Ticks2Us
342  * Description : Transform timer ticks to microseconds using the given
343  * prescaler settings. Clock frequency must be valid (different from 0).
344  * Possible return values:
345  * - true: conversion success
346  * - false: conversion failed, result did not fit in 32-bit.
347  *
348  *END**************************************************************************/
349 static bool lptmr_Ticks2Us(
350  const uint32_t clkfreq,
351  const lptmr_prescaler_t pval,
352  const bool bypass,
353  const uint16_t ticks,
354  uint32_t* const us
355  )
356 {
357  bool success = true;
358  uint8_t p = lptmr_cfg2p(pval, bypass);
359  uint64_t nn = ( (uint64_t)ticks + 1u ) << p;
360  uint64_t us_real = (nn * 1000000u) / (clkfreq);
361  uint32_t us_local;
362 
363  if ( us_real <= (0xFFFFFFFFu) )
364  {
365  us_local = (uint32_t)us_real;
366  }
367  else
368  {
369  us_local = 0xFFFFFFFFu;
370  success = false;
371  }
372 
373  *us = us_local;
374  return success;
375 }

377 /*FUNCTION**********************************************************************
378  *
379  * Function Name : lptmr_Us2Ticks
380  * Description : Transform microseconds to timer ticks using the given
381  * prescaler settings. Input clock frequency, clkfreq, must be greater than 0.
382  * Possible return values:
383  * - true: conversion completed successfully
384  * - false: conversion failed, value did not fit in 16-bit.
385  *
386  *END**************************************************************************/
387 static bool lptmr_Us2Ticks(
388  const uint32_t clkfreq,
389  const lptmr_prescaler_t prescval,
390  const bool bypass,
391  const uint32_t us,
392  uint16_t* const ticks
393  )
394 {
395  bool success = true;
396  /* Transform prescaler configuration to prescaler coefficient p */
397  uint8_t p = lptmr_cfg2p(prescval, bypass);
398  /* Compute nn, the number of ticks necessary for the period in microseconds
399  without any prescaler */
400  uint64_t nn = lptmr_us2nn(clkfreq, us);
401  /* Compute nticks, total number of ticks with prescaler */
402  uint64_t nticks = lptmr_compute_nticks(nn, p);
403  /* Transform nticks to value to be written to register */
404  success = nticks2compare_ticks(nticks, ticks);
405  return success;
406 }

408 /*FUNCTION**********************************************************************
409  *
410  * Function Name : lptmr_ChooseClkConfig
411  * Description : Choose clocking configuration (prescaler value, timer ticks)
412  * for the desired timeout period, given in microseconds. Input clock frequency,
413  * clkfreq, must be greater than 0.
414  * Possible return values:
415  * - true: configuration found
416  * - false: configuration mismatch, desired timeout period is too small or too
417  * big for the clock settings.
418  *
419  *END**************************************************************************/
420 static bool lptmr_ChooseClkConfig(
421  const uint32_t clkfreq,
422  const uint32_t us,
423  lptmr_prescaler_t* const prescval,
424  bool* const bypass,
425  uint16_t* const ticks
426  )
427 {
428  uint8_t p;
429  uint64_t nticks = 0ULL;
430  bool success;
431 
432  uint64_t nn = lptmr_us2nn(clkfreq, us);
433 
434  /* Find the lowest prescaler value that allows the compare value in 16-bits */
435  for (p = 0u; p <= LPTMR_MAX_PRESCALER; p++)
436  {
437  nticks = lptmr_compute_nticks(nn, p);
438 
439  if (nticks <= LPTMR_MAX_CMR_NTICKS)
440  {
441  /* Search finished, value will fit in the 16-bit register */
442  break;
443  }
444  }
445 
446  success = nticks2compare_ticks(nticks, ticks);
447 
448  /* Convert p to prescaler configuration */
449  if (p == 0u)
450  {
451  /* Prescaler value of 1 */
452  *bypass = true;
453  *prescval = LPTMR_PRESCALE_2;
454  }
455  else{
456  *bypass = false;
457  p--; /* Decrement to match lptmr_prescaler_t. */
458  *prescval = (lptmr_prescaler_t) p;
459  }
460 
461  return success;
462 }

499 /*FUNCTION**********************************************************************
500  *
501  * Function Name : LPTMR_DRV_Init
502  * Description : Initialize a LPTMR instance based on the input configuration
503  * structure.
504  *
505  * When (counterUnits == LPTMR_COUNTER_UNITS_MICROSECONDS) the function will
506  * automatically configure the timer for the input compareValue in microseconds.
507  * The input parameters for 'prescaler' and 'bypassPrescaler' will be ignored
508  * - their values will be adapted by the function, to best fit the input compareValue
509  * (in microseconds) for the operating clock frequency.
510  *
511  * LPTMR_COUNTER_UNITS_MICROSECONDS may only be used for LPTMR_WORKMODE_TIMER mode.
512  * Otherwise the function shall not convert 'compareValue' in ticks
513  * and this is likely to cause erroneous behavior.
514  *
515  * When (counterUnits == LPTMR_COUNTER_UNITS_TICKS) the function will use the
516  * 'prescaler' and 'bypassPrescaler' provided in the input configuration structure.
517  *
518  * When (counterUnits == LPTMR_COUNTER_UNITS_TICKS), 'compareValue' must be lower
519  * than 0xFFFFu. Only the least significant 16bits of 'compareValue' will be used.
520  * When (counterUnits == LPTMR_COUNTER_UNITS_MICROSECONDS), 'compareValue'
521  * may take any 32bits unsigned value.
522  *
523  * Implements : LPTMR_DRV_Init_Activity
524  *END**************************************************************************/
525 void LPTMR_DRV_Init(const uint32_t instance,
526  const lptmr_config_t * const config,
527  const bool startCounter)
528 {
529  DEV_ASSERT(instance < LPTMR_INSTANCE_COUNT);
530  DEV_ASSERT(config != NULL);
531 
532  LPTMR_Type* const base = g_lptmrBase[instance];
533 
534  LPTMR_DRV_SetConfig(instance, config);
535 
536  /* Start the counter if requested */
537  if (startCounter)
538  {
539  LPTMR_Enable(base);
540  }
541 }

556  * Otherwise the function shall not convert 'compareValue' in ticks
557  * and this is likely to cause erroneous behavior.
558  *
559  * When (counterUnits == LPTMR_COUNTER_UNITS_TICKS) the function will use the
560  * 'prescaler' and 'bypassPrescaler' provided in the input configuration structure.
561  *
562  * When (counterUnits == LPTMR_COUNTER_UNITS_TICKS), 'compareValue' must be lower
563  * than 0xFFFFu. Only the least significant 16bits of 'compareValue' will be used.
564  * When (counterUnits == LPTMR_COUNTER_UNITS_MICROSECONDS), 'compareValue'
565  * may take any 32bits unsigned value.
566  *
567  * Implements : LPTMR_DRV_SetConfig_Activity
568  *END**************************************************************************/
569 void LPTMR_DRV_SetConfig(const uint32_t instance,
570  const lptmr_config_t * const config)
571 {
572  DEV_ASSERT(instance < LPTMR_INSTANCE_COUNT);
573  DEV_ASSERT(config != NULL);
574 
575  LPTMR_Type* const base = g_lptmrBase[instance];
576  uint32_t configCmpValue = config->compareValue;
577  lptmr_workmode_t configWorkMode = config->workMode;
578  uint16_t cmpValueTicks = 0U;
579  lptmr_prescaler_t prescVal = config->prescaler;
580  bool prescBypass = config->bypassPrescaler;
581  lptmr_counter_units_t configCounterUnits = config->counterUnits;
582 
583  if(configWorkMode == LPTMR_WORKMODE_TIMER)
584  {
585  /* A valid clock must be selected when used in Timer Mode. */
586  uint32_t clkFreq;
587  clkFreq = lptmr_GetClkFreq(config->clockSelect, instance);
588  DEV_ASSERT(clkFreq != 0U); /* Clock frequency equal to '0', signals invalid value. */
589 
590  if(configCounterUnits == LPTMR_COUNTER_UNITS_MICROSECONDS)
591  {
592  bool chooseClkConfigStatus;
593 
594  /* When workmode is set to Timer Mode and compare value is provided in microseconds,
595  * then the input parameters for prescale value and prescaleBypass are ignored.
596  * The prescaleValue, prescaleBypass and cmpValue in ticks, are calculated to best fit
597  * the input configCmpValue (in us) for the current operating clk frequency. */
598  chooseClkConfigStatus = lptmr_ChooseClkConfig(clkFreq, configCmpValue, &prescVal, &prescBypass, &cmpValueTicks);
599  DEV_ASSERT(chooseClkConfigStatus == true);
600  (void) chooseClkConfigStatus;
601  }
602  else
603  {
604  DEV_ASSERT(configCounterUnits == LPTMR_COUNTER_UNITS_TICKS);
605  DEV_ASSERT(configCmpValue <= LPTMR_CMR_COMPARE_MASK); /* Compare Value in Tick Units must fit in CMR. */
606 
607  cmpValueTicks = (uint16_t)(configCmpValue & LPTMR_CMR_COMPARE_MASK);
608  }
609  }
610  else
611  {
612  /* If configWorkMode is not LPTMR_WORKMODE_TIMER, then it must be LPTMR_WORKMODE_PULSECOUNTER. */
613  DEV_ASSERT(configWorkMode == LPTMR_WORKMODE_PULSECOUNTER);
614 
615  /* Only LPTMR_COUNTER_UNITS_TICKS can be used when LPTMR is configured as Pulse Counter. */
616  DEV_ASSERT(config->counterUnits == LPTMR_COUNTER_UNITS_TICKS);
617  /* A valid clock must be selected when glitch filter is enabled (prescaler not bypassed). */
618  DEV_ASSERT((lptmr_GetClkFreq(config->clockSelect, instance) != 0u) || prescBypass);
619  /* Glitch filter does not support LPTMR_PRESCALE_2. */
620  DEV_ASSERT(prescBypass || (prescVal != LPTMR_PRESCALE_2));
621 
622  DEV_ASSERT(configCmpValue <= LPTMR_CMR_COMPARE_MASK); /* Compare Value in Tick Units must fit in CMR. */
623 
624  cmpValueTicks = (uint16_t)(configCmpValue & LPTMR_CMR_COMPARE_MASK);
625  }
626 
627  /* Initialize and write configuration parameters. */
628  LPTMR_Init(base);
629 
630  LPTMR_SetDmaRequest (base, config->dmaRequest);
631  LPTMR_SetInterrupt (base, config->interruptEnable);
632  LPTMR_SetFreeRunning (base, config->freeRun);
633  LPTMR_SetWorkMode (base, configWorkMode);
634  LPTMR_SetPrescaler (base, prescVal);
635  LPTMR_SetBypass (base, prescBypass);
636  LPTMR_SetClockSelect (base, config->clockSelect);
637  LPTMR_SetCompareValue (base, cmpValueTicks);
638  LPTMR_SetPinSelect (base, config->pinSelect);
639  LPTMR_SetPinPolarity (base, config->pinPolarity);
640 }

697  * - STATUS_ERROR: cannot reconfigure compare value (TCF not set)

702 status_t LPTMR_DRV_SetCompareValueByCount(const uint32_t instance,

704 {

708  status_t statusCode = STATUS_SUCCESS;

710  bool timerEnabled = LPTMR_GetEnable(base);
711  bool compareFlag = LPTMR_GetCompareFlag(base);
712 
713  uint16_t counterVal;
714 
715  /* Check if a valid clock is selected for the timer/glitch filter */

766  * Function Name : LPTMR_DRV_SetCompareValueUs
767  * Description : Set the compare value for Timer Mode in microseconds,
768  * for a LPTMR instance.
769  * Can be used only in Timer Mode.
770  * Possible return values:
771  * - STATUS_SUCCESS: completed successfully
772  * - STATUS_ERROR: cannot reconfigure compare value
773  * - STATUS_TIMEOUT: compare value greater then current counter value
774  *
775  * Implements : LPTMR_DRV_SetCompareValueByUs_Activity
776  *END**************************************************************************/
777 status_t LPTMR_DRV_SetCompareValueByUs(const uint32_t instance,
778  const uint32_t compareValueUs)
779 {
780  DEV_ASSERT(instance < LPTMR_INSTANCE_COUNT);
781 
782  status_t returnCode = STATUS_SUCCESS;
783  LPTMR_Type* const base = g_lptmrBase[instance];
784  bool timerEnabled, compareFlag;
785 
786  lptmr_clocksource_t clkSrc;
787  uint32_t clkFreq;
788  uint16_t cmpValTicks, currentCounterVal;
789  lptmr_prescaler_t prescVal;
790  bool prescBypass, conversionStatus;
791 
792  /* This function can only be used if LPTMR is configured in Timer Mode. */
793  DEV_ASSERT(LPTMR_GetWorkMode(base) == LPTMR_WORKMODE_TIMER);
794 
795  timerEnabled = LPTMR_GetEnable(base);
796  compareFlag = LPTMR_GetCompareFlag(base);
797  /* The compare value can only be written if counter is disabled or the compare flag is set. */
798  if (timerEnabled && !compareFlag)
799  {
800  returnCode = STATUS_ERROR;
801  }
802  else
803  {
804  clkSrc = LPTMR_GetClockSelect(base);
805  clkFreq = lptmr_GetClkFreq(clkSrc, instance);
806  DEV_ASSERT(clkFreq != 0U); /* Check the calculated clock frequency: '0' - invalid*/
807 
808  /* Get prescaler value and prescaler bypass state.*/
809  prescVal = LPTMR_GetPrescaler(base);
810  prescBypass = LPTMR_GetBypass(base);
811  /* Convert new compare value from microseconds to ticks. */
812  conversionStatus = lptmr_Us2Ticks(clkFreq, prescVal, prescBypass, compareValueUs, &cmpValTicks);
813  DEV_ASSERT(conversionStatus == true); /* Check the conversion status: compareValueUs doesn't fit for current prescaller. */
814  (void) conversionStatus;
815 
816  /* Write value and check if written successfully */
817  LPTMR_SetCompareValue(base, cmpValTicks);
818  currentCounterVal = LPTMR_GetCounterValue(base);
819 
820  if (currentCounterVal >= cmpValTicks)
821  {
822  returnCode = STATUS_TIMEOUT;
823  }
824  }
825 
826  return returnCode;
827 }
