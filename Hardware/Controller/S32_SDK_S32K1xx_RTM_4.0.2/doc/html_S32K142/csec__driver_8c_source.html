1093  *
1094  * Implements : CSEC_DRV_MPCompress_Activity
1095  * END**************************************************************************/
1096 status_t CSEC_DRV_MPCompress(const uint8_t * msg,
1097  uint16_t msgLen,
1098  uint8_t * mpCompress,
1099  uint32_t timeout)
1100 {
1101  DEV_ASSERT(msg != NULL);
1102  DEV_ASSERT(mpCompress != NULL);
1103  DEV_ASSERT(g_csecStatePtr != NULL);
1104 
1105  uint32_t startTime = 0;
1106  uint32_t crtTime = 0;
1107  status_t stat = STATUS_SUCCESS;
1108  uint16_t numPagesLeft = msgLen;
1109 
1110  if (g_csecStatePtr->cmdInProgress)
1111  {
1112  return STATUS_BUSY;
1113  }
1114 
1115  /* Initialize the internal state of the driver */
1116  CSEC_DRV_InitState(CSEC_SECRET_KEY, CSEC_CMD_MP_COMPRESS, msg, mpCompress, (uint32_t)msgLen << CSEC_BYTES_TO_FROM_PAGES_SHIFT);
1117 
1118  startTime = OSIF_GetMilliseconds();
1119 
1120  /* Loop and launch commands until the end of the message */
1121  while (numPagesLeft > 0U)
1122  {
1123  uint8_t numPages = (uint8_t)((numPagesLeft > CSEC_DATA_PAGES_AVAILABLE) ?
1124  CSEC_DATA_PAGES_AVAILABLE : numPagesLeft);
1125  uint8_t numBytes = (uint8_t)(numPages << CSEC_BYTES_TO_FROM_PAGES_SHIFT);
1126 
1127  /* Write the message */
1128  CSEC_WriteCommandBytes(FEATURE_CSEC_PAGE_1_OFFSET, &msg[g_csecStatePtr->index], numBytes);
1129  /* Write the size of the message */

1139  {
1140  CSEC_DRV_CancelCommand();
1141 
1142  stat = STATUS_TIMEOUT;
1143  }
1144 
1145  if (stat == STATUS_SUCCESS)
1146  {
1147  /* Read the status of the execution */
1148  stat = CSEC_ReadErrorBits();
1149  }
1150

1153  break;
1154  }
1155

1162  }

1168  }
1169
