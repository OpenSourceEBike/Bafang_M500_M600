96 /*FUNCTION**********************************************************************
97  *
98  * Function Name : INT_SYS_InstallHandler
99  * Description : Install an interrupt handler routine for a given IRQ number
100  * This function will let application register/replace the interrupt
101  * handler for specified IRQ number. IRQ number is different than Vector
102  * number. IRQ 0 will start from Vector 16 address. Refer to reference
103  * manual for details. Also refer to startup_<CPU>.s file for each chip
104  * family to find out the default interrupt handler for each device. This
105  * function will convert the IRQ number to vector number by adding 16 to
106  * it.
107  *
108  * Note : This method is applicable only if interrupt vector is
109  * copied in RAM, __flash_vector_table__ symbol is used to
110  * control this from linker options.
111  * Implements INT_SYS_InstallHandler_Activity
112  *
113  *END**************************************************************************/
114 void INT_SYS_InstallHandler(IRQn_Type irqNumber,
115  const isr_t newHandler,
116  isr_t* const oldHandler)
117 {
118 
119 #if (defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT))
120 
121  /* Check IRQ number - dev_irqNumber is used to avoid compiler warning */
122  int32_t dev_irqNumber = (int32_t)irqNumber;
123  DEV_ASSERT(FEATURE_INTERRUPT_IRQ_MIN <= irqNumber);
124  DEV_ASSERT(dev_irqNumber <= (int32_t)FEATURE_INTERRUPT_IRQ_MAX);
125 
126 #endif /*(defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT)) */
127 
128  const uint32_t * aVectorRam = (uint32_t *)__VECTOR_RAM;
129  const uint32_t * aVectorTable = (uint32_t *)__VECTOR_TABLE;
130 
131  /* Check whether there is vector table in RAM */
132 #if defined(__ARMCC_VERSION)
133  if ((aVectorRam != aVectorTable) || (__DATA_ROM > __RAM_START))
134 #elif (defined(__ICCARM__))
135  if ((aVectorRam != aVectorTable) || ((uint32_t)CODE_ROM_SECTION_START > (uint32_t)__RAM_START))
136 #else
137  const uint32_t * aDataRom = (uint32_t *)__DATA_ROM;
138  const uint32_t * aDataRam = (uint32_t *)__DATA_END;
139  if ((aVectorRam != aVectorTable) || (aDataRom == aDataRam))
140 #endif
141  {
142  uint32_t * pVectorRam = (uint32_t *)__VECTOR_RAM;
143 
144  /* Save the former handler pointer */
145  if (oldHandler != (isr_t *) 0)
146  {
147  *oldHandler = (isr_t)pVectorRam[((int32_t)irqNumber) + 16];
148  }
149 
150 #if FEATURE_MSCM_HAS_INTERRUPT_ROUTER
151 
152  #if (defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT))
153 
154  DEV_ASSERT((int32_t)irqNumber < (int32_t)MSCM_IRSPRC_COUNT);
155  /* Check routing is not read-only in case it needs to be written */
156  uint16_t cpu_enable = (uint16_t)(1UL << (MSCM->CPXNUM));
157  if ((MSCM->IRSPRC[irqNumber] & cpu_enable) == 0U)
158  {
159  DEV_ASSERT((MSCM->IRSPRC[irqNumber] & (uint16_t)(MSCM_IRSPRC_RO_MASK)) == (uint16_t)MSCM_IRSPRC_RO(0));
160  }
161 
162  #endif /*(defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT)) */
163 
164 #endif /* FEATURE_MSCM_HAS_INTERRUPT_ROUTER */
165 
166  /* Set handler into vector table */
167  pVectorRam[((int32_t)irqNumber) + 16] = (uint32_t)newHandler;
168  }
169  else
170  {
171  #if (defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT))
172  const uint32_t * pVectorRam = (uint32_t *)__VECTOR_RAM;
173  /* Check if the existing handler is the same as the one already present in the vector */
174  DEV_ASSERT(pVectorRam[((int32_t)irqNumber) + 16] == (uint32_t)newHandler);
175 
176  #endif /*(defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT)) */
177  }
178 }

182  * Function Name : INT_SYS_EnableIRQ
183  * Description : Enables an interrupt for a given IRQ number.
184  * It calls the system NVIC API to access the interrupt control
185  * register and MSCM (if available) API for interrupt routing.
186  * The input IRQ number does not include the core interrupt, only
187  * the peripheral interrupt, from 0 to a maximum supported IRQ.
188  * Implements INT_SYS_EnableIRQ_Activity
189  *END**************************************************************************/
190 void INT_SYS_EnableIRQ(IRQn_Type irqNumber)
191 {
192 #if (defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT))
193  /* Check IRQ number - dev_irqNumber is used to avoid compiler warning */
194  int32_t dev_irqNumber = (int32_t)irqNumber;
195  DEV_ASSERT(0 <= (int32_t)irqNumber);
196  DEV_ASSERT(dev_irqNumber <= (int32_t)FEATURE_INTERRUPT_IRQ_MAX);
197 #endif /*(defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT)) */
198  /* Enable interrupt */
199  S32_NVIC->ISER[(uint32_t)(irqNumber) >> 5U] = (uint32_t)(1UL << ((uint32_t)(irqNumber) & (uint32_t)0x1FU));
200 
201 #if FEATURE_MSCM_HAS_INTERRUPT_ROUTER
202 
203  /* Enable routing to current CPU */
204  uint16_t cpu_enable = (uint16_t)(1UL << (MSCM->CPXNUM));
205  MSCM->IRSPRC[irqNumber] |= cpu_enable;
206 
207 #endif /* FEATURE_MSCM_HAS_INTERRUPT_ROUTER */
208 }

280 /*FUNCTION**********************************************************************
281  *
282  * Function Name : INT_SYS_SetPriority
283  * Description : Set the priority of an interrupt
284  * This function will set the priority of an interrupt.
285  * Note: The priority cannot be set for every core interrupt.
286  * Implements INT_SYS_SetPriority_Activity
287  *
288  *END**************************************************************************/
289 void INT_SYS_SetPriority(IRQn_Type irqNumber, uint8_t priority)
290 {
291 
292 #if (defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT))
293 
294  /* Check IRQ number and priority - dev_irqNumber is used to avoid compiler warning */
295  int32_t dev_irqNumber = (int32_t)irqNumber;
296  DEV_ASSERT(FEATURE_INTERRUPT_IRQ_MIN <= irqNumber);
297  DEV_ASSERT(dev_irqNumber <= (int32_t)FEATURE_INTERRUPT_IRQ_MAX);
298  DEV_ASSERT(priority < (uint8_t)(1U << FEATURE_NVIC_PRIO_BITS));
299 
300 #endif /*(defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT)) */
301 
302  uint8_t shift = (uint8_t) (8U - FEATURE_NVIC_PRIO_BITS);
303 
304  if ((int32_t)irqNumber < 0)
305  {
306  uint32_t intVectorId = ((uint32_t)(irqNumber) & 0xFU);
307  uint32_t regId = intVectorId / 4U;
308 
309  /* Compute pointer to SHPR register - avoid MISRA violation. */
310  #if defined (S32K11x_SERIES)
311  DEV_ASSERT((int32_t)regId != 1);
312 
313  volatile uint32_t * shpr_reg_ptr = ((regId == 2U) ? (volatile uint32_t *)&S32_SCB->SHPR2 : (volatile uint32_t *)&S32_SCB->SHPR3);
314  uint8_t priByteShift = (uint8_t)(((uint8_t)(intVectorId) & 0x3U) << 3U);
315 
316  /* Clear the old value from the register */
317  *shpr_reg_ptr &= ~(0xFFUL << priByteShift);
318 
319  /* Set Priority for Cortex-M0P System Interrupts */
320  *shpr_reg_ptr |= ((uint32_t)(((((uint32_t)priority) << shift)) & 0xFFUL)) << priByteShift;
321  #else
322  volatile uint8_t * shpr_reg_ptr = ((regId == 1U) ? (volatile uint8_t *)&S32_SCB->SHPR1 : ((regId == 2U) ? (volatile uint8_t *)&S32_SCB->SHPR2 : (volatile uint8_t *)&S32_SCB->SHPR3));
323 
324  /* Set Priority for Cortex-M4 System Interrupts */
325  shpr_reg_ptr[intVectorId % 4U] = (uint8_t)(((((uint32_t)priority) << shift)) & 0xffUL);
326  #endif /* defined (S32K11x_SERIES) */
327  }
328  else
329  {
330  /* Set Priority for device specific Interrupts */
331  #if defined (S32K11x_SERIES)
332  uint32_t iprVectorId = (uint32_t)(irqNumber) >> 2U;
333  uint8_t priByteShift = (uint8_t)((((uint8_t)irqNumber) & 0x3U) << 3U);
334 
335  /* Clear the old value from the register */
336  S32_NVIC->IPR[iprVectorId] &= ~(0xFFUL << priByteShift);
337 
338  S32_NVIC->IPR[iprVectorId] |= ((uint32_t)(((((uint32_t)priority) << shift)) & 0xFFUL)) << priByteShift;
339  #else
340  S32_NVIC->IP[(uint32_t)(irqNumber)] = (uint8_t)(((((uint32_t)priority) << shift)) & 0xFFUL);
341  #endif /* defined (S32K11x_SERIES) */
342  }
343 }

366  uint8_t priority = 0U;
367  uint8_t shift = (uint8_t) (8U - FEATURE_NVIC_PRIO_BITS);
368 
369  if ((int32_t)irqNumber < 0)
370  {
371  uint32_t intVectorId = ((uint32_t)(irqNumber) & 0xFU);
372  uint32_t regId = intVectorId / 4U;
373 
374  /* Compute pointer to SHPR register - avoid MISRA violation. */
375  #if defined (S32K11x_SERIES)
376  DEV_ASSERT((int32_t)regId != 1);
377

379  uint8_t priByteShift = (uint8_t)(((uint8_t)(intVectorId) & 0x3U) << 3U);
380 
381  priority = ((uint8_t)(*shpr_reg_ptr >> priByteShift)) >> shift;
382  #else
383  volatile const uint8_t * shpr_reg_ptr = ((regId == 1U) ? (volatile uint8_t *)&S32_SCB->SHPR1 : ((regId == 2U) ? (volatile uint8_t *)&S32_SCB->SHPR2 : (volatile uint8_t *)&S32_SCB->SHPR3));
384 
385  /* Get Priority from Cortex-M System Interrupts */
386  priority = (uint8_t)(shpr_reg_ptr[intVectorId % 4U] >> (shift));
387  #endif /* defined (S32K11x_SERIES) */
388  }
389  else
390  {
391  /* Get Priority for device specific Interrupts */
392  #if defined (S32K11x_SERIES)
393  uint32_t iprVectorId = (uint32_t)(irqNumber) >> 2U;
394  uint8_t priByteShift = (uint8_t)((((uint8_t)(irqNumber)) & 0x3U) << 3U);
395  priority = ((uint8_t)(S32_NVIC->IPR[iprVectorId] >> priByteShift)) >> shift;
396  #else
397  priority = (uint8_t)(S32_NVIC->IP[(uint32_t)(irqNumber)] >> shift);
398  #endif /* defined (S32K11x_SERIES) */
399  }
400 
401  return priority;
402 }
