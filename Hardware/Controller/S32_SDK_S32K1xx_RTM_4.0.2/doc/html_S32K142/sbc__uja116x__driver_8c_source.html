181  * regulator.supplyEvnt.v2oe = SBC_UJA_SUPPLY_EVNT_V2OE_DIS;

185  * Implements : SBC_SetVreg_Activity
186  *END**************************************************************************/

188 {
189  status_t status = STATUS_SUCCESS;
190  status_t spiStat = STATUS_SUCCESS;
191 
192  DEV_ASSERT(regulatorCtr != NULL);
193 
194  const sbc_regulator_t *regulator = &regulatorCtr->regulator;
195  uint8_t regData = (uint8_t)((uint8_t)regulator->pdc | (uint8_t)regulator->v2c
196  | (uint8_t)regulator->v1rtc);
197 
198  /* Send configuration to Regulator control register. */
199  spiStat = SBC_DataTransfer(SBC_UJA_REGULATOR, &regData, NULL);

201

205  /* Send configuration to Supply Event register. */
206  regData = (uint8_t)((uint8_t)sEvnt->v2oe | (uint8_t)sEvnt->v2ue
207  | (uint8_t)sEvnt->v1ue);
208  spiStat = SBC_DataTransfer(SBC_UJA_SUPPLY_EVNT, &regData, NULL);
209  }
210 
211  if(spiStat != STATUS_SUCCESS)

213  status = SBC_COMM_ERROR;

270 /*FUNCTION**********************************************************************
271  *
272  * Function Name : SBC_SetWatchdog
273  * Description : This function writes to Watchdog control register (0x00).
274  * This function can select Watchdog mode control, and nominal watchdog period.
275  * The UJA116xA contains a watchdog that supports three operating modes: Window,
276  * Timeout and Autonomous. In Window mode (available only in SBC Normal mode), a
277  * watchdog trigger event within a defined watchdog window triggers and resets
278  * the watchdog timer. In Timeout mode, the watchdog runs continuously and can
279  * be triggered and reset at any time within the watchdog period by a watchdog
280  * trigger. Watchdog time-out mode can also be used for cyclic wake-up
281  * of the microcontroller. In Autonomous mode, the watchdog can be off or
282  * autonomously in Timeout mode, depending on the selected SBC mode.
283  * Note SBC mode will temporary set to StandBy while WatchDog configuration
284  * is changed.
285  * The following is an example of how to set up Watchdog register by
286  * passing wtdog parameter to SBC_SetWatchdog function:
287  * wtdog.modeControl = SBC_UJA_WTDOG_CTR_WMC_AUTO;
288  * wtdog.nominalPeriod = SBC_UJA_WTDOG_CTR_NWP_4096;
289  * status_t status = SBC_SetWatchdog(&wtdog);
290  *
291  * Implements : SBC_SetWatchdog_Activity
292  *END**************************************************************************/
293 status_t SBC_SetWatchdog(const sbc_wtdog_ctr_t* const wtdog)
294 {
295  status_t status = STATUS_SUCCESS;
296  status_t spiStat = STATUS_SUCCESS;
297 
298  DEV_ASSERT(wtdog != NULL);
299 
300  bool newWdMode = false;
301  /* Variable for storing current Mode settings. */
302  sbc_mode_mc_t currentMode = SBC_UJA_MODE_MC_SLEEP;
303 
304  /* Test if new Watchdog value is different. */
305  if((g_drvConfig.watchdogCtr.modeControl != wtdog->modeControl) ||
306  (g_drvConfig.watchdogCtr.nominalPeriod != wtdog->nominalPeriod)
307  || g_drvConfig.isInit)
308  {
309  newWdMode = true;
310  /* Read current mode. */
311  status = SBC_GetMode(&currentMode);
312  }
313 
314  /* Set device to stand by mode. */
315  if(newWdMode && (status == STATUS_SUCCESS) &&
316  (currentMode != SBC_UJA_MODE_MC_STANDBY))
317  {
318  status = SBC_SetMode(SBC_UJA_MODE_MC_STANDBY);
319  }
320 
321  if(status == STATUS_SUCCESS)
322  {
323  uint8_t regData = (uint8_t)((uint8_t)wtdog->modeControl
324  | (uint8_t)wtdog->nominalPeriod);
325 
326  /* Send configuration to Watchdog control register. */
327  spiStat = SBC_DataTransfer(SBC_UJA_WTDOG_CTR, &regData, NULL);
328  }
329 
330  if(newWdMode && (spiStat == STATUS_SUCCESS) &&
331  (status == STATUS_SUCCESS) &&
332  (currentMode != SBC_UJA_MODE_MC_STANDBY))
333  {
334  /* Return to previous mode. */
335  status = SBC_SetMode(currentMode);
336 
337  g_drvConfig.watchdogCtr.modeControl = wtdog->modeControl;
338  g_drvConfig.watchdogCtr.nominalPeriod = wtdog->nominalPeriod;
339  }
340 
341  if(spiStat != STATUS_SUCCESS)
342  {
343  status = SBC_COMM_ERROR;
344  }
345 
346  return status;
347 }

417 /*FUNCTION**********************************************************************
418  *
419  * Function Name : SBC_SetMode
420  * Description : This function write to Mode control register (0x01).
421  * Normal mode is the active operating mode. In this mode, all the hardware on
422  * the device is available and can be activated. Voltage regulator
423  * V1 is enabled to supply the microcontroller.
424  * Standby mode is the first-level power-saving mode of the UJA116xA, offering
425  * reduced current consumption. The transceiver is unable to transmit or
426  * receive data in Standby mode. The SPI remains enabled and V1 is still active;
427  * the watchdog is active (in Timeout mode) if enabled. The behavior
428  * of V2/VEXT is determined by the SPI setting.
429  * Sleep mode is the second-level power-saving mode of the UJA116xA.
430  * The difference between Sleep and Standby modes is that V1 is off in
431  * Sleep mode and temperature protection is inactive.
432  * Note event status are cleared before device move to sleep mode. At least
433  * one wake up event must be enabled before moving to sleep mode otherwise SBC
434  * will be reseted.
435  * The following is an example of how to set up Mode register by
436  * passing mode parameter to SBC_SetMode function:
437  * sbc_mode_mc_t mode;
438  * mode = SBC_UJA_MODE_MC_STANDBY;
439  * status_t status = SBC_SetMode(&mode);
440  *
441  * Implements : SBC_SetMode_Activity
442  *END**************************************************************************/
443 status_t SBC_SetMode(const sbc_mode_mc_t mode)
444 {
445  status_t status = STATUS_SUCCESS;
446  status_t spiStat = STATUS_SUCCESS;
447 
448  uint8_t regData = (uint8_t)mode;
449 
450  if((mode == SBC_UJA_MODE_MC_SLEEP) || (mode == SBC_UJA_MODE_MC_STANDBY)
451  || (mode == SBC_UJA_MODE_MC_NORMAL))
452  {
453  if((mode == SBC_UJA_MODE_MC_SLEEP) || (mode == SBC_UJA_MODE_MC_STANDBY))
454  {
455  /* Clear all events and move to sleep mode. */
456  sbc_evn_capt_t events;
457  status = SBC_GetEventsStatus(&events);
458 
459  if(status == STATUS_SUCCESS)
460  {
461  status = SBC_CleanEvents(&events);
462  }
463  }
464  }
465  else
466  {
467  status = SBC_CMD_ERROR;
468  }
469 
470  if(status == STATUS_SUCCESS)
471  {
472  /* Write mode. */
473  spiStat = SBC_DataTransfer(SBC_UJA_MODE, &regData, NULL);
474  }
475 
476  if(spiStat != STATUS_SUCCESS)
477  {
478  status = SBC_COMM_ERROR;
479  }
480 
481  return status;
482 }

531  *
532  * Function Name : SBC_SetFailSafe
533  * Description : This function writes to Fail-safe control register (0x02).
534  * The dedicated LIMP pin can be used to enable so called limp home hardware
535  * in the event of a serious ECU failure. Detectable failure conditions include
536  * SBC overtemperature events, loss of watchdog service, short-circuits on pins
537  * RSTN or V1 and user-initiated or external reset events.
538  * The LIMP pin is a battery-robust, active-LOW, open-drain output.
539  * The LIMP pin can also be forced LOW by setting bit LHC in the Fail-safe
540  * control register.
541  * The limp-home function and the reset counter are disabled in
542  * Forced Normal mode. The LIMP pin is floating, RCC remains unchanged
543  * and bit LHC = 0.
544  * RCC -reset counter control.
545  * incremented every time the SBC enters Reset mode while FNMC = 0;
546  * RCC overflows from 11 to 00; default at power-on is 00. For ignore settings
547  * of rcc use NULL pointer or otherwise send pointer to variable.
548  * This function is not available on UJA1168 device variants.
549  * The following is an example of how to write to limp home control by
550  * passing lhc parameter to SBC_SetFailSafe function:
551  * sbc_fail_safe_lhc_t lhc;
552  * lhc = SBC_UJA_FAIL_SAFE_LHC_FLOAT;
553  * status_t status = SBC_SetFailSafe(&lhc, NULL);
554  *
555  * Implements : SBC_SetFailSafe_Activity
556  *END**************************************************************************/

558  const sbc_fail_safe_rcc_t* const rcc)
559 {

562
