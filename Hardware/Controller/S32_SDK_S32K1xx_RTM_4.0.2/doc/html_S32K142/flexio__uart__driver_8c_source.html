103 /*FUNCTION**********************************************************************
104  *
105  * Function Name : FLEXIO_UART_DRV_ComputeBaudRateDivider
106  * Description : Computes the baud rate divider for a target baud rate
107  *
108  *END**************************************************************************/
109 static void FLEXIO_UART_DRV_ComputeBaudRateDivider(uint32_t baudRate,
110  uint16_t *divider,
111  uint32_t inputClock)
112 {
113  int32_t tmpDiv;
114 
115  /* Compute divider: ((input_clock / baud_rate) / 2) - 1. Round to nearest integer */
116  tmpDiv = (((int32_t)inputClock + (int32_t)baudRate) / (2 * (int32_t)baudRate)) - 1;
117  /* Enforce upper/lower limits */
118  if (tmpDiv < DIVIDER_MIN_VALUE)
119  {
120  tmpDiv = DIVIDER_MIN_VALUE;
121  }
122  if (tmpDiv > DIVIDER_MAX_VALUE)
123  {
124  tmpDiv = DIVIDER_MAX_VALUE;
125  }
126 
127  *divider = (uint16_t)tmpDiv;
128 }

252 /*FUNCTION**********************************************************************
253  *
254  * Function Name : FLEXIO_UART_DRV_EndTransfer
255  * Description : End the current transfer
256  *
257  *END**************************************************************************/
258 static void FLEXIO_UART_DRV_EndTransfer(flexio_uart_state_t *state)
259 {
260  FLEXIO_Type *baseAddr;
261  uint8_t resourceIndex; /* Index of first used internal resource instance (shifter and timer) */
262 
263  baseAddr = g_flexioBase[state->flexioCommon.instance];
264  resourceIndex = state->flexioCommon.resourceIndex;
265 
266  /* Disable transfer engine */
267  switch (state->driverType)
268  {
269  case FLEXIO_DRIVER_TYPE_INTERRUPTS:
270  /* Disable interrupts for Rx / Tx shifter */
271  FLEXIO_SetShifterInterrupt(baseAddr, (uint8_t)(1U << TX_SHIFTER(resourceIndex)), false);
272  FLEXIO_SetShifterErrorInterrupt(baseAddr, (uint8_t)(1U << TX_SHIFTER(resourceIndex)), false);
273  /* Disable timer interrupt */
274  FLEXIO_SetTimerInterrupt(baseAddr, (uint8_t)(1U << TX_TIMER(resourceIndex)), false);
275  break;
276  case FLEXIO_DRIVER_TYPE_POLLING:
277  /* Nothing to do here */
278  break;
279  case FLEXIO_DRIVER_TYPE_DMA:
280  /* For Tx we need to disable timer interrupt */
281  FLEXIO_SetTimerInterrupt(baseAddr, (uint8_t)(1U << TX_TIMER(resourceIndex)), false);
282  /* Stop DMA channels */
283  (void)EDMA_DRV_StopChannel(state->dmaChannel);
284  /* Disable the FlexIO DMA request */
285  FLEXIO_SetShifterDMARequest(baseAddr, (uint8_t)(1U << TX_SHIFTER(resourceIndex)), false);
286  break;
287  default:
288  /* Impossible type - do nothing */
289  break;
290  }
291 
292  state->remainingBytes = 0U;
293  state->driverIdle = true;
294  /* Signal transfer end for blocking transfers */
295  if (state->blocking == true)
296  {
297  (void)OSIF_SemaPost(&(state->idleSemaphore));
298  }
299 }

361 /*FUNCTION**********************************************************************
362  *
363  * Function Name : FLEXIO_UART_DRV_WaitTransferEnd
364  * Description : waits for the end of a blocking transfer
365  *
366  *END**************************************************************************/
367 static status_t FLEXIO_UART_DRV_WaitTransferEnd(flexio_uart_state_t *state, uint32_t timeout)
368 {
369  status_t osifError = STATUS_SUCCESS;
370 
371  switch (state->driverType)
372  {
373  case FLEXIO_DRIVER_TYPE_INTERRUPTS:
374  /* Wait for transfer to be completed by the IRQ */
375  osifError = OSIF_SemaWait(&(state->idleSemaphore), timeout);
376  break;
377  case FLEXIO_DRIVER_TYPE_POLLING:
378  /* Call FLEXIO_UART_DRV_GetStatus() to do the transfer */
379  while (FLEXIO_UART_DRV_GetStatus(state, NULL) == STATUS_BUSY) {}
380  break;
381  case FLEXIO_DRIVER_TYPE_DMA:
382  /* Wait for transfer completion to be signaled by the DMA or IRQ */
383  osifError = OSIF_SemaWait(&(state->idleSemaphore), timeout);
384  break;
385  default:
386  /* Impossible type - do nothing */
387  break;
388  }
389 
390  /* Blocking transfer is over */
391  state->blocking = false;
392  if (osifError == STATUS_TIMEOUT)
393  {
394  /* Abort current transfer */
395  state->status = STATUS_TIMEOUT;
396  FLEXIO_UART_DRV_StopTransfer(state);
397  }
398 
399  return state->status;
400 }

405  * Function Name : FLEXIO_UART_DRV_ReadData
406  * Description : reads data received by the module
407  *
408  *END**************************************************************************/
409 static void FLEXIO_UART_DRV_ReadData(flexio_uart_state_t *state)
410 {
411  const FLEXIO_Type *baseAddr;
412  uint32_t data;
413  uint8_t resourceIndex; /* Index of first used internal resource instance (shifter and timer) */
414 
415  DEV_ASSERT(state->remainingBytes > 0U);
416  DEV_ASSERT(state->rxData != NULL);
417 
418  baseAddr = g_flexioBase[state->flexioCommon.instance];
419  resourceIndex = state->flexioCommon.resourceIndex;
420 
421  /* Read data from shifter buffer */
422  data = FLEXIO_ReadShifterBuffer(baseAddr, RX_SHIFTER(resourceIndex), FLEXIO_SHIFTER_RW_MODE_NORMAL);
423  data >>= 32U - (uint32_t)(state->bitCount);
424 
425  if (state->bitCount <= 8U)
426  {
427  *(uint8_t *)state->rxData = (uint8_t)data;
428  /* Update rx buffer pointer and remaining bytes count */
429  state->rxData ++;
430  state->remainingBytes -= 1U;
431  }
432  else
433  {
434  /* For more than 8 bits per word 2 bytes are needed */
435  *(uint16_t *)state->rxData = (uint16_t)data;
436  /* Update rx buffer pointer and remaining bytes count */
437  state->rxData = &state->rxData[2U];
438  state->remainingBytes -= 2U;
439  }
440 }

479 /*FUNCTION**********************************************************************
480  *
481  * Function Name : FLEXIO_UART_DRV_CheckStatusTx
482  * Description : Check status of the UART transmission. This function can be
483  * called either in an interrupt routine or directly in polling
484  * mode to advance the transfer.
485  *
486  *END**************************************************************************/
487 static void FLEXIO_UART_DRV_CheckStatusTx(void *stateStruct)
488 {
489  FLEXIO_Type *baseAddr;
490  uint8_t resourceIndex; /* Index of first used internal resource instance (shifter and timer) */
491  flexio_uart_state_t *state;
492 
493  DEV_ASSERT(stateStruct != NULL);
494 
495  state = (flexio_uart_state_t *)stateStruct;
496  baseAddr = g_flexioBase[state->flexioCommon.instance];
497  resourceIndex = state->flexioCommon.resourceIndex;
498 
499  /* No need to check for Tx underflow since timer is controlled by the shifter status flag */
500  /* Check for transfer end */
501  if (state->remainingBytes == 0U)
502  {
503  if (FLEXIO_GetTimerStatus(baseAddr, TX_TIMER(resourceIndex)))
504  {
505  /* Clear timer status */
506  FLEXIO_ClearTimerStatus(baseAddr, TX_TIMER(resourceIndex));
507  state->txFlush--;
508  if (state->txFlush == 0U)
509  {
510  /* Done flushing the Tx buffer, end transfer */
511  /* Record success if there was no error */
512  if (state->status == STATUS_BUSY)
513  {
514  state->status = STATUS_SUCCESS;
515  }
516  FLEXIO_UART_DRV_StopTransfer(state);
517  /* Call callback to announce the end transfer event to the user */
518  if (state->callback != NULL)
519  {
520  state->callback(state, UART_EVENT_END_TRANSFER, state->callbackParam);
521  }
522  }
523  else if (FLEXIO_GetShifterStatus(baseAddr, TX_SHIFTER(resourceIndex)))
524  {
525  /* txFlush == 1, but last byte was already transferred from buffer to shifter. There is a
526  danger that the transmission is over and we end up never reporting the end event.
527  To avoid this, send one extra dummy byte */
528  /* Set start bit to 1 and send an 0xFF byte, this way the line will appear idle */
529  FLEXIO_SetShifterStartBit(baseAddr, TX_SHIFTER(resourceIndex), FLEXIO_SHIFTER_START_BIT_1);
530  FLEXIO_WriteShifterBuffer(baseAddr, TX_SHIFTER(resourceIndex), 0xFFFFFFFFUL, FLEXIO_SHIFTER_RW_MODE_NORMAL);
531  }
532  else
533  {
534  /* txFlush == 1, and last byte was not yet transferred from buffer to shifter.
535  No need to do anything, just wait for the next timer event. */
536  }
537  }
538  }
539  /* Check if transmitter needs more data */
540  else if (FLEXIO_GetShifterStatus(baseAddr, TX_SHIFTER(resourceIndex)) && (state->remainingBytes > 0U))
541  {
542  FLEXIO_UART_DRV_WriteData(state);
543  if (state->remainingBytes == 0U)
544  {
545  /* Out of data, call callback to allow user to provide a new buffer */
546  if (state->callback != NULL)
547  {
548  state->callback(state, UART_EVENT_TX_EMPTY, state->callbackParam);
549  }
550  }
551  if (state->remainingBytes == 0U)
552  {
553  /* No more data, transmission will stop after the last bytes are sent.
554  The timer event will indicate when the send is complete */
555  /* Clear any previous timer events */
556  FLEXIO_ClearTimerStatus(baseAddr, TX_TIMER(resourceIndex));
557  if (state->driverType == FLEXIO_DRIVER_TYPE_INTERRUPTS)
558  {
559  /* Transmission completed; disable interrupt */
560  FLEXIO_SetShifterInterrupt(baseAddr, (uint8_t)(1U << TX_SHIFTER(resourceIndex)), false);
561  /* Enable timer interrupt to ensure that transfer is completed */
562  FLEXIO_SetTimerInterrupt(baseAddr, (uint8_t)(1U << TX_TIMER(resourceIndex)), true);
563  }
564  }
565  }
566  else
567  {
568  /* No relevant events - nothing to do */
569  }
570 
571 }

573 /*FUNCTION**********************************************************************
574  *
575  * Function Name : FLEXIO_UART_DRV_CheckStatusRx
576  * Description : Check status of the UART reception. This function can be
577  * called either in an interrupt routine or directly in polling
578  * mode to advance the transfer.
579  *
580  *END**************************************************************************/

583  FLEXIO_Type *baseAddr;

585  flexio_uart_state_t *state;

587  DEV_ASSERT(stateStruct != NULL);

589  state = (flexio_uart_state_t *)stateStruct;
590  baseAddr = g_flexioBase[state->flexioCommon.instance];
591  resourceIndex = state->flexioCommon.resourceIndex;
592 
593  /* Check for errors */
594  if (FLEXIO_GetShifterErrorStatus(baseAddr, RX_SHIFTER(resourceIndex)))
595  {
596  state->status = STATUS_UART_RX_OVERRUN;
597  FLEXIO_ClearShifterErrorStatus(baseAddr, RX_SHIFTER(resourceIndex));
598  state->remainingBytes = 0U;
599  /* Continue processing events */
600  }
601  /* Check if data was received */
602  else if (FLEXIO_GetShifterStatus(baseAddr, RX_SHIFTER(resourceIndex)))
603  {
604  FLEXIO_UART_DRV_ReadData(state);
605  if (state->remainingBytes == 0U)
606  {
607  /* Out of data, call callback to allow user to provide a new buffer */
608  if (state->callback != NULL)
609  {
610  state->callback(state, UART_EVENT_RX_FULL, state->callbackParam);
611  }
612  }
613  }
614  else
615  {
616  /* No events - nothing to do */
617  }
618  /* Check if transfer is over */
619  if (state->remainingBytes == 0U)
620  {
621  /* Record success if there was no error */
622  if (state->status == STATUS_BUSY)
623  {
624  state->status = STATUS_SUCCESS;
625  }
626  /* Discard any leftover rx. data */
627  FLEXIO_ClearShifterStatus(baseAddr, RX_SHIFTER(resourceIndex));
628  /* End transfer */
629  FLEXIO_UART_DRV_StopTransfer(state);
630  /* Call callback to announce the event to the user */
631  if (state->callback != NULL)
632  {
633  state->callback(state, UART_EVENT_END_TRANSFER, state->callbackParam);
634  }
635  }
636 }
