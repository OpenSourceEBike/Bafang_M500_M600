247 /*FUNCTION**********************************************************************
248  *
249  * Function Name : ADC_Init
250  * Description : This function initializes the ADC PAL instance, including the
251  * other platform specific HW units used together with ADC.
252  *
253  * Implements : ADC_Init_Activity
254  * END**************************************************************************/
255 status_t ADC_Init(const adc_instance_t * const instance,
256  const adc_config_t * const config)
257 {
258  DEV_ASSERT(instance != NULL);
259  DEV_ASSERT(config != NULL);
260  DEV_ASSERT(config->extension != NULL);
261  DEV_ASSERT(config->groupConfigArray != NULL);
262 
263  uint32_t instIdx = instance->instIdx;
264  DEV_ASSERT(instIdx < NUMBER_OF_ADC_PAL_INSTANCES);
265 
266  adc_pal_state_t * const palState = &(adcPalState[instIdx]);
267  status_t status = STATUS_SUCCESS;
268  uint8_t idx = 0u;
269 
270  /* Initialize PAL state structure members common for all platforms */
271  palState->groupArray = config->groupConfigArray;
272  palState->numGroups = config->numGroups;
273  for (idx = 0; idx < ADC_PAL_MAX_NUM_HW_GROUPS_EN; idx++)
274  {
275  palState->hwTrigGroupState[idx].active = false;
276  palState->hwTrigGroupState[idx].bufferLength = 0u;
277  palState->hwTrigGroupState[idx].currentBufferOffset = 0u;
278  palState->hwTrigGroupState[idx].notificationEn = false;
279  }
280 
281  palState->swTrigGroupState.active = false;
282  palState->swTrigGroupState.bufferLength = 0u;
283  palState->swTrigGroupState.currentBufferOffset = 0u;
284  palState->swTrigGroupState.notificationEn = false;
285  palState->latestGroupIdx = 0u;
286 
287 #if defined (ADC_PAL_S32K1xx)
288 
289  /* Initialize state members specific for this platform */
290  const extension_adc_s32k1xx_t * const extension = (extension_adc_s32k1xx_t *)(config->extension);
291  palState->pdbPrescaler = extension->pdbPrescaler;
292 
293  status = ADC_Init_S32K1xx(instIdx, config);
294 
295 #elif (defined (ADC_PAL_MPC574xC_G_R) || defined (ADC_PAL_SAR_CTU))
296 
297  /* Initialize state members specific for this platform */
298  uint8_t hwTrigGroupIdx = 0;
299  for (idx = 0; idx < palState->numGroups; idx++)
300  {
301  if (palState->groupArray[idx].hwTriggerSupport == true)
302  {
303  /* For hwTrigGroups, stateIdxMapping holds the index in hwTrigGroupState array */
304  palState->stateIdxMapping[idx] = hwTrigGroupIdx;
305  hwTrigGroupIdx++;
306  }
307  else
308  {
309  /* A single SW triggered group active is supported,
310  * so for swTrigGroups, stateIdxMapping holds a value which is not used. */
311  palState->stateIdxMapping[idx] = 0u;
312  }
313  }
314 
315 #if defined (ADC_PAL_MPC574xC_G_R)
316 
317  status = ADC_Init_MPC574xC_G_R(instIdx, config);
318 
319 #else
320 
321  status = ADC_SAR_CTU_Init(instIdx, config);
322 
323 #endif /* defined(ADC_PAL_MPC574xC_G_R) */
324 
325 #endif /* (defined(ADC_PAL_MPC574xC_G_R) || defined(ADC_PAL_SAR_CTU)) */
326 
327  return status;
328 }

330 /*FUNCTION**********************************************************************
331  *
332  * Function Name : ADC_Deinit
333  * Description : This function resets the ADC PAL instance, including
334  * the other platform specific HW units used together with ADC.
335  *
336  * Implements : ADC_Deinit_Activity
337  * END**************************************************************************/
338 status_t ADC_Deinit(const adc_instance_t * const instance)
339 {
340  DEV_ASSERT(instance != NULL);
341 
342  uint32_t instIdx = instance->instIdx;
343  DEV_ASSERT(instIdx < NUMBER_OF_ADC_PAL_INSTANCES);
344 
345  adc_pal_state_t * const palState = &(adcPalState[instIdx]);
346  bool hwTrigGroupActive = false;
347  status_t status;
348  uint16_t idx;
349 
350  hwTrigGroupActive = ADC_AnyHwTrigGroupActive(instIdx);
351 
352  if ((palState->swTrigGroupState.active == true) || (hwTrigGroupActive == true))
353  {
354  status = STATUS_BUSY;
355  }
356  else
357  {
358 #if defined (ADC_PAL_S32K1xx)
359 
360  /* No state members specific for this platform to reset */
361 
362  PDB_DRV_Deinit(instIdx);
363 
364  status = TRGMUX_DRV_SetTrigSourceForTargetModule(ADC_PAL_TRGMUX_IDX, TRGMUX_TRIG_SOURCE_DISABLED, adcPalTrgmuxTarget[instIdx]);
365  DEV_ASSERT(status == STATUS_SUCCESS);
366 
367  ADC_DRV_Reset(instIdx);
368 
369  /* Disable interrupt from INT manager */
370  IRQn_Type adcIrqId;
371  adcIrqId = ADC_DRV_GetInterruptNumber(instIdx);
372  INT_SYS_DisableIRQ(adcIrqId);
373 
374  status = STATUS_SUCCESS;
375 
376 #elif defined (ADC_PAL_MPC574xC_G_R)
377 
378  /* Reset state members specific for this platform */
379  for (idx = 0; idx < palState->numGroups; idx++)
380  {
381  palState->stateIdxMapping[idx] = 0u;
382  }
383 
384  /* Deinit BCTU and ADCs */
385  status_t bctuStatus;
386  bctuStatus = BCTU_DRV_Reset(instIdx, /* default timeout value 2ms*/ 2u);
387  if (bctuStatus == STATUS_TIMEOUT)
388  {
389  status = STATUS_BUSY;
390  }
391  else
392  {
393  ADC_ResetAdcMPC574x(instIdx);
394 
395  status = STATUS_SUCCESS;
396  }
397 
398 #elif defined (ADC_PAL_SAR_CTU)
399 
400  /* Reset state members specific for this platform */
401  for (idx = 0; idx < palState->numGroups; idx++)
402  {
403  palState->stateIdxMapping[idx] = 0u;
404  }
405 
406  CTU_DRV_Reset(instIdx);
407 
408  ADC_ResetAdcMPC574x(instIdx);
409 
410  status = STATUS_SUCCESS;
411 
412 #endif /* defined(ADC_PAL_SAR_CTU) */
413 
414  /* Reset PAL state structure members common for all platforms */
415  palState->groupArray = NULL;
416  palState->numGroups = 0u;
417  for (idx = 0; idx < ADC_PAL_MAX_NUM_HW_GROUPS_EN; idx++)
418  {
419  palState->hwTrigGroupState[idx].active = false;
420  }
421 
422  palState->swTrigGroupState.active = false;
423  palState->latestGroupIdx = 0u;
424  }
425 
426  return status;
427 }

429 /*FUNCTION**********************************************************************
430  *
431  * Function Name : ADC_EnableHardwareTrigger
432  * Description : Enables the selected HW trigger for a conversion group,
433  * if the conversion group has support for HW trigger.
434  *
435  * Implements : ADC_EnableHardwareTrigger_Activity
436  * END**************************************************************************/
437 status_t ADC_EnableHardwareTrigger(const adc_instance_t * const instance,
438  const uint32_t groupIdx)
439 {
440  DEV_ASSERT(instance != NULL);
441 
442  uint32_t instIdx = instance->instIdx;
443  DEV_ASSERT(instIdx < NUMBER_OF_ADC_PAL_INSTANCES);
444  DEV_ASSERT(groupIdx < adcPalState[instIdx].numGroups);
445  DEV_ASSERT(adcPalState[instIdx].groupArray != NULL);
446 
447  const adc_pal_state_t * const palState = &(adcPalState[instIdx]);
448  const adc_group_config_t * const currentGroupCfg = &(palState->groupArray[groupIdx]);
449  const adc_group_state_t * const currentGroupState = ADC_GetHwGroupStatePtr(instIdx, groupIdx);
450  status_t status = STATUS_SUCCESS;
451  status_t statusTrgmux = STATUS_SUCCESS;
452  (void)statusTrgmux;
453 
454  DEV_ASSERT(currentGroupCfg->hwTriggerSupport == true);
455  (void)currentGroupCfg;
456 
457  if ((palState->swTrigGroupState.active == true) || (currentGroupState->active == true))
458  {
459  /* A conversion group is already active */
460  status = STATUS_BUSY;
461  }
462  else
463  {
464  /* Update ADC PAL and group state structures */
465  ADC_PalStateUpdateStart(instIdx, groupIdx);
466 
467 #if defined (ADC_PAL_S32K1xx)
468 
469  /* Enable the ADC interrupt from Interrupt Manager */
470  IRQn_Type adcIrqId;
471  adcIrqId = ADC_DRV_GetInterruptNumber(instIdx);
472  INT_SYS_EnableIRQ(adcIrqId);
473 
474  /* The group shall be configured each time it is enabled. */
475  ADC_ConfigGroup(instIdx, groupIdx, true);
476 
477  /* Enable in TRGMUX the selected HW trigger source for PDB.
478  * Must be called after all PDB pre-triggers have been configured, to make sure no triggers occur during configuration. */
479  statusTrgmux = TRGMUX_DRV_SetTrigSourceForTargetModule(ADC_PAL_TRGMUX_IDX, currentGroupCfg->triggerSource, adcPalTrgmuxTarget[instIdx]);
480  DEV_ASSERT(statusTrgmux == STATUS_SUCCESS);
481 
482 #elif defined (ADC_PAL_MPC574xC_G_R)
483  const uint8_t numChans = (palState->groupArray[groupIdx]).numChannels;
484 
485  if (numChans == 1u)
486  {
487  INT_SYS_EnableIRQ(BCTU_ConvUpdate_IRQn);
488  }
489  else
490  {
491  INT_SYS_EnableIRQ(BCTU_ListLast_IRQn);
492  }
493 
494  /* De-activate low-power mode. Any pending triggers will become active, immediately triggering a conversion. */
495  BCTU_DRV_SetLowPowerMode(instIdx, false);
496 
497  /* All groups get configured at ADC PAL initialization,
498  * so here they only need to be enabled. */
499  BCTU_DRV_EnableHwTrigger(instIdx, currentGroupCfg->triggerSource);
500 
501 #elif defined (ADC_PAL_SAR_CTU)
502  /* internalTrigIdx is mapped 1:1 with stateIdx */
503  const uint32_t internalTrigIdx = palState->stateIdxMapping[groupIdx];
504  const uint32_t shift = internalTrigIdx * CTU_THCR1_T1_ADCE_SHIFT;
505  CTU_Type * const ctuInstBase = ctuBase[instIdx];
506 
507  /* Enable internal trigger corresponding to current groupIdx */
508  ctuInstBase->THCR1 |= (uint32_t)CTU_OUTPUT_TRIG_EN << shift; /* Assumes: internalTrigIdx < 4 */
509 
510  /* THCR is double-buffered, so mark that MRS can reload the value */
511  CTU_DRV_EnableGeneralReload(instIdx);
512 
513 #endif /* defined(ADC_PAL_SAR_CTU) */
514 
515  status = STATUS_SUCCESS;
516  }
517 
518  return status;
519 }

521 /*FUNCTION**********************************************************************
522  *
523  * Function Name : ADC_DisableHardwareTrigger
524  * Description : Disables the selected HW trigger for a conversion group,
525  * if the conversion group has support for HW trigger.
526  *
527  * Implements : ADC_DisableHardwareTrigger_Activity
528  * END**************************************************************************/
529 status_t ADC_DisableHardwareTrigger(const adc_instance_t * const instance,
530  const uint32_t groupIdx,
531  const uint32_t timeout)
532 {
533  DEV_ASSERT(instance != NULL);
534 
535  uint32_t instIdx = instance->instIdx;
536  adc_group_state_t * const currentGroupState = ADC_GetHwGroupStatePtr(instIdx, groupIdx);
537 
538  DEV_ASSERT(instIdx < NUMBER_OF_ADC_PAL_INSTANCES);
539  DEV_ASSERT(groupIdx < adcPalState[instIdx].numGroups);
540  DEV_ASSERT(adcPalState[instIdx].groupArray != NULL);
541  DEV_ASSERT(adcPalState[instIdx].groupArray[groupIdx].hwTriggerSupport == true);
542 
543  status_t status = STATUS_SUCCESS;
544 
545  /* Make sure no more notifications are received from current groupIdx */
546  status = ADC_DisableNotification(instance, groupIdx);
547 
548  if ((currentGroupState->active == true) && (status == STATUS_SUCCESS))
549  {
550 #if defined (ADC_PAL_S32K1xx)
551  /* Disable from TRGMUX the trigger source for PDB */
552  status = TRGMUX_DRV_SetTrigSourceForTargetModule(ADC_PAL_TRGMUX_IDX, TRGMUX_TRIG_SOURCE_DISABLED, adcPalTrgmuxTarget[instIdx]);
553  DEV_ASSERT(status == STATUS_SUCCESS);
554 
555  status = ADC_StopGroupBlocking(instIdx, timeout);
556 
557 #elif defined (ADC_PAL_MPC574xC_G_R)
558  const adc_pal_state_t * const palState = &(adcPalState[instIdx]);
559  const adc_trigger_source_t triggerSource = palState->groupArray[groupIdx].triggerSource;
560  const uint8_t numChans = palState->groupArray[groupIdx].numChannels;
561  uint32_t startTime, deltaTime;
562 
563  if (numChans == 1u)
564  {
565  INT_SYS_DisableIRQ(BCTU_ConvUpdate_IRQn);
566  }
567  else
568  {
569  INT_SYS_DisableIRQ(BCTU_ListLast_IRQn);
570  }
571 
572  OSIF_TimeDelay(0u); /* Make sure OSIF timer is initialized. */
573 
574  startTime = OSIF_GetMilliseconds();
575  deltaTime = 0u;
576 
577  /* Prevent new conversions from starting */
578  BCTU_DRV_DisableHwTrigger(instIdx, triggerSource);
579 
580  /* Check if conversion is still running until execution finished or timeout occurred */
581  while ((BCTU_DRV_IsConvRunning(instIdx, triggerSource) == true) && (deltaTime < timeout))
582  {
583  deltaTime = OSIF_GetMilliseconds() - startTime;
584  }
585 
586  if (deltaTime >= timeout)
587  {
588  status = STATUS_TIMEOUT;
589  }
590  else
591  {
592  status = STATUS_SUCCESS;
593 
594  /* Activate low-power mode */
595  BCTU_DRV_SetLowPowerMode(instIdx, true);
596  }
597 
598 #elif defined (ADC_PAL_SAR_CTU)
599  const adc_pal_state_t * const palState = &(adcPalState[instIdx]);
600  /* internalTrigIdx is mapped 1:1 with stateIdx */
601  const uint32_t internalTrigIdx = palState->stateIdxMapping[groupIdx];
602  const uint32_t shift = internalTrigIdx * CTU_THCR1_T1_ADCE_SHIFT;
603  CTU_Type * const ctuInstBase = ctuBase[instIdx];
604 
605  (void)timeout;
606 
607  /* Disable internal trigger corresponding to current groupIdx */
608  /* Register is double-buffered so the new value will not have effect in the current control cycle
609  * - only after a new MRS occurs */
610  ctuInstBase->THCR1 &= ~((uint32_t)CTU_OUTPUT_TRIG_EN << shift); /* Assumes: fifoIdx < 4 */
611 
612  /* THCR is double-buffered, so mark that MRS can reload the value. */
613  CTU_DRV_EnableGeneralReload(instIdx);
614 
615  currentGroupState->active = false;
616 
617 #endif /* defined(ADC_PAL_SAR_CTU) */
618 
619 #ifndef ADC_PAL_SAR_CTU
620 
621  if (status == STATUS_SUCCESS)
622  {
623  currentGroupState->active = false;
624  }
625 
626 #endif /* ADC_PAL_SAR_CTU */
627  }
628 
629 
630  return status;
631 }

644

647 
648  const adc_pal_state_t * const palState = &(adcPalState[instIdx]);

651  DEV_ASSERT(palState->groupArray[groupIdx].hwTriggerSupport == false);
652 
653  status_t status;
654 
655  if ((ADC_AnyHwTrigGroupActive(instIdx) == true) || (palState->swTrigGroupState.active == true))

657  /* Any other conversion group is already active */
658  status = STATUS_BUSY;
659  }

661  {
662  /* Update ADC PAL and group state structures */

667 
668  bool hwTriggerEnabled = false;
669 
670  /* Enable the ADC interrupt from Interrupt Manager */
