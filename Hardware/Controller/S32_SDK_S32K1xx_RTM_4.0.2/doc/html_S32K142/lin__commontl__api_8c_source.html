49 void ld_init(l_ifc_handle iii)
50 {
51  DEV_ASSERT((l_u8)iii < LIN_NUM_OF_IFCS);
52  lin_transport_layer_queue_t * tl_queue;
53  static lin_product_id_t product_id_data[LIN_NUM_OF_IFCS];
54  lin_tl_descriptor_t * tl_desc_ptr = &g_lin_tl_descriptor_array[iii];
55  l_u16 length = g_lin_protocol_user_cfg_array[iii].max_message_length;
56  l_u16 max_queue_size;
57 
58  /* Calculate max_queue_size from max_message_length */
59  if (length <= 6U)
60  {
61  max_queue_size = 1U;
62  }
63  else
64  {
65  if (((length - 5U) % 6U) == 0U)
66  {
67  max_queue_size = (l_u16)(((length - 5U) / 6U) + 1U);
68  }
69  else
70  {
71  max_queue_size = (l_u16)(((length - 5U) / 6U) + 2U);
72  }
73  }
74 
75  /* Initial global variables */
76  tl_desc_ptr->tl_tx_queue.tl_pdu_ptr = g_lin_protocol_user_cfg_array[iii].tl_tx_queue_data_ptr;
77  tl_desc_ptr->tl_rx_queue.tl_pdu_ptr = g_lin_protocol_user_cfg_array[iii].tl_rx_queue_data_ptr;
78 
79  tl_desc_ptr->product_id_ptr = &product_id_data[iii];
80  tl_desc_ptr->tl_tx_queue.queue_max_size = max_queue_size;
81  tl_desc_ptr->tl_rx_queue.queue_max_size = max_queue_size;
82 
83  /* Initialize transmit queue */
84  /* Get TL tx queue */
85  tl_queue = &(tl_desc_ptr->tl_tx_queue);
86  tl_queue->queue_header = 0;
87  tl_queue->queue_tail = 0;
88  tl_queue->queue_status = LD_QUEUE_EMPTY;
89  tl_queue->queue_current_size = 0;
90  /* Initialize receive queue */
91  /* Get TL rx queue */
92  tl_queue = &(tl_desc_ptr->tl_rx_queue);
93  tl_queue->queue_header = 0;
94  tl_queue->queue_tail = 0;
95  tl_queue->queue_status = LD_NO_DATA;
96  tl_queue->queue_current_size = 0;
97  /* Initialize transmit message */
98  tl_desc_ptr->rx_msg_status = LD_COMPLETED;
99  /* Initialize receive message */
100  tl_desc_ptr->tx_msg_status = LD_COMPLETED;
101 
102  tl_desc_ptr->last_cfg_result = LD_SUCCESS;
103  tl_desc_ptr->last_RSID = 0;
104  tl_desc_ptr->ld_error_code = 0;
105 
106  tl_desc_ptr->frame_counter = 0;
107  tl_desc_ptr->num_of_pdu = 0;
108  tl_desc_ptr->slave_resp_cnt = 0;
109 
110  tl_desc_ptr->check_timeout_type = LD_NO_CHECK_TIMEOUT;
111  tl_desc_ptr->check_timeout = 0;
112 
113  tl_desc_ptr->diag_state = LD_DIAG_IDLE;
114  tl_desc_ptr->service_status = LD_SERVICE_IDLE;
115  tl_desc_ptr->diag_interleave_state = DIAG_NOT_START;
116  tl_desc_ptr->interleave_timeout_counter = 0;
117  tl_desc_ptr->FF_pdu_received = false;

200 /*FUNCTION**********************************************************************
201  *
202  * Function Name : ld_send_message
203  * Description : Pack the information specified by data and length into one or multiple diagnostic frames.
204  *
205  * Implements : ld_send_message_Activity
206  *END**************************************************************************/
207 void ld_send_message(l_ifc_handle iii,
208  l_u16 length,
209  l_u8 NAD,
210  const l_u8 * const data)
211 {
212  DEV_ASSERT((l_u8)iii < LIN_NUM_OF_IFCS);
213  DEV_ASSERT(data != NULL);
214 
215  const lin_transport_layer_queue_t * tl_queue;
216  lin_tl_descriptor_t * tl_desc_ptr = &g_lin_tl_descriptor_array[iii];
217  l_u8 message_size;
218 
219  /* Get transmit queue */
220  tl_queue = &(tl_desc_ptr->tl_tx_queue);
221  /* check message status in queue */
222  if (LD_COMPLETED == tl_desc_ptr->tx_msg_status)
223  {
224  /* calculate number of PDU message_size for this message */
225  if (length <= 6U)
226  {
227  message_size = 1U;
228  }
229  else
230  {
231  if (((length - 5U) % 6U) == 0U)
232  {
233  message_size = (l_u8)(((length - 5U) / 6U) + 1U);
234  }
235  else
236  {
237  message_size = (l_u8)(((length - 5U) / 6U) + 2U);
238  }
239  }
240 
241  /* Check if the message can be put into queue */
242  if (message_size <= (tl_queue->queue_max_size - tl_queue->queue_current_size))
243  {
244  /* update information of message in queue */
245  tl_desc_ptr->tx_msg_status = LD_IN_PROGRESS;
246  tl_desc_ptr->service_status = LD_SERVICE_BUSY;
247 
248  lin_tl_package_pdu(iii, length, message_size, NAD, data);
249  } /* end of check message size */
250  } /* end of (LD_COMPLETED == tl_conf->tl_message_status) */
251 } /* end of ld_send_message */

271 
272  if((tl_desc_ptr->rx_msg_status == LD_COMPLETED) && (*length != 0U))
273  {

275  tl_desc_ptr->rx_msg_status = LD_IN_PROGRESS;
276

281  }
