63 /*FUNCTION**********************************************************************
64  *
65  * Function Name : l_sch_set
66  * Description : This function sets up the next schedule to be followed by the
67  * l_sch_tick function for a certain interface iii. The new schedule will be activated as soon as the current schedule reaches
68  * its next schedule entry point. The entry defines the starting entry point in the new schedule table. The value should
69  * be in the range 0 to N if the schedule table has N entries
70  *
71  * Implements : l_sch_set_Activity
72  *END**************************************************************************/
73 void l_sch_set(l_ifc_handle iii,
74  l_schedule_handle schedule_iii,
75  l_u8 entry)
76 {
77  DEV_ASSERT((l_u8)iii < LIN_NUM_OF_IFCS);
78  DEV_ASSERT((l_u8)schedule_iii < LIN_NUM_OF_SCHD_TBL);
79  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
80 
81  /* Check if the interface is LIN Master */
82  if (prot_user_config_ptr->function == (bool)LIN_MASTER)
83  {
84  /* Check if input entry is in range from 0 to N = (Number of frame slots) */
85  if (entry <= prot_user_config_ptr->schedule_tbl[schedule_iii].num_slots)
86  {
87  l_u8 tmp_entry = entry;
88  lin_master_data_t * master_data_ptr = &g_lin_master_data_array[g_lin_protocol_user_cfg_array[iii].master_ifc_handle];
89  /* save the new schedule which is activated after the */
90  /* current schedule reaches its next schedule entry point */
91  master_data_ptr->previous_schedule_id = master_data_ptr->active_schedule_id;
92  /* Insert new schedule at that point of the current schedule */
93  master_data_ptr->active_schedule_id = (l_u8)schedule_iii;
94 
95  /* if entry is 0 or 1 the new schedule table will be started from the beginning. */
96  if (tmp_entry == 0U)
97  {
98  /* Set entry equal to 1 */
99  tmp_entry = 1U;
100  }
101 
102  /* Assign the start entry of new schedule table by argument entry */
103  master_data_ptr->schedule_start_entry_ptr[schedule_iii] = (l_u8)(tmp_entry - 1U);
104  }
105  }
106 }

108 /*FUNCTION**********************************************************************
109  *
110  * Function Name : lin_master_update_signal
111  * Description : This function handles update signal for master node.
112  * This function is implemented for Master.
113  *
114  * Implements : lin_master_update_signal_Activity
115  *END**************************************************************************/
116 static void lin_master_update_signal(l_ifc_handle iii)
117 {
118  l_u8 i;
119  l_u8 size;
120  l_u16 offset;
121  lin_master_data_t * master_data_ptr;
122  master_data_ptr = &g_lin_master_data_array[g_lin_protocol_user_cfg_array[iii].master_ifc_handle];
123 
124  if (master_data_ptr->frm_offset != 0xFFFFU)
125  {
126  /* Copy data from master_data_buffer to g_lin_frame_data_buffer */
127  /* Do this to satisfy Requirement: The master node updates its
128  * received signals periodically at the time base start */
129  size = master_data_ptr->frm_size;
130  offset = master_data_ptr->frm_offset;
131 
132  /* Check if size and offset are valid */
133  if ((size <= 8U) && ((size + offset) <= LIN_FRAME_BUF_SIZE))
134  {
135  /* Copy data from master_data_buffer to g_lin_frame_data_buffer */
136  for (i = 0U; i < size; i++)
137  {
138  g_lin_frame_data_buffer[offset + i] = master_data_ptr->master_data_buffer[i];
139  }
140  }
141 
142  master_data_ptr->frm_offset = 0xFFFFU;
143  }
144 
145  if (master_data_ptr->flag_offset != 0xFFU)
146  {
147  /* Do this to satisfy Requirement: The master node updates its
148  * received signals periodically at the time base start */
149  size = master_data_ptr->flag_size;
150  offset = master_data_ptr->flag_offset;
151 
152  /* Check if size and offset are valid */
153  if ((size <= LIN_FLAG_BUF_SIZE) && ((size + offset) <= LIN_FLAG_BUF_SIZE))
154  {
155  /* Update flag */
156  for (i = 0U; i < size; i++)
157  {
158  g_lin_flag_handle_tbl[offset + i] = 0xFF;
159  }
160 
161  master_data_ptr->flag_offset = 0xFF;
162  }
163  }
164 }

166 /*FUNCTION**********************************************************************
167  *
168  * Function Name : lin_master_update_frame_before_transmit
169  * Description : This function update signal in transmitted frame of master.
170  * This function is implemented for Master.
171  *
172  * Implements : lin_master_update_frame_before_transmit_Activity
173  *END**************************************************************************/
174 static void lin_master_update_frame_before_transmit(l_ifc_handle iii, l_u8 cur_id, l_u8 frame_index)
175 {
176  l_u8 i;
177  l_u8 size;
178  l_u16 offset;
179  l_u8 active_schedule_id;
180  l_u8 flag;
181  const l_u8 * current_entry;
182  const lin_protocol_state_t * prot_state_ptr;
183  const lin_schedule_data_t * sch_data_ptr;
184  const lin_protocol_user_config_t * prot_user_config_ptr;
185  const lin_schedule_t * active_schedule_struct_ptr;
186  lin_master_data_t * master_data_ptr = &g_lin_master_data_array[g_lin_protocol_user_cfg_array[iii].master_ifc_handle];
187 
188  if (0x3CU == cur_id)
189  {
190  active_schedule_id = master_data_ptr->active_schedule_id;
191  active_schedule_struct_ptr = (const lin_schedule_t *)&(g_lin_protocol_user_cfg_array[iii].schedule_tbl[active_schedule_id]);
192  if (LIN_SCH_TBL_DIAG != active_schedule_struct_ptr->sch_tbl_type)
193  {
194  current_entry = (l_u8 *)&master_data_ptr->schedule_start_entry_ptr[active_schedule_id];
195  prot_state_ptr = &g_lin_protocol_state_array[iii];
196  sch_data_ptr = &active_schedule_struct_ptr->ptr_sch_data_ptr[*current_entry];
197  /* Update transmitted data in case diagnostic frame is not added in diagnostic schedule table */
198  for (i = 0U; i < 8U; i++)
199  {
200  prot_state_ptr->response_buffer_ptr[i] = sch_data_ptr->tl_queue_data[i];
201  }
202  }
203  }
204  else
205  {
206  prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
207  /* Update buffer data if master is publisher and not transmit diagnostic request */
208  if (prot_user_config_ptr->frame_tbl_ptr[frame_index].frm_response == LIN_RES_PUB)
209  {
210  size = prot_user_config_ptr->frame_tbl_ptr[frame_index].frm_len;
211  offset = prot_user_config_ptr->frame_tbl_ptr[frame_index].frm_offset;
212  flag = g_lin_frame_updating_flag_tbl[frame_index];
213  /* Update data */
214  for (i = 0U; i < size; i++)
215  {
216  if((flag & (1U << i)) != 0U)
217  {
218  master_data_ptr->master_data_buffer[i] = g_buffer_backup_data[i];
219  }
220  else
221  {
222  master_data_ptr->master_data_buffer[i] = g_lin_frame_data_buffer[offset];
223  }
224  offset++;
225  }
226  }
227  }
228 }

230 /*FUNCTION**********************************************************************
231  *
232  * Function Name : l_sch_tick
233  * Description : This function follows a schedule. When a frame becomes due, its
234  * transmission is initiated. When the end of the current schedule is reached,
235  * this function starts again at the beginning of the schedule.
236  * This API is for Master interface only.
237  *
238  * Implements : l_sch_tick_Activity
239  *END**************************************************************************/
240 l_u8 l_sch_tick(l_ifc_handle iii)
241 {
242  DEV_ASSERT((l_u8)iii < LIN_NUM_OF_IFCS);
243  const lin_schedule_t * active_schedule_struct_ptr;
244  l_u8 * current_entry;
245  l_u8 frame_index;
246  l_u8 cur_id;
247  l_u8 active_schedule_id;
248  l_u8 retVal = 0U;
249 
250  const lin_tl_descriptor_t * tl_desc_ptr = &g_lin_tl_descriptor_array[iii];
251  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];
252  lin_protocol_state_t * prot_state_ptr = &g_lin_protocol_state_array[iii];
253 
254  lin_master_data_t * master_data_ptr;
255 
256  /* Check if the interface is LIN Master */
257  if (prot_user_config_ptr->function == (bool)LIN_MASTER)
258  {
259  master_data_ptr = &g_lin_master_data_array[g_lin_protocol_user_cfg_array[iii].master_ifc_handle];
260 
261  lin_master_update_signal(iii);
262 
263  /* Get active_schedule_id */
264  active_schedule_id = master_data_ptr->active_schedule_id;
265  cur_id = prot_state_ptr->current_id;
266 
267  /* process diagnostic interleaved mode */
268  if ((tl_desc_ptr->diag_interleave_state == DIAG_NO_RESPONSE) && (prot_state_ptr->diagnostic_mode == DIAG_INTERLEAVE_MODE) &&
269  (cur_id == 0x3DU))
270  {
271  /* go back normal schedule table */
272  active_schedule_id = master_data_ptr->previous_schedule_id;
273  master_data_ptr->active_schedule_id = active_schedule_id;
274  master_data_ptr->schedule_start_entry_ptr[active_schedule_id] = 0U;
275  }
276 
277  active_schedule_struct_ptr = (const lin_schedule_t *)&prot_user_config_ptr->schedule_tbl[active_schedule_id];
278  current_entry = (l_u8 *)&master_data_ptr->schedule_start_entry_ptr[active_schedule_id];
279 
280  if (LIN_SCH_TBL_NULL != active_schedule_struct_ptr->sch_tbl_type)
281  {
282  /* Check if next_transmit is 0 */
283  if (0U == prot_state_ptr->next_transmit_tick)
284  {
285  /* Check protocol and event trigger collision flag */
286  if ((prot_user_config_ptr->protocol_version != LIN_PROTOCOL_J2602) && (master_data_ptr->event_trigger_collision_flg == true))
287  {
288  /* Call collision resolver */
289  lin_collision_resolve(iii, cur_id);
290  /* Update active schedule table */
291  active_schedule_id = master_data_ptr->active_schedule_id;
292  active_schedule_struct_ptr = (const lin_schedule_t *)&prot_user_config_ptr->schedule_tbl[active_schedule_id];
293  /* Re-calculate current entry due to change table to collision */
294  current_entry = (l_u8 *)&master_data_ptr->schedule_start_entry_ptr[active_schedule_id];
295  master_data_ptr->event_trigger_collision_flg = false;
296  }
297 
298  /* Set new transmit tick */
299  prot_state_ptr->next_transmit_tick = active_schedule_struct_ptr->ptr_sch_data_ptr[*current_entry].delay_integer;
300  /* Get frame index to send */
301  frame_index = (l_u8)active_schedule_struct_ptr->ptr_sch_data_ptr[*current_entry].frm_id;
302 
303  if ((frame_index - prot_user_config_ptr->frame_start) < prot_user_config_ptr->number_of_configurable_frames)
304  {
305  cur_id = prot_user_config_ptr->list_identifiers_RAM_ptr[(frame_index - prot_user_config_ptr->frame_start) + 1U];
306  }
307 
308  if (LIN_FRM_SPRDC == prot_user_config_ptr->frame_tbl_ptr[frame_index].frm_type)
309  {
310  /* Sporadic frame */
311  cur_id = lin_check_sporadic_update(iii, frame_index);
312  frame_index = lin_get_frame_index(iii, cur_id);
313  }
314 
315  if ((0xFFU != cur_id) && (frame_index < LIN_NUM_OF_FRMS))
316  {
317  /* The latest point master can update value of signal in transmitted frame */
318  lin_master_update_frame_before_transmit(iii, cur_id, frame_index);
319 
320  /* Start transmit header */
321  (void)lin_lld_tx_header(iii, cur_id);
322  }
323  /* increase start entry */
324  *current_entry = (l_u8)(*current_entry + 1U);
325  /* Check if it is the last entry */
326  if (*current_entry >= active_schedule_struct_ptr->num_slots)
327  {
328  /* switch schedule table */
329  lin_switch_sch_table(iii);
330  }
331  }
332 
333  /* Decrease next_transmit_tick */
334  prot_state_ptr->next_transmit_tick -= 1U;
335 
336  /* if the next call of l_sch_tick will start the transmission of the frame in the next schedule table entry */
337  /* The return value will in this case be the next schedule table entry's number */
338  /* counted from the beginning of the schedule table) in the schedule table */
339  /* The return value will be in range 1 to N if the schedule table has N entries */
340  if (0U == prot_state_ptr->next_transmit_tick)
341  {
342  /* The return value will be in range 1 to N if the schedule table has N entries */
343  retVal = (l_u8)(*current_entry + 1U);
344  }
345  } /* End (LIN_SCH_TBL_NULL != active_schedule_struct_ptr->sch_tbl_type) */
346 
347  /* In the master node the status word is updated in the l_sch_tick function*/
348  switch (prot_user_config_ptr->protocol_version)
349  {
350  case LIN_PROTOCOL_20:
351  case LIN_PROTOCOL_21:
352  lin_update_word_status_lin21(iii, LIN_LLD_PID_OK);
353  break;
354  case LIN_PROTOCOL_J2602:
355  lin_update_word_status_j2602(iii);
356  break;
357  default:
358  /*do nothing*/
359  break;
360  }
361 
362  prot_state_ptr->current_id = cur_id;
363  } /* END (prot_user_config_ptr->function == LIN_MASTER) */
364 
365  return retVal;
366 }

397 /*FUNCTION**********************************************************************
398  *
399  * Function Name : l_ifc_init
400  * Description : Initialize the interface specified by name, i.e. sets up internal

403  * Implements : l_ifc_init_Activity

406 {
407  DEV_ASSERT((l_u8)iii < LIN_NUM_OF_IFCS);
408  lin_protocol_state_t * prot_state_ptr = &g_lin_protocol_state_array[iii];
409 #if (SUPPORT_MASTER_MODE == 1U)
410  const lin_protocol_user_config_t * prot_user_config_ptr = &g_lin_protocol_user_cfg_array[iii];

417  /* Call Low level initialization function */

420  if (ret_val == (l_bool)LIN_LLD_OK)
421  {
422  #if (SUPPORT_MASTER_MODE == 1U)
423  if (prot_user_config_ptr->function == (bool)LIN_MASTER)

428  master_data_ptr->frm_offset = 0xFFFFU;
429  master_data_ptr->flag_size = 0x00U;
430  master_data_ptr->frm_size = 0x00U;
431  master_data_ptr->schedule_start_entry_ptr = &lin_schedule_start_entry[0U];
432  master_data_ptr->send_slave_res_flg = (bool)1;
433  master_data_ptr->send_functional_request_flg = (bool)0;
434  /* For LIN Master, set NULL schedule table after initialization*/
435  /* current schedule reaches its next schedule entry point */
436  master_data_ptr->previous_schedule_id = master_data_ptr->active_schedule_id;
437  /* Insert new schedule at that point of the current schedule */
438  master_data_ptr->active_schedule_id = prot_user_config_ptr->schedule_start;
439  /* Assign the start entry of new schedule table by argument entry */
440  master_data_ptr->schedule_start_entry_ptr[prot_user_config_ptr->schedule_start] = 0;
441  }
442  #endif /* End (SUPPORT_MASTER_MODE == 1U) */
443  }
444 
445  return ret_val;

462 /*FUNCTION**********************************************************************
463  *
464  * Function Name : l_ifc_read_status
465  * Description : This function will return the status of the previous communication.
466  *
467  * Implements : l_ifc_read_status_Activity
468  *END**************************************************************************/
469 l_u16 l_ifc_read_status(l_ifc_handle iii)
470 {
471  DEV_ASSERT((l_u8)iii < LIN_NUM_OF_IFCS);
472 
473  l_u16 tmp_word_status = 0U;
474  lin_protocol_state_t * prot_state_ptr = &g_lin_protocol_state_array[iii];
475  tmp_word_status = (l_u16)(prot_state_ptr->word_status.last_pid << 8);
476  if (prot_state_ptr->word_status.save_config_flg == 1U)
477  {
478  tmp_word_status |= SAVE_CONFIG_SET;
479  }
480 
481  if (prot_state_ptr->word_status.event_trigger_collision_flg == 1U)
482  {
483  tmp_word_status |= EVENT_TRIGGER_COLLISION_SET;
484  }
485 
486  if (prot_state_ptr->word_status.bus_activity == 1U)
487  {
488  tmp_word_status |= BUS_ACTIVITY_SET;
489  }
490 
491  if (prot_state_ptr->word_status.go_to_sleep_flg == 1U)
492  {
493  tmp_word_status |= GO_TO_SLEEP_SET;
494  }
495 
496  if (prot_state_ptr->word_status.overrun == 1U)
497  {
498  tmp_word_status |= OVERRUN;
499  }
500 
501  if (prot_state_ptr->word_status.successful_transfer == 1U)
502  {
503  tmp_word_status |= SUCCESSFULL_TRANSFER;
504  }
505 
506  if (prot_state_ptr->word_status.error_in_res == 1U)
507  {
508  tmp_word_status |= ERROR_IN_RESPONSE;
509  }
510 
511  /* Read-reset call; meaning that after the call has returned,
512  * the status word is set to 0 */
513  /* Clear word status */
514  prot_state_ptr->word_status.last_pid = 0x00U;
515  prot_state_ptr->word_status.reserved = 0U;
516  prot_state_ptr->word_status.save_config_flg = 0U;
517  prot_state_ptr->word_status.event_trigger_collision_flg = 0U;
518  prot_state_ptr->word_status.bus_activity = 0U;
519  prot_state_ptr->word_status.go_to_sleep_flg = 0U;
520  prot_state_ptr->word_status.overrun = 0U;
521  prot_state_ptr->word_status.successful_transfer = 0U;
522  prot_state_ptr->word_status.error_in_res = 0U;
523  /* Clear save configuration flag value */
524  prot_state_ptr->save_config_flg = (bool)0U;
525  /* reset word status flags in g_lin_protocol_state_array for new session*/
526  prot_state_ptr->go_to_sleep_flg = false;
527  prot_state_ptr->overrun_flg = 0U;
528  prot_state_ptr->successful_transfer = 0U;
529 
530  /* Reset number of processed frames to 0 */
531  prot_state_ptr->num_of_processed_frame = 0U;
532 
533  return tmp_word_status;
534 }
