62  * Description : This function set all CMP registers to reset values.

64  * Implements : CMP_DRV_Reset_Activity
65  *END**************************************************************************/
66 status_t CMP_DRV_Reset(const uint32_t instance)
67 {
68  status_t status = STATUS_SUCCESS;
69  DEV_ASSERT(instance < CMP_INSTANCE_COUNT);
70  CMP_Type* baseAddr = g_cmpBase[instance];
71  baseAddr->C0 = C0_RESET_VALUE;
72  baseAddr->C1 = C1_RESET_VALUE;
73  baseAddr->C2 = C1_RESET_VALUE;
74  return status;

492 /*FUNCTION**********************************************************************
493  *
494  * Function Name : CMP_DRV_GetOutputFlags
495  * Description : Return in <flags> comparator output flags(rising and falling edge on output).
496  * Implements : CMP_DRV_GetOutputFlags_Activity
497  *
498  *END**************************************************************************/
499 status_t CMP_DRV_GetOutputFlags(const uint32_t instance, cmp_output_trigger_t *flags)
500 {
501  status_t status = STATUS_SUCCESS;
502  DEV_ASSERT(flags != NULL);
503  DEV_ASSERT(instance < CMP_INSTANCE_COUNT);
504  const CMP_Type* baseAddr = g_cmpBase[instance];
505  bool rising_enabled = (bool)(((baseAddr->C0) >> CMP_C0_CFR_SHIFT) & (uint32_t)1U);
506  bool falling_enabled = (bool)(((baseAddr->C0) >> CMP_C0_CFF_SHIFT) & (uint32_t)1U);
507  *flags = (cmp_output_trigger_t) (((uint32_t)rising_enabled << (uint32_t)1U) | ((uint32_t)falling_enabled));
508  return status;
509 }

527  tmp |= CMP_C0_CFF(1);

532  #endif
