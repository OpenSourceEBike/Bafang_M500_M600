57 /*FUNCTION**********************************************************************
58  *
59  * Function Name : CRC_DRV_Init
60  * Description : This function initializes CRC driver based on user configuration input.
61  * The user must make sure that the clock is enabled.
62  *
63  * Implements : CRC_DRV_Init_Activity
64  *END**************************************************************************/
65 status_t CRC_DRV_Init(uint32_t instance,
66  const crc_user_config_t * userConfigPtr)
67 {
68  DEV_ASSERT(instance < CRC_INSTANCE_COUNT);
69  DEV_ASSERT(userConfigPtr != NULL);
70  CRC_Type * base = s_crcBase[instance];
71  status_t retStatus = STATUS_SUCCESS;
72 
73  /* Set the default configuration */
74  CRC_Init(base);
75  /* Set the CRC configuration */
76  retStatus = CRC_DRV_Configure(instance, userConfigPtr);
77 
78  return retStatus;
79 }

99 /*FUNCTION**********************************************************************
100  *
101  * Function Name : CRC_DRV_GetCrc32
102  * Description : This function appends 32-bit data to the current CRC calculation
103  * and returns new result. If the newSeed is true, seed set and result are calculated
104  * from the seed new value (new CRC calculation).
105  *
106  * Implements : CRC_DRV_GetCrc32_Activity
107  *END**************************************************************************/
108 uint32_t CRC_DRV_GetCrc32(uint32_t instance,
109  uint32_t data,
110  bool newSeed,
111  uint32_t seed)
112 {
113  DEV_ASSERT(instance < CRC_INSTANCE_COUNT);
114  CRC_Type * base = s_crcBase[instance];
115 
116  /* If newSeed is true then write a seed to initial checksum */
117  if (newSeed)
118  {
119  /* Write a seed - initial checksum */
120  CRC_SetSeedReg(base, seed);
121  }
122 
123  /* Write 32-bit data */
124  CRC_SetDataReg(base, data);
125 
126  /* Result of the CRC calculation */
127  return CRC_GetCrcResult(base);
128 }

229 /*FUNCTION**********************************************************************
230  *
231  * Function Name : CRC_DRV_Configure
232  * Description : This function configures the CRC module from a user configuration structure.
233  *
234  * Implements : CRC_DRV_Configure_Activity
235  *END**************************************************************************/
236 status_t CRC_DRV_Configure(uint32_t instance,
237  const crc_user_config_t * userConfigPtr)
238 {
239  DEV_ASSERT(instance < CRC_INSTANCE_COUNT);
240  DEV_ASSERT(userConfigPtr != NULL);
241  CRC_Type * base = s_crcBase[instance];
242 
243 #if defined(FEATURE_CRC_DRIVER_SOFT_POLYNOMIAL)
244  /* Set CRC mode */
245  CRC_SetProtocolWidth(base, userConfigPtr->crcWidth);
246 #endif /* FEATURE_CRC_DRIVER_SOFT_POLYNOMIAL */
247  /* Set transposes options */
248  CRC_SetReadTranspose(base, userConfigPtr->readTranspose);
249  /* Set CRC polynomial */
250  CRC_SetPolyReg(base, userConfigPtr->polynomial);
251  /* Set writes transposes */
252  CRC_SetWriteTranspose(base, userConfigPtr->writeTranspose);
253  /* Sets complement or inversion checksum */
254  CRC_SetFXorMode(base, userConfigPtr->complementChecksum);
255  /* Write a seed - initial checksum */
256  CRC_SetSeedReg(base, userConfigPtr->seed);
257 
258  return STATUS_SUCCESS;
259 }

287  /* Get a seed - initial checksum */
